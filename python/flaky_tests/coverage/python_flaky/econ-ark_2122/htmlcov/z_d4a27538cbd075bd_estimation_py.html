16def minimizeNelderMead(objectiveFunction, parameter_guess, verbose=False, which_vars=None, **kwargs): 

17 ''' 

18 Minimizes the objective function using the Nelder-Mead simplex algorithm, 

19 starting from an initial parameter guess. 

20  

21 Parameters 

22 ---------- 

23 objectiveFunction : function 

24 The function to be minimized. It should take only a single argument, which 

25 should be a list representing the parameters to be estimated. 

26 parameter_guess : [float] 

27 A starting point for the Nelder-Mead algorithm, which must be a valid 

28 input for objectiveFunction. 

29 which_vars : np.array or None 

30 Array of booleans indicating which parameters should be estimated. When 

31 not provided, estimation is performed on all parameters. 

32 verbose : boolean 

33 A flag for the amount of output to print. 

34  

35 Returns 

36 ------- 

37 xopt : [float] 

38 The values that minimize objectiveFunction. 

39 ''' 

40 # Specify a temporary "modified objective function" that restricts parameters to be estimated 

41 if which_vars is None: 

42 which_vars = np.ones(len(parameter_guess),dtype=bool) 

43 def objectiveFunctionMod(params): 

44 params_full = np.copy(parameter_guess) 

45 params_full[which_vars] = params 

46 out = objectiveFunction(params_full) 

47 return out 

48 # convert parameter guess to np array to slice it with boolean array 

49 parameter_guess_mod = np.array(parameter_guess)[which_vars] 

50 

51 # Execute the minimization, starting from the given parameter guess 

52 t0 = time() # Time the process 

53 OUTPUT = fmin(objectiveFunctionMod, parameter_guess_mod, full_output=1, disp=verbose, **kwargs) 

54 t1 = time() 

55 

56 # Extract values from optimization output: 

57 xopt = OUTPUT[0] # Parameters that minimize function. 

58 fopt = OUTPUT[1] # Value of function at minimum: ``fopt = func(xopt)``. 

59 optiter = OUTPUT[2] # Number of iterations performed. 

60 funcalls = OUTPUT[3] # Number of function calls made. 

61 warnflag = OUTPUT[4] # warnflag : int 

62 # 1 : Maximum number of function evaluations made. 

63 # 2 : Maximum number of iterations reached. 

64 # Check that optimization succeeded: 

65 if warnflag != 0: 

66 warnings.warn("Minimization failed! xopt=" + str(xopt) + ', fopt=' + str(fopt) + 

67 ', optiter=' + str(optiter) +', funcalls=' + str(funcalls) + 

68 ', warnflag=' + str(warnflag)) 

69 xopt_full = np.copy(parameter_guess) 

70 xopt_full[which_vars] = xopt 

71 

72 # Display and return the results: 

73 if verbose: 

74 print("Time to estimate is " + str(t1-t0) + " seconds.") 

75 return xopt_full

78def minimizePowell(objectiveFunction, parameter_guess, verbose=False): 

79 '''

91 verbose : boolean 

92 A flag for the amount of output to print. 

93 

94 Returns 

95 ------- 

96 xopt : [float] 

97 The values that minimize objectiveFunction. 

98 ''' 

99 

100 # Execute the minimization, starting from the given parameter guess 

101 t0 = time() # Time the process 

102 OUTPUT = fmin_powell(objectiveFunction, parameter_guess, full_output=1, maxiter=1000, disp=verbose) 

103 t1 = time() 

104 

105 # Extract values from optimization output: 

106 xopt = OUTPUT[0] # Parameters that minimize function.

126def bootstrapSampleFromData(data, weights=None, seed=0): 

127 ''' 

128 Samples rows from the input array of data, generating a new data array with 

129 an equal number of rows (records). Rows are drawn with equal probability 

130 by default, but probabilities can be specified with weights (must sum to 1). 

131 

132 Parameters 

133 ---------- 

134 data : np.array 

135 An array of data, with each row representing a record. 

136 weights : np.array 

137 A weighting array with length equal to data.shape[0]. 

138 seed : int 

139 A seed for the random number generator. 

140 

141 Returns 

142 ------- 

143 new_data : np.array 

144 A resampled version of input data. 

145 ''' 

146 # Set up the random number generator 

147 RNG = np.random.RandomState(seed) 

148 N = data.shape[0] 

149 

150 # Set up weights 

151 if weights is not None: 

152 cutoffs = np.cumsum(weights) 

153 else: 

154 cutoffs = np.linspace(0, 1, N) 

155 

156 # Draw random indices 

157 indices = np.searchsorted(cutoffs, RNG.uniform(size=N)) 

158 

159 # Create a bootstrapped sample 

160 new_data = deepcopy(data[indices, ]) 

161 return new_data

164def parallelNelderMead( 

165 objFunc, 

166 guess, 

167 perturb=None, 

168 P=1, 

169 ftol=0.000001, 

170 xtol=0.00000001, 

171 maxiter=np.inf, 

172 maxeval=np.inf, 

173 r_param=1.0, 

174 e_param=1.0, 

175 c_param=0.5, 

176 s_param=0.5, 

177 maxthreads=None, 

178 name=None, 

179 resume=False, 

180 savefreq=None, 

181 verbose=1, 

182): 

183 """ 

184 

185 A parallel implementation of the Nelder-Mead minimization algorithm, as 

186 described in Lee and Wiswall. For long optimization procedures, it can 

187 save progress between iterations and resume later. 

188  

189 Parameters 

190 ---------- 

191 objFunc : function 

192 The objective function to be minimized. Takes a single 1D array as input. 

193 guess : np.array 

194 Initial starting point for the simplex, representing an input for objFunc. 

195 perturb : np.array 

196 Perturbation vector for the simplex, of the same length as an input to 

197 objFunc. If perturb[j] is non-zero, a simplex point will be created 

198 that perturbs the j-th element of guess by perturb[j]; if it is zero, 

199 then the j-th parameter of objFunc will not be optimized over. By 

200 default, perturb=None, indicating that all parameters should be optimized, 

201 with an initial perturbation of 0.1*guess. 

202 P : int 

203 Degree of parallelization: the number of vertices of the simplex to try 

204 to update on each iteration of the process. 

205 ftol : float 

206 Absolute tolerance of the objective function for convergence. If suc- 

207 cessive iterations return minimum function values that differ by less 

208 than ftol, the process terminates successfully. 

209 xtol : float 

210 Absolute tolerance of the input values for convergence. If the maximum 

211 distance between the current minimum point and the worst point in the 

212 simplex is less than xtol, then the process terminates successfully. 

213 maxiter : int 

214 Maximum number of Nelder-Mead iterations; reaching iters=maxiter is 

215 reported as an "unsuccessful" minimization. 

216 maxeval : int 

217 Maximum number of evaluations of objFunc (across all processes); reaching 

218 evals=maxeval is reported as an "unsuccessful" minimization. 

219 r_param: float 

220 Parameter indicating magnitude of the reflection point calculation. 

221 e_param: float 

222 Parameter indicating magnitude of the expansion point calculation. 

223 c_param: float 

224 Parameter indicating magnitude of the contraction point calculation. 

225 s_param: float 

226 Parameter indicating magnitude of the shrink calculation. 

227 maxthreads : int 

228 The maximum number of CPU cores that the optimization should use, 

229 regardless of the size of the problem. 

230 name : string 

231 A filename for (optionally) saving the progress of the Nelder-Mead search, 

232 and for resuming a previous search (when resume=True). Useful for long 

233 searches that could potentially be interrupted by computer down time. 

234 resume : boolean 

235 An indicator for whether the search should resume from earlier progress. 

236 When True, the process will load a progress file named in input name. 

237 savefreq : int 

238 When not None, search progress will be saved to name.txt every savefreq 

239 iterations, to be loaded later with resume=True). 

240 verbose : int 

241 Indicator for the verbosity of the optimization routine. Higher values 

242 generate more text output; verbose=0 produces no text output. 

243  

244 Returns 

245 ------- 

246 min_point : np.array 

247 The input that minimizes objFunc, as found by the minimization. 

248 fmin : float 

249 The minimum of objFunc; fmin = objFunc(min_point). 

250 """ 

251 # If this is a resumed search, load the data 

252 if resume: 

253 simplex, fvals, iters, evals = loadNelderMeadData(name) 

254 dim_count = fvals.size - 1 

255 N = dim_count + 1 # Number of points in simplex 

256 K = simplex.shape[1] # Total number of parameters 

257 

258 # Otherwise, construct the initial simplex and array of function values 

259 else: 

260 if perturb is None: # Default: perturb each parameter by 10% 

261 perturb = 0.1 * guess 

262 guess[guess == 0] = 0.1 

263 

264 params_to_opt = np.where(perturb != 0)[ 

265 0 

266 ] # Indices of which parameters to optimize 

267 dim_count = params_to_opt.size # Number of parameters to search over 

268 N = dim_count + 1 # Number of points in simplex 

269 K = guess.size # Total number of parameters 

270 simplex = np.tile(guess, (N, 1)) 

271 for j in range( 

272 dim_count 

273 ): # Perturb each parameter to optimize by the specified distance 

274 simplex[j + 1, params_to_opt[j]] = ( 

275 simplex[j + 1, params_to_opt[j]] + perturb[params_to_opt[j]] 

276 ) 

277 

278 # Initialize iteration and evaluation counts, plus a 1D array of function values 

279 fvals = np.zeros(dim_count + 1) + np.nan 

280 

281 iters = 0 

282 evals = 0 

283 

284 # Make sure degree of parallelization is not illegal 

285 if P > N - 1: 

286 print( 

287 "Requested degree of simplex parallelization is " 

288 + str(P) 

289 + ", but dimension of optimization problem is only " 

290 + str(N - 1) 

291 + "." 

292 ) 

293 print("Degree of parallelization has been reduced to " + str(N - 1) + ".") 

294 P = N - 1 

295 

296 # Create the pool of worker processes 

297 cpu_cores = multiprocessing.cpu_count() # Total number of available CPU cores 

298 cores_to_use = min(cpu_cores, dim_count) 

299 if maxthreads is not None: # Cap the number of cores if desired 

300 cores_to_use = min(cores_to_use, maxthreads) 

301 parallel = Parallel(n_jobs=cores_to_use) 

302 

303 # Begin a new Nelder-Mead search 

304 if not resume: 

305 temp_simplex = list(simplex) # Evaluate the initial simplex 

306 fvals = np.array(parallel(delayed(objFunc)(params) for params in temp_simplex)) 

307 evals += N 

308 # Reorder the initial simplex 

309 order = np.argsort(fvals) 

310 fvals = fvals[order] 

311 simplex = simplex[order, :] 

312 fmin = fvals[0] 

313 f_dist = np.abs(fmin - fvals[-1]) 

314 x_dist = np.max( 

315 np.sqrt(np.sum((simplex - np.tile(simplex[0, :], (N, 1))) ** 2.0, axis=1)) 

316 ) 

317 if verbose > 0: 

318 print( 

319 "Evaluated the initial simplex: fmin=" 

320 + str(fmin) 

321 + ", f_dist=" 

322 + str(f_dist) 

323 + ", x_dist=" 

324 + str(x_dist) 

325 ) 

326 if savefreq is not None: 

327 saveNelderMeadData(name, simplex, fvals, iters, evals) 

328 if verbose > 0: 

329 print("Saved search progress in " + name + ".txt") 

330 else: # Resume an existing search that was cut short 

331 if verbose > 0: 

332 print( 

333 "Resuming search after " 

334 + str(iters) 

335 + " iterations and " 

336 + str(evals) 

337 + " function evaluations." 

338 ) 

339 

340 # Initialize some inputs for the multithreader 

341 j_list = range(N - P, N) 

342 opt_params = [r_param, c_param, e_param] 

343 

344 # Run the Nelder-Mead algorithm until a terminal condition is met 

345 go = True 

346 while go: 

347 t_start = time() 

348 iters += 1 

349 if verbose > 0: 

350 print("Beginning iteration #" + str(iters) + " now.") 

351 

352 # Update the P worst points of the simplex 

353 output = parallel( 

354 delayed(parallelNelderMeadWorker)(objFunc, simplex, fvals, j, P, opt_params) 

355 for j in j_list 

356 ) 

357 new_subsimplex = np.zeros((P, K)) + np.nan 

358 new_vals = np.zeros(P) + np.nan 

359 new_evals = 0 

360 for i in range(P): 

361 new_subsimplex[i, :] = output[i][0] 

362 new_vals[i] = output[i][1] 

363 new_evals += output[i][2] 

364 evals += new_evals 

365 

366 # Check whether any updates actually happened 

367 old_subsimplex = simplex[(N - P) : N, :] 

368 if np.max(np.abs(new_subsimplex - old_subsimplex)) == 0: 

369 if verbose > 0: 

370 print("Updated the simplex, but must perform a shrink step.") 

371 # If every attempted update was unsuccessful, must shrink the simplex 

372 simplex = ( 

373 s_param * np.tile(simplex[0, :], (N, 1)) + (1.0 - s_param) * simplex 

374 ) 

375 temp_simplex = list(simplex[1:N, :]) 

376 fvals = np.array( 

377 [fvals[0]] 

378 + parallel(delayed(objFunc)(params) for params in temp_simplex) 

379 ) 

380 new_evals += N - 1 

381 evals += N - 1 

382 else: 

383 if verbose > 0: 

384 print("Updated the simplex successfully.") 

385 # Otherwise, update the simplex with the new results 

386 simplex[(N - P) : N, :] = new_subsimplex 

387 fvals[(N - P) : N] = new_vals 

388 

389 # Reorder the simplex from best to worst 

390 order = np.argsort(fvals) 

391 fvals = fvals[order] 

392 simplex = simplex[order, :] 

393 fmin = fvals[0] 

394 f_dist = np.abs(fmin - fvals[-1]) 

395 x_dist = np.max( 

396 np.sqrt(np.sum((simplex - np.tile(simplex[0, :], (N, 1))) ** 2.0, axis=1)) 

397 ) 

398 t_end = time() 

399 if verbose > 0: 

400 t_iter = t_end - t_start 

401 print( 

402 "Finished iteration #" 

403 + str(iters) 

404 + " with " 

405 + str(new_evals) 

406 + " evaluations (" 

407 + str(evals) 

408 + " cumulative) in " 

409 + str(t_iter) 

410 + " seconds." 

411 ) 

412 print( 

413 "Simplex status: fmin=" 

414 + str(fmin) 

415 + ", f_dist=" 

416 + str(f_dist) 

417 + ", x_dist=" 

418 + str(x_dist) 

419 ) 

420 

421 # Check for terminal conditions 

422 if iters >= maxiter: 

423 go = False 

424 print("Maximum iterations reached, terminating unsuccessfully.") 

425 if evals >= maxeval: 

426 go = False 

427 print("Maximum evaluations reached, terminating unsuccessfully.") 

428 if f_dist < ftol: 

429 go = False 

430 print("Function tolerance reached, terminating successfully.") 

431 if x_dist < xtol: 

432 go = False 

433 print("Parameter tolerance reached, terminating successfully.") 

434 

435 # Save the progress of the estimation if desired 

436 if savefreq is not None: 

437 if (iters % savefreq) == 0: 

438 saveNelderMeadData(name, simplex, fvals, iters, evals) 

439 if verbose > 0: 

440 print("Saved search progress in " + name + ".txt") 

441 

442 # Return the results 

443 xopt = simplex[0, :] 

444 return xopt, fmin

447def saveNelderMeadData(name, simplex, fvals, iters, evals): 

448 """ 

449 Stores the progress of a parallel Nelder-Mead search in a text file so that 

450 it can be resumed later (after manual termination or a crash). 

451  

452 Parameters 

453 ---------- 

454 name : string 

455 Name of the txt file in which to store search progress. 

456 simplex : np.array 

457 The current state of the simplex of parameter guesses. 

458 fvals : np.array 

459 The objective function value at each row of simplex. 

460 iters : int 

461 The number of completed Nelder-Mead iterations. 

462 evals : int 

463 The cumulative number of function evaluations in the search process. 

464  

465 Returns 

466 ------- 

467 None 

468 """ 

469 N = simplex.shape[0] # Number of points in simplex 

470 K = simplex.shape[1] # Total number of parameters 

471 

472 with open(name + ".txt", "w") as f: 

473 my_writer = csv.writer(f, delimiter=",") 

474 my_writer.writerow(simplex.shape) 

475 my_writer.writerow([iters, evals]) 

476 for n in range(N): 

477 my_writer.writerow(simplex[n, :]) 

478 my_writer.writerow(fvals)

481def loadNelderMeadData(name): 

482 """ 

483 Reads the progress of a parallel Nelder-Mead search from a text file, as 

484 created by saveNelderMeadData(). 

485  

486 Parameters 

487 ---------- 

488 name : string 

489 Name of the txt file from which to read search progress. 

490  

491 Returns 

492 ------- 

493 simplex : np.array 

494 The current state of the simplex of parameter guesses. 

495 fvals : np.array 

496 The objective function value at each row of simplex. 

497 iters : int 

498 The number of completed Nelder-Mead iterations. 

499 evals : int 

500 The cumulative number of function evaluations in the search process. 

501 """ 

502 # Open the Nelder-Mead progress file 

503 with open(name + ".txt", "r") as f: 

504 my_reader = csv.reader(f, delimiter=",") 

505 

506 # Get the shape of the simplex and initialize it 

507 my_shape_txt = my_reader.next() 

508 N = int(my_shape_txt[0]) 

509 K = int(my_shape_txt[1]) 

510 simplex = np.zeros((N, K)) + np.nan 

511 

512 # Get number of iterations and cumulative evaluations from the next line 

513 my_nums_txt = my_reader.next() 

514 iters = int(my_nums_txt[0]) 

515 evals = int(my_nums_txt[1]) 

516 

517 # Read one line per point of the simplex 

518 for n in range(N): 

519 simplex[n, :] = np.array(my_reader.next(), dtype=float) 

520 

521 # Read the final line to get function values 

522 fvals = np.array(my_reader.next(), dtype=float) 

523 

524 return simplex, fvals, iters, evals

527def parallelNelderMeadWorker(objFunc, simplex, f_vals, j, P, opt_params): 

528 """ 

529 A worker process for the parallel Nelder-Mead algorithm. Updates one point 

530 in the simplex, returning its function value as well. Should basically 

531 never be called directly, only by parallelNelderMead(). 

532  

533 Parameters 

534 ---------- 

535 objFunc : function 

536 The function to be minimized; takes a single 1D array as input. 

537 simplex : numpy.array 

538 The current simplex for minimization; simplex[k,:] is an input for objFunc. 

539 f_vals : numpy.array 

540 The values of the objective function at each point of the simplex: 

541 f_vals[k] = objFunc(simplex[k,:]) 

542 j : int 

543 Index of the point in the simplex to update: simplex[j,:] 

544 P : int 

545 Degree of parallelization of the algorithm. 

546 opt_params : numpy.array 

547 Three element array with parameters for reflection, contraction, expansion. 

548  

549 Returns 

550 ------- 

551 new_point : numpy.array 

552 An updated point for the simplex; might be the same as simplex[j,:]. 

553 new_val : float 

554 The value of the objective function at the new point: objFunc(new_point). 

555 evals : int 

556 Number of evaluations of objFunc by this worker. 

557 """ 

558 # Unpack the input parameters 

559 alpha = opt_params[0] # reflection parameter 

560 beta = opt_params[1] # contraction parameter 

561 gamma = opt_params[2] # expansion parameter 

562 my_point = simplex[j, :] # vertex to update 

563 my_val = f_vals[j] # value at the vertex to update 

564 best_val = f_vals[0] # best value in the vertex 

565 next_val = f_vals[j - 1] # next best point in the simplex 

566 evals = 0 

567 

568 # Calculate the centroid of the "good" simplex points 

569 N = simplex.shape[0] # number of points in simplex 

570 centroid = np.mean(simplex[0 : (N - P), :], axis=0) 

571 

572 # Calculate the reflection point and its function value 

573 r_point = centroid + alpha * (centroid - my_point) 

574 r_val = objFunc(r_point) 

575 evals += 1 

576 

577 # Case 1: the reflection point is better than best point 

578 if r_val < best_val: 

579 e_point = r_point + gamma * (r_point - centroid) 

580 e_val = objFunc(e_point) # Calculate expansion point 

581 evals += 1 

582 if e_val < r_val: 

583 new_point = e_point 

584 new_val = e_val 

585 else: 

586 new_point = r_point 

587 new_val = r_val 

588 # Case 2: the reflection point is better than the next best point 

589 elif r_val < next_val: 

590 new_point = r_point # Report reflection point 

591 new_val = r_val 

592 # Case 3: the reflection point is worse than the next best point 

593 else: 

594 if r_val < my_val: 

595 temp_point = r_point # Check whether reflection or original point 

596 temp_val = r_val # is better and use it temporarily 

597 else: 

598 temp_point = my_point 

599 temp_val = my_val 

600 c_point = temp_point + beta * (centroid - temp_point) 

601 c_val = objFunc(c_point) # Calculate contraction point 

602 evals += 1 

603 if c_val < temp_val: 

604 new_point = c_point 

605 new_val = c_val # Check whether the contraction point is better 

606 else: 

607 new_point = temp_point 

608 new_val = temp_val 

609 

610 # Return the outputs 

611 return new_point, new_val, evals
