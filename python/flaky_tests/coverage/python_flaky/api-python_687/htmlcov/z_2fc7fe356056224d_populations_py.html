32def _flatten_results(result, default_value=None): 

33 """ Formats results to map to a single value or default value if empty. """ 

34 for k in list(result): 

35 v = result[k] 

36 if len(v) > 1: 

37 raise ValueError( 

38 'Expected one result, but more returned for "{}": {}'.format(k, v)) 

39 if len(v) == 1: 

40 result[k] = v[0] 

41 else: 

42 if default_value is not None: 

43 result[k] = default_value 

44 else: 

45 del result[k] 

46 return result

49def get_populations(dcids, population_type, constraining_properties={}): 

50 """ Returns :obj:`StatisticalPopulation`'s located at the given :code:`dcids`. 

51 

52 Args: 

53 dcids (:obj:`iterable` of :obj:`str`): Dcids 

54 identifying :obj:`Place`'s of populations to query for. These dcids are 

55 treated as the property value associated with returned :obj:`Population`'s 

56 by the property 

57 `location <https://browser.datacommons.org/kg?dcid=location>`_ 

58 population_type (:obj:`str`): The population type of the 

59 :obj:`StatisticalPopulation` 

60 constraining_properties (:obj:`map` from :obj:`str` to :obj:`str`, optional): 

61 A map from constraining property to the value that the 

62 :obj:`StatisticalPopulation` should be constrained by. 

63 

64 Returns: 

65 The returned :obj:`StatisticalPopulation` are formatted as a :obj:`dict` from a given 

66 dcid to the unique :obj:`StatisticalPopulation` located at the dcid as 

67 specified by the `population_type` and `constraining_properties` *if such 

68 exists*. A given dcid will *NOT* be a member of the :obj:`dict` if such 

69 a population does not exist. 

70 

71 Raises: 

72 ValueError: If the payload returned by the Data Commons REST API is 

73 malformed. 

74 

75 Examples: 

76 We would like to get 

77 

78 - The `population of employed persons in California <https://browser.datacommons.org/kg?dcid=dc/p/x6t44d8jd95rd>`_ 

79 - The `population of employed persons in Kentucky <https://browser.datacommons.org/kg?dcid=dc/p/fs929fynprzs>`_ 

80 - The `population of employed persons in Maryland <https://browser.datacommons.org/kg?dcid=dc/p/lr52m1yr46r44>`_. 

81 

82 These populations are specified as having a 

83 `population_type` as :obj:`Person` and the `constraining_properties` 

84 as `employment <https://browser.datacommons.org/kg?dcid=employment>`_ 

85 = BLS_Employed 

86 

87 With a :obj:`list` of dcids for our states, we can get the populations we 

88 want as follows. 

89 

90 >>> dcids = ["geoId/06", "geoId/21", "geoId/24"] 

91 >>> pvs = {'employment': 'BLS_Employed'} 

92 >>> dc.get_populations(dcids, 'Person', constraining_properties=pvs) 

93 { 

94 "geoId/06": "dc/p/x6t44d8jd95rd", 

95 "geoId/21": "dc/p/fs929fynprzs", 

96 "geoId/24": "dc/p/lr52m1yr46r44" 

97 } 

98 """ 

99 # Convert the dcids field and format the request to GetPopulations 

100 dcids = filter(lambda v: v==v, dcids) # Filter out NaN values 

101 dcids = list(dcids) 

102 pv = [{'property': k, 'value': v} for k, v in constraining_properties.items()] 

103 url = utils._API_ROOT + utils._API_ENDPOINTS['get_populations'] 

104 payload = utils._send_request(url, req_json={ 

105 'dcids': dcids, 

106 'population_type': population_type, 

107 'pvs': pv, 

108 }) 

109 

110 # Create the results and format it appropriately 

111 result = utils._format_expand_payload( 

112 payload, 'population', must_exist=dcids) 

113 

114 # Drop empty results while flattening 

115 return _flatten_results(result)

118def get_observations(dcids, 

119 measured_property, 

120 stats_type, 

121 observation_date, 

122 observation_period=None, 

123 measurement_method=None): 

124 """ Returns values of :obj:`Observation`'s observing the given :code:`dcids`. 

125 

126 Args: 

127 dcids (:obj:`iterable` of :obj:`str`): Dcids 

128 identifying nodes that returning :obj:`Observation`'s observe. These dcids 

129 are treated as the property value associated with returned 

130 :obj:`Observation`'s by the property 

131 `observedNode <https://browser.datacommons.org/kg?dcid=observedNode>`_ 

132 measured_property (:obj:`str`): The measured property. 

133 stats_type (:obj:`str`): The statistical type for the observation. 

134 observation_date (:obj:`str`): The associated observation date in ISO8601 

135 format. 

136 observation_period (:obj:`str`, optional): An optional parameter specifying 

137 the observation period. 

138 measurement_method (:obj:`str`, optional): An optional parameter specifying 

139 the measurement method. 

140 

141 Raises: 

142 ValueError: If the payload returned by the Data Commons REST API is 

143 malformed. 

144 

145 Returns: 

146 When :code:`dcids` is an instance of :obj:`list`, the returned 

147 :obj:`Observation`'s are formatted as a :obj:`dict` from a given dcid to the 

148 unique :obj:`Observation` observing the dcid where the observation is 

149 specified by what is given in the other parameters *if such exists*. A given 

150 dcid will *NOT* be a member of the :obj:`dict` if such an observation does 

151 not exist. 

152 

153 Examples: 

154 We would like to get the following for December, 2018: 

155 

156 - The `total count of employed persons in California <https://browser.datacommons.org/kg?dcid=dc/o/wetnm9026gf73>`_ 

157 - The `total count of employed persons in Kentucky <https://browser.datacommons.org/kg?dcid=dc/o/4nklvdnkfq835>`_ 

158 - The `total count of employed persons in Maryland <https://browser.datacommons.org/kg?dcid=dc/o/nkntbc4vpshn9>`_. 

159 

160 The observations we want are observations of the populations representing 

161 employed individuals in each state (to get these, see 

162 :any:module-datacommons.populations.get_populations). With a list of these 

163 population dcids, we can get the observations like so. 

164 

165 >>> dcids = [ 

166 ... "dc/p/x6t44d8jd95rd", # Employed individuals in California 

167 ... "dc/p/fs929fynprzs", # Employed individuals in Kentucky 

168 ... "dc/p/lr52m1yr46r44" # Employed individuals in Maryland 

169 ... ] 

170 >>> get_observations(dcids, 'count', 'measuredValue', '2018-12', 

171 ... observation_period='P1M', 

172 ... measurement_method='BLSSeasonallyAdjusted' 

173 ... ) 

174 { 

175 "dc/p/x6t44d8jd95rd": 18704962.0, 

176 "dc/p/fs929fynprzs": 1973955.0, 

177 "dc/p/lr52m1yr46r44": 3075662.0 

178 } 

179 """ 

180 dcids = filter(lambda v: v==v, dcids) # Filter out NaN values 

181 dcids = list(dcids) 

182 req_json = { 

183 'dcids': dcids, 

184 'measured_property': measured_property, 

185 'stats_type': stats_type, 

186 'observation_date': observation_date, 

187 } 

188 if observation_period: 

189 req_json['observation_period'] = observation_period 

190 if measurement_method: 

191 req_json['measurement_method'] = measurement_method 

192 

193 # Issue the request to GetObservation 

194 url = utils._API_ROOT + utils._API_ENDPOINTS['get_observations'] 

195 payload = utils._send_request(url, req_json=req_json) 

196 

197 # Create the results and format it appropriately 

198 result = utils._format_expand_payload( 

199 payload, 'observation', must_exist=dcids) 

200 

201 # Drop empty results by calling _flatten_results without default_value, then 

202 # coerce the type to float if possible. 

203 typed_results = {} 

204 for k, v in _flatten_results(result).items(): 

205 try: 

206 typed_results[k] = float(v) 

207 except ValueError: 

208 typed_results[k] = v 

209 return typed_results

212def get_pop_obs(dcid): 

213 """ Returns all :obj:`StatisticalPopulation` and :obj:`Observation` \ 

214 of a :obj:`Thing`. 

215 

216 Args: 

217 dcid (:obj:`str`): Dcid of the thing. 

218 

219 Returns: 

220 A :obj:`dict` of :obj:`StatisticalPopulation` and :obj:`Observation` that 

221 are associated to the thing identified by the given :code:`dcid`. The given 

222 dcid is linked to the returned :obj:`StatisticalPopulation`, 

223 which are the :obj:`observedNode` of the returned :obj:`Observation`. 

224 See example below for more detail about how the returned :obj:`dict` is 

225 structured. 

226 

227 Raises: 

228 ValueError: If the payload returned by the Data Commons REST API is 

229 malformed. 

230 

231 Examples: 

232 We would like to get all :obj:`StatisticalPopulation` and 

233 :obj:`Observations` of 

234 `Santa Clara <https://browser.datacommons.org/kg?dcid=geoId/06085>`_. 

235 

236 >>> get_pop_obs("geoId/06085") 

237 { 

238 'name': 'Santa Clara', 

239 'placeType': 'County', 

240 'populations': { 

241 'dc/p/zzlmxxtp1el87': { 

242 'popType': 'Household', 

243 'numConstraints': 3, 

244 'propertyValues': { 

245 'householderAge': 'Years45To64', 

246 'householderRace': 'USC_AsianAlone', 

247 'income': 'USDollar35000To39999' 

248 }, 

249 'observations': [ 

250 { 

251 'marginOfError': 274, 

252 'measuredProp': 'count', 

253 'measuredValue': 1352, 

254 'measurementMethod': 'CensusACS5yrSurvey', 

255 'observationDate': '2017' 

256 }, 

257 { 

258 'marginOfError': 226, 

259 'measuredProp': 'count', 

260 'measuredValue': 1388, 

261 'measurementMethod': 'CensusACS5yrSurvey', 

262 'observationDate': '2013' 

263 } 

264 ], 

265 }, 

266 }, 

267 'observations': [ 

268 { 

269 'meanValue': 4.1583, 

270 'measuredProp': 'particulateMatter25', 

271 'measurementMethod': 'CDCHealthTracking', 

272 'observationDate': '2014-04-04', 

273 'observedNode': 'geoId/06085' 

274 }, 

275 { 

276 'meanValue': 9.4461, 

277 'measuredProp': 'particulateMatter25', 

278 'measurementMethod': 'CDCHealthTracking', 

279 'observationDate': '2014-03-20', 

280 'observedNode': 'geoId/06085' 

281 } 

282 ] 

283 } 

284 

285 Notice that the return value is a multi-level :obj:`dict`. The top level 

286 contains the following keys. 

287 

288 - :code:`name` and :code:`placeType` provides the name and type of the 

289 :obj:`Place` identified by the given :code:`dcid`. 

290 - :code:`populations` maps to a :obj:`dict` containing all 

291 :obj:`StatisticalPopulation` that have the given :code:`dcid` as its 

292 :obj:`location`. 

293 - :code:`observations` maps to a :obj:`list` containing all 

294 :obj:`Observation` that have the given :code:`dcid` as its 

295 :obj:`observedNode`. 

296 

297 The :code:`populations` dictionary is keyed by the dcid of each 

298 :obj:`StatisticalPopulation`. The mapped dictionary contains the following 

299 keys. 

300 

301 - :code:`popType` which gives the population type of the 

302 :obj:`StatisticalPopulation` identified by the key. 

303 - :code:`numConstraints` which gives the number of constraining properties 

304 defined for the identified :obj:`StatisticalPopulation`. 

305 - :code:`propertyValues` which gives a :obj:`dict` mapping a constraining 

306 property to its value for the identified :obj:`StatisticalPopulation`. 

307 - :code:`observations` which gives a list of all :obj:`Observation`'s that 

308 have the identified :obj:`StatisticalPopulation` as their 

309 :obj:`observedNode`. 

310 

311 Each :obj:`Observation` is represented by a :code:`dict` that have the keys: 

312 

313 - :code:`measuredProp`: The property measured by the :obj:`Observation`. 

314 - :code:`observationDate`: The date when the :obj:`Observation` was made. 

315 - :code:`observationPeriod` (optional): The period over which the 

316 :obj:`Observation` was made. 

317 - :code:`measurementMethod` (optional): A field providing additional 

318 information on how the :obj:`Observation` was collected. 

319 - Additional fields that denote values measured by the :obj:`Observation`. 

320 These may include the following: :code:`measuredValue`, :code:`meanValue`, 

321 :code:`medianValue`, :code:`maxValue`, :code:`minValue`, :code:`sumValue`, 

322 :code:`marginOfError`, :code:`stdError`, :code:`meanStdError`, and others. 

323 """ 

324 url = utils._API_ROOT + utils._API_ENDPOINTS['get_pop_obs'] + '?dcid={}'.format(dcid) 

325 return utils._send_request(url, compress=True, post=False)

327def get_place_obs( 

328 place_type, observation_date, population_type, constraining_properties={}): 

329 """ Returns all :obj:`Observation`'s for all places given the place type, 

330 observation date and the :obj:`StatisticalPopulation` constraints. 

331 

332 Args: 

333 place_type (:obj:`str`): The type of places to query 

334 :obj:`StatisticalPopulation`'s and :obj:`Observation`'s for. 

335 observation_date (:obj:`str`): The observation date in ISO-8601 format. 

336 population_type (:obj:`str`): The population type of the 

337 :obj:`StatisticalPopulation` 

338 constraining_properties (:obj:`map` from :obj:`str` to :obj:`str`, optional): 

339 A map from constraining property to the value that the 

340 :obj:`StatisticalPopulation` should be constrained by. 

341 

342 Returns: 

343 A list of dictionaries, with each dictionary containng *all* 

344 :obj:`Observation`'s of a place that conform to the :obj:`StatisticalPopulation` 

345 constraints. See examples for more details on how the format of the 

346 return value is structured. 

347 

348 Raises: 

349 ValueError: If the payload is malformed. 

350 

351 Examples: 

352 We would like to get all :obj:`StatisticalPopulation` and 

353 :obj:`Observations` for all places of type :obj:`City` in year 2017 where 

354 the populations have a population type of :obj:`Person` is specified by the 

355 following constraining properties. 

356 

357 - Persons should have `age <https://browser.datacommons.org/kg?dcid=age>`_ 

358 with value `Years5To17 <https://browser.datacommons.org/kg?dcid=Years5To17>`_ 

359 - Persons should have `placeOfBirth <https://browser.datacommons.org/kg?dcid=placeOfBirth>`_ 

360 with value BornInOtherStateInTheUnitedStates. 

361 

362 >>> props = { 

363 ... 'age': 'Years5To17', 

364 ... 'placeOfBirth': 'BornInOtherStateInTheUnitedStates' 

365 ... } 

366 >>> get_place_obs('City', '2017', Person', constraining_properties=props) 

367 [ 

368 { 

369 'name': 'Marcus Hook borough', 

370 'place': 'geoId/4247344', 

371 'populations': { 

372 'dc/p/pq6frs32sfvk': { 

373 'observations': [ 

374 { 

375 'marginOfError': 39, 

376 'measuredProp': 'count', 

377 'measuredValue': 67, 

378 'type': 'Observation' 

379 }, 

380 # More observations... 

381 ], 

382 } 

383 } 

384 }, 

385 # Entries for more cities... 

386 ] 

387 

388 The value returned by :code:`get_place_obs` is a :obj:`list` of 

389 :obj:`dict`'s. Each dictionary corresponds to a :obj:`StatisticalPopulation` 

390 matching the given :code:`population_type` and 

391 :code:`constraining_properties` for a single place of the given 

392 :code:`place_type`. The dictionary contains the following keys. 

393 

394 - :code:`name`: The name of the place being described. 

395 - :code:`place`: The dcid associated with the place being described. 

396 - :code:`populations`: A :obj:`dict` mapping :code:`StatisticalPopulation` 

397 dcids to a a :obj:`dict` with a list of :code:`observations`. 

398 

399 Each :obj:`Observation` is represented by a :obj:`dict` with the following 

400 keys. 

401 - :code:`measuredProp`: The property measured by the :obj:`Observation`. 

402 - :code:`measurementMethod` (optional): A field identifying how the 

403 :obj:`Observation` was made 

404 - Additional fields that denote values measured by the :obj:`Observation`. 

405 These may include the following: :code:`measuredValue`, :code:`meanValue`, 

406 :code:`medianValue`, :code:`maxValue`, :code:`minValue`, :code:`sumValue`, 

407 :code:`marginOfError`, :code:`stdError`, :code:`meanStdError`, and others. 

408 """ 

409 # Create the json payload and send it to the REST API. 

410 pv = [{'property': k, 'value': v} for k, v in constraining_properties.items()] 

411 url = utils._API_ROOT + utils._API_ENDPOINTS['get_place_obs'] 

412 payload = utils._send_request(url, req_json={ 

413 'place_type': place_type, 

414 'observation_date': observation_date, 

415 'population_type': population_type, 

416 'pvs': pv, 

417 }, compress=True) 

418 return payload['places']
