12def add_edge_speeds(G, hwy_speeds=None, fallback=None, precision=1): 

13 """ 

14 Add edge speeds (km per hour) to graph as new `speed_kph` edge attributes. 

15 

16 Imputes free-flow travel speeds for all edges based on mean `maxspeed` 

17 value of edges, per highway type. For highway types in graph that have no 

18 `maxspeed` value on any edge, function assigns the mean of all `maxspeed` 

19 values in graph. 

20 

21 This mean-imputation can obviously be imprecise, and the caller can 

22 override it by passing in `hwy_speeds` and/or `fallback` arguments that 

23 correspond to local speed limit standards. 

24 

25 If edge `maxspeed` attribute has "mph" in it, value will automatically be 

26 converted from miles per hour to km per hour. Any other speed units should 

27 be manually converted to km per hour prior to running this function, 

28 otherwise there could be unexpected results. If "mph" does not appear in 

29 the edge's maxspeed attribute string, then function assumes kph, per OSM 

30 guidelines: https://wiki.openstreetmap.org/wiki/Map_Features/Units 

31 

32 Parameters 

33 ---------- 

34 G : networkx.MultiDiGraph 

35 input graph 

36 hwy_speeds : dict 

37 dict keys = OSM highway types and values = typical speeds (km per 

38 hour) to assign to edges of that highway type for any edges missing 

39 speed data. Any edges with highway type not in `hwy_speeds` will be 

40 assigned the mean preexisting speed value of all edges of that highway 

41 type. 

42 fallback : numeric 

43 default speed value (km per hour) to assign to edges whose highway 

44 type did not appear in `hwy_speeds` and had no preexisting speed 

45 values on any edge 

46 precision : int 

47 decimal precision to round speed_kph 

48 

49 Returns 

50 ------- 

51 G : networkx.MultiDiGraph 

52 graph with speed_kph attributes on all edges 

53 """ 

54 if fallback is None: 

55 fallback = np.nan 

56 

57 edges = utils_graph.graph_to_gdfs(G, nodes=False, fill_edge_geometry=False) 

58 

59 # collapse any highway lists (can happen during graph simplification) 

60 # into string values simply by keeping just the first element of the list 

61 edges["highway"] = edges["highway"].map(lambda x: x[0] if isinstance(x, list) else x) 

62 

63 if "maxspeed" in edges.columns: 

64 # collapse any maxspeed lists (can happen during graph simplification) 

65 # into a single value 

66 edges["maxspeed"] = edges["maxspeed"].map(_collapse_multiple_maxspeed_values) 

67 

68 # create speed_kph by cleaning maxspeed strings and converting mph to 

69 # kph if necessary 

70 edges["speed_kph"] = edges["maxspeed"].astype(str).map(_clean_maxspeed).astype(float) 

71 else: 

72 # if no edges in graph had a maxspeed attribute 

73 edges["speed_kph"] = None 

74 

75 # if user provided hwy_speeds, use them as default values, otherwise 

76 # initialize an empty series to populate with values 

77 if hwy_speeds is None: 

78 hwy_speed_avg = pd.Series(dtype=float) 

79 else: 

80 hwy_speed_avg = pd.Series(hwy_speeds).dropna() 

81 

82 # for each highway type that caller did not provide in hwy_speeds, impute 

83 # speed of type by taking the mean of the preexisting speed values of that 

84 # highway type 

85 for hwy, group in edges.groupby("highway"): 

86 if hwy not in hwy_speed_avg: 

87 hwy_speed_avg.loc[hwy] = group["speed_kph"].mean() 

88 

89 # if any highway types had no preexisting speed values, impute their speed 

90 # with fallback value provided by caller. if fallback=np.nan, impute speed 

91 # as the mean speed of all highway types that did have preexisting values 

92 hwy_speed_avg = hwy_speed_avg.fillna(fallback).fillna(hwy_speed_avg.mean()) 

93 

94 # for each edge missing speed data, assign it the imputed value for its 

95 # highway type 

96 speed_kph = ( 

97 edges[["highway", "speed_kph"]].set_index("highway").iloc[:, 0].fillna(hwy_speed_avg) 

98 ) 

99 

100 # all speeds will be null if edges had no preexisting maxspeed data and 

101 # caller did not pass in hwy_speeds or fallback arguments 

102 if pd.isnull(speed_kph).all(): 

103 raise ValueError( 

104 ( 

105 "this graph's edges have no preexisting `maxspeed` " 

106 "attribute values so you must pass `hwy_speeds` or " 

107 "`fallback` arguments." 

108 ) 

109 ) 

110 

111 # add speed kph attribute to graph edges 

112 edges["speed_kph"] = speed_kph.round(precision).values 

113 edge_speed_kph = edges[["u", "v", "key", "speed_kph"]].set_index(["u", "v", "key"]).iloc[:, 0] 

114 nx.set_edge_attributes(G, values=edge_speed_kph, name="speed_kph") 

115 

116 return G

119def add_edge_travel_times(G, precision=1): 

120 """ 

121 Add edge travel time (seconds) to graph as new `travel_time` edge attributes. 

122 

123 Calculates free-flow travel time along each edge, based on `length` and 

124 `speed_kph` attributes. Note: run `add_edge_speeds` first to generate the 

125 `speed_kph` attribute. All edges must have `length` and `speed_kph` 

126 attributes and all their values must be non-null. 

127 

128 Parameters 

129 ---------- 

130 G : networkx.MultiDiGraph 

131 input graph 

132 precision : int 

133 decimal precision to round travel_time 

134 

135 Returns 

136 ------- 

137 G : networkx.MultiDiGraph 

138 graph with travel_time attributes on all edges 

139 """ 

140 edges = utils_graph.graph_to_gdfs(G, nodes=False) 

141 

142 # verify edge length and speed_kph attributes exist and contain no nulls 

143 if not ("length" in edges.columns and "speed_kph" in edges.columns): 

144 raise KeyError("all edges must have `length` and `speed_kph` attributes.") 

145 else: 

146 if pd.isnull(edges["length"]).any() or pd.isnull(edges["speed_kph"]).any(): 

147 raise ValueError("edge `length` and `speed_kph` values must be non-null.") 

148 

149 # convert distance km to meters, and speed km per hour to km per second 

150 distance_km = edges["length"] / 1000 

151 speed_km_sec = edges["speed_kph"] / (60 * 60) 

152 

153 # calculate edge travel time in seconds 

154 travel_time = distance_km / speed_km_sec 

155 

156 # add travel time attribute to graph edges 

157 edges["travel_time"] = travel_time.round(precision).values 

158 edge_times = edges[["u", "v", "key", "travel_time"]].set_index(["u", "v", "key"]).iloc[:, 0] 

159 nx.set_edge_attributes(G, values=edge_times, name="travel_time") 

160 

161 return G

164def _clean_maxspeed(value, convert_mph=True): 

165 """ 

166 Clean a maxspeed string and convert mph to kph if necessary. 

167 

168 Parameters 

169 ---------- 

170 value : string 

171 an OSM way maxspeed value 

172 convert_mph : bool 

173 if True, convert mph to kph 

174 

175 Returns 

176 ------- 

177 value_clean : string 

178 """ 

179 MPH_TO_KPH = 1.60934 

180 pattern = re.compile(r"[^\d\.,;]") 

181 

182 try: 

183 # strip out everything but numbers, periods, commas, semicolons 

184 value_clean = float(re.sub(pattern, "", value).replace(",", ".")) 

185 if convert_mph and "mph" in value.lower(): 

186 value_clean = value_clean * MPH_TO_KPH 

187 return value_clean 

188 

189 except ValueError: 

190 return None

193def _collapse_multiple_maxspeed_values(value): 

194 """ 

195 Collapse a list of maxspeed values into its mean value. 

196 

197 Parameters 

198 ---------- 

199 value : list or string 

200 an OSM way maxspeed value, or a list of them 

201 

202 Returns 

203 ------- 

204 mean_value : int 

205 an integer representation of the mean value in the list, converted 

206 to kph if original value was in mph. 

207 """ 

208 # if this isn't a list, just return it right back to the caller 

209 if not isinstance(value, list): 

210 return value 

211 

212 else: 

213 try: 

214 # clean each value in list and convert to kph if it is mph then 

215 # return mean value 

216 values = [_clean_maxspeed(x) for x in value] 

217 mean_value = int(pd.Series(values).dropna().mean()) 

218 return mean_value 

219 except ValueError: 

220 return None
