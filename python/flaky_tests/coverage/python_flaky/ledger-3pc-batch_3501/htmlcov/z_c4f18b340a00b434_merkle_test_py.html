471 def test_verify_leaf_inclusion_all_nodes_all_tree_sizes_up_to_4(self): 

472 leaves = ["aa", "bb", "cc", "dd"] 

473 hh = HexTreeHasher() 

474 leaf_hashes = [hh.hash_leaf(l) for l in leaves] 

475 hc = hh.hash_children 

476 proofs_per_tree_size = { 

477 1: [[] ], 

478 2: [[leaf_hashes[1]], [leaf_hashes[0]]], 

479 3: [[leaf_hashes[1], leaf_hashes[2]], # leaf 0 

480 [leaf_hashes[0], leaf_hashes[2]], # leaf 1 

481 [hc(leaf_hashes[0], leaf_hashes[1])]], # leaf 2 

482 4: [[leaf_hashes[1], hc(leaf_hashes[2], leaf_hashes[3])], # leaf 0 

483 [leaf_hashes[0], hc(leaf_hashes[2], leaf_hashes[3])], # leaf 1 

484 [leaf_hashes[3], hc(leaf_hashes[0], leaf_hashes[1])], # leaf 2 

485 [leaf_hashes[2], hc(leaf_hashes[0], leaf_hashes[1])], # leaf 3 

486 ] 

487 } 

488 tree = compact_merkle_tree.CompactMerkleTree(hasher=HexTreeHasher()) 

489 verifier = ledger.merkle_verifier.MerkleVerifier(HexTreeHasher()) 

490 # Increase the tree by one leaf each time 

491 for i in range(4): 

492 tree.append(leaves[i]) 

493 tree_size = i + 1 

494 # ... and check inclusion proof validates for each node 

495 # of the tree 

496 for j in range(tree_size): 

497 proof = proofs_per_tree_size[tree_size][j] 

498 sth = self.STH(tree.root_hash, tree_size) 

499 self.assertTrue( 

500 verifier.verify_leaf_inclusion( 

501 leaves[j], j, proof, sth))
