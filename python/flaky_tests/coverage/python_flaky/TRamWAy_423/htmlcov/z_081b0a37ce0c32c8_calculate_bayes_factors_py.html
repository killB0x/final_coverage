75def calculate_bayes_factors(zeta_ts, zeta_sps, ns, Vs, Vs_pi, loc_error, dim=2, B_threshold=10, verbose=True): 

76 """ 

77 Calculate the Bayes factor for a set of bins given a uniform localization error. 

78 

79 Input: 

80 zeta_ts --- signal-to-noise ratios for the total force = dx_mean / sqrt(var(dx)) in bins. Size: M x 2, 

81 zeta_sps --- signal-to-noise ratios for the spurious force = grad(D) / sqrt(var(dx)) in bins. Size: M x 2, 

82 ns --- number of jumps in each bin. Size: M x 1, 

83 Vs --- jump variance in bins = E((dx - dx_mean) ** 2). Size: M x 1, 

84 Vs_pi --- jump variance in all other bins relative to the current bin. Size: M x 1, 

85 loc_error --- localization error. Same units as variance. Set to 0 if localization error can be ignored; 

86 dim --- dimensionality of the problem; 

87 B_threshold --- the values of Bayes factor for thresholding. 

88 

89 Output: 

90 Bs, forces, min_ns 

91 

92 lg_Bs --- log_10 Bayes factor values in the bins. Size: M x 1, 

93 forces --- Returns 1 if there is strong evidence for the presence of a conservative forces, 

94 -1 for strong evidence for a spurious force, and 0 if the is not enough evidence. Size: M x 1; 

95 min_ns --- minimum number of data points in a bin to support the currently favored model at the required evidence level. Size: M x 1. 

96 

97 Notation: 

98 M --- number of bins. 

99 """ 

100 check_dimensionality(dim) 

101 

102 if verbose: 

103 def _trange(x): return trange(x, desc='Bayes factor calculation') 

104 else: 

105 _trange = range 

106 

107 # Convert to numpy 

108 vars = [zeta_ts, zeta_sps, Vs, Vs_pi, ns] 

109 vars = map(np.asarray, vars) 

110 

111 # Check that the 2nd dimension has size 2 

112 if np.shape(zeta_ts)[1] != 2 or np.shape(zeta_sps)[1] != 2: 

113 raise ValueError("zeta_ts and zeta_sps must be matrices of size (M x 2)") 

114 

115 M = len(ns) 

116 

117 # Calculate 

118 lg_Bs = np.zeros_like(ns) * np.nan 

119 forces = np.zeros_like(ns) * np.nan 

120 min_ns = np.zeros_like(ns, dtype=int) * np.nan 

121 nan_cells_list = [] 

122 with stopwatch("Bayes factor calculation", verbose): 

123 for i in _trange(M): 

124 lg_Bs[i], forces[i], min_ns[i] = _calculate_one_bayes_factor( 

125 zeta_ts[i, :], zeta_sps[i, :], ns[i], Vs[i], Vs_pi[i], loc_error, dim) 

126 try: 

127 lg_Bs[i], forces[i], min_ns[i] = _calculate_one_bayes_factor( 

128 zeta_ts[i, :], zeta_sps[i, :], ns[i], Vs[i], Vs_pi[i], loc_error, dim) 

129 except NaNInputError: 

130 nan_cells_list.append(i) 

131 

132 # Report error if any 

133 try: 

134 logging.warn( 

135 "A NaN value was present in the input parameters for the following cells: {nan_cells_list}.\nBayes factor calculations were skipped for them".format(nan_cells_list=nan_cells_list)) 

136 except NameError: 

137 pass 

138 

139 return [lg_Bs, forces, min_ns]

142def _calculate_one_bayes_factor(zeta_t, zeta_sp, n, V, V_pi, loc_error, dim, B_threshold=10, bl_need_min_n=True): 

143 """Calculate the Bayes factor for one bin.""" 

144 

145 test = check_for_nan(zeta_t, zeta_sp, n, V, V_pi, loc_error) 

146 if test is not 'ok': 

147 raise NaNInputError() 

148 # logging.warning( 

149 # f'>>A {test} value is present in the input parameters for _calculate_one_bayes_factor.\nSkipping Bayes factor calculation for the current bin.\nCall parameters: zeta_t={zeta_t}, zeta_sp={zeta_sp}, n={n}, V={V}, V_pi={V_pi}, loc_error={loc_error}<<') 

150 # return [np.nan] * 3 

151 

152 # # Check if None is present 

153 # if any(var is None for var in [zeta_t, zeta_sp, n, V, V_pi, loc_error]): 

154 # logging.info('None values encountered in cell. Skipping cell.') 

155 # return [np.nan] * 3 

156 

157 # Parameter combinations 

158 n_pi = n_pi_func(dim) 

159 p = pow(n, dim) 

160 u = V_pi / V 

161 v = 1.0 + n_pi / n * u 

162 eta = np.sqrt(n_pi / (n + n_pi)) 

163 

164 if loc_error > 0: 

165 rel_loc_error = n * V / (4 * loc_error) 

166 else: 

167 rel_loc_error = np.inf 

168 

169 def upstairs(l): 

170 return v + eta**2 * norm(zeta_t - l * zeta_sp)**2 

171 

172 def downstairs(l): 

173 return v + norm(zeta_t - l * zeta_sp)**2 

174 

175 # Identify break points 

176 with np.errstate(invalid='ignore', divide='ignore'): 

177 bps = set(zeta_t / zeta_sp) 

178 bps = [bp for bp in bps if bp >= 0 and bp <= 1] 

179 

180 ln_B = calculate_integral_ratio(arg_func_up=upstairs, arg_func_down=downstairs, 

181 pow_up=p, pow_down=p, v=v, 

182 rel_loc_error=rel_loc_error, 

183 break_points=bps, lamb='marg') 

184 lg_B = ln_B / np.log(10) + dim * np.log10(eta) 

185 

186 if bl_need_min_n: 

187 min_n = calculate_minimal_n( 

188 zeta_t=zeta_t, zeta_sp=zeta_sp, n0=n, V=V, V_pi=V_pi, loc_error=loc_error, dim=dim, B_threshold=B_threshold) 

189 else: 

190 min_n = np.nan 

191 

192 # Threshold 

193 force = 1 * (lg_B >= np.log10(B_threshold)) - 1 * (lg_B <= -np.log10(B_threshold)) 

194 

195 return [lg_B, force, min_n]

198def calculate_minimal_n(zeta_t, zeta_sp, n0, V, V_pi, loc_error, dim=2, B_threshold=10): 

199 """ 

200 Calculate the minimal number of jumps per bin needed to obtain strong evidence for the active force or the spurious force model. 

201 

202 Input: 

203 zeta_t, zeta_sp - -- vectors of length 2 for one bin 

204 n0 - -- initial number of jumps(evidence already available). This way the "next" strong evidence can be found, i.e. the minimal number of data points to support the current conclusion 

205 

206 Output: 

207 min_n - -- minimal number of jumps to obtain strong evidence for the conservative force model. 

208 Return - 1 if unable to find the min_n 

209 """ 

210 

211 test = check_for_nan(zeta_t, zeta_sp, n0, V, V_pi, loc_error) 

212 if test is not 'ok': 

213 logging.warning( 

214 '>>A {test} value is present in the input parameters for calculate_minimal_n.\nSkipping minimal n calculation for the current bin.\nCall parameters: zeta_t={zeta_t}, zeta_sp={zeta_sp}, n0={n0}, V={V}, V_pi={V_pi}, loc_error={loc_error}<<'.format( 

215 test=test, zeta_t=zeta_t, zeta_sp=zeta_sp, n0=n0, V=V, V_pi=V_pi, loc_error=loc_error)) 

216 return np.nan 

217 

218 if np.isnan(n0): 

219 logging.error( 

220 'Invalid initial jumps number supplied for minimal n calculation: {}'.format(n0)) 

221 return np.nan 

222 # Local constants 

223 increase_factor = 2 # initial search interval increase with each iteration 

224 max_attempts = 40 

225 xtol = 1.0 

226 rtol = 0.001 

227 

228 lg_B_threshold = np.log10(B_threshold) 

229 

230 # Define the Bayes factor 

231 def lg_B(n): 

232 """A wrapper for the Bayes factor as a function of n.""" 

233 lg_B, _, _ = _calculate_one_bayes_factor( 

234 zeta_t=zeta_t, zeta_sp=zeta_sp, n=n, V=V, V_pi=V_pi, loc_error=loc_error, dim=dim, bl_need_min_n=False) 

235 return lg_B 

236 

237 if abs(lg_B(n0)) >= abs(lg_B_threshold): 

238 return n0 

239 

240 # Find the initial search interval 

241 bl_found = False 

242 n = n0 

243 # print( 

244 # f'Start min n search. Params: zeta_t={zeta_t}, zeta_sp={zeta_sp}, n={n}, V={V}, V_pi={V_pi}, loc_error={loc_error}') 

245 for attempt in range(max_attempts): 

246 n = n0 - 1 + increase_factor ** attempt 

247 # print( 

248 # f'Min_n search. Iteration: {attempt}, n: {n}, lg_B(n): {lg_B(n)}, abs(lg_B_threshold): {abs(lg_B_threshold)}') 

249 if abs(lg_B(n)) >= abs(lg_B_threshold): 

250 bl_found = True 

251 break 

252 

253 if not bl_found: 

254 logging.warning( 

255 "Unable to find the minimal number of data points to provide strong evidence. Search parameters: zeta_t={zeta_t}, zeta_sp={zeta_sp}, n={n}, V={V}, V_pi={V_pi}, loc_error={loc_error}") 

256 return np.nan 

257 

258 # Find a more accurate location 

259 n_interval = [n0, n] 

260 sign = np.sign(lg_B(n)) 

261 

262 def solve_me(n): 

263 return lg_B(n) - sign * lg_B_threshold 

264 

265 min_n = scipy.optimize.brentq(solve_me, n_interval[0], n_interval[1], xtol=xtol, rtol=rtol) 

266 min_n = np.int(np.ceil(min_n)) 

267 

268 return min_n

276def check_for_nan(*args): 

277 """Return 'nan' if a nan value is present in any of the variables. 

278 Return 'None' if any of the variables is None. 

279 Return 'ok' otherwise""" 

280 for i, var in enumerate(args): 

281 try: 

282 if var is None: 

283 return 'None' 

284 elif np.any(np.isnan(var)): 

285 return 'nan' 

286 except: 

287 logging.warning('Unable to check for nan value for variable #{i}'.format(i=i)) 

288 return 'unidentified' 

289 return 'ok'
