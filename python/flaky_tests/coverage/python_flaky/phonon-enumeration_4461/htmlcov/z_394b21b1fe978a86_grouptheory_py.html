73def _make_member_list(n): 

74 """Takes the length of three cyclic group and constructs the member 

75 list so that the permutations can be determined. Each member has 

76 three components, corresponding to the entries for each of the 

77 three cyclic groups. 

78  

79 Args: 

80 n (int): array containing the diagonal elements of the SNF. 

81 

82 Returns: 

83 p (list): The member list for the symmetry group. 

84 """ 

85 from operator import mul 

86 from functools import reduce 

87 

88 depth = int(round(reduce(mul,n,1))) 

89 p = np.zeros([depth,3]) 

90 for im in range(1,depth): # Loop over the members of the translation group 

91 p[im] = list(p[im-1]) # Start with the same digits as in the previous increment  

92 p[im][2] = (p[im-1][2]+1)%n[2] # Increment the first cyclic group 

93 if (p[im][2]==0): # If it rolled over then 

94 p[im][1] = (p[im-1][1] +1) % n[1] # increment the next cyclic group 

95 if (p[im][1]==0): # If this one rolled over too 

96 p[im][0] = (p[im-1][0]+1)%n[0] # Then increment the third one 

97 return p

123def _is_equiv_lattice(lat1,lat2,eps): 

124 """This function determines whether two lattices are equivalent. That 

125 is, if one is an equal-volume derivative lattice of the other. It 

126 uses the idea of Santoro and Mighell (Acta. Cryst. 1972) that, for 

127 equivalent lattices, the transformation matrix that takes the 

128 vectors of str2 to str1 has determinant of 1 and all integer 

129 elements. [LV2]=[LV1][S]=&gt;S=LV1^-1*LV2. But I had to use abs on 

130 the determinant. This is not mentioned by Santoro and Mighell 

131 (issue of opposite handedness---which doesn't matter to me). 

132 

133 Args: 

134 lat1 (array-like): 2D array containing the first lattice. 

135 lat2 (array-like): 2D array containing the second lattice. 

136 eps (float): The finite precision tolerance. 

137 

138 Returns: 

139 is_equiv_lattice (bool): True if lattice are equivalent. 

140 """ 

141 from numpy import linalg, allclose, matmul 

142 is_equiv_lattice = False 

143 lat1inv = linalg.inv(lat1) 

144 S = matmul(lat1inv,lat2).tolist() 

145 if (allclose(abs(linalg.det(S)),1.0,rtol=0,atol=eps) and 

146 allclose(S,[[int(round(S[i][j])) for j in range(3)] for i in range(3)],rtol=0,atol=eps)): 

147 is_equiv_lattice = True 

148 return is_equiv_lattice

150def _get_sLV_fixing_operations(HNF,pLV,rot,shift,dPerm,eps): 

151 """This subroutine deteremines which operations of the parent cell 

152 don't change the supercell. These operations are the symmetry 

153 operations of the supercell. 

154 

155 Args: 

156 HNF (list): A 2D integer list containing the hermite normal form matrix. 

157 pLV (array-like): A 2D array containing the parent lattice vectors. 

158 rot (array-like): A 3D array containing all the rotation matrices. 

159 shift (array-like): A 2D array containing the translations.  

160 dPerm (list): List of basis vector permutations. 

161 eps (float): Finite precision tolerance. 

162 

163 Returns: 

164 fix_op (OpList): A list of the lattice fixing symmetry operations. 

165 rot_perm (list): The rotation permutations. 

166 degeneracy (int): Degeneracy of the supercell. 

167 """ 

168 

169 n_rot = len(rot) 

170 degen_lattices = np.zeros([n_rot,3,3]) 

171 

172 c_degen = 0 

173 ic = 0 # Counter for the fixing operations 

174 tv = 0 

175 tmp_op_rot = [] 

176 tmp_op_shift = [] 

177 tv = [] 

178 t_index = [] # temp variables 

179 for iRot in range(n_rot): # Loop over each rotation 

180 this_rot = rot[iRot] # Store the rotation 

181 orig_lat = np.matmul(np.transpose(pLV),HNF) # Compute the superlattice 

182 rot_lat = np.matmul(this_rot,orig_lat) # Compute the rotated superlattice 

183 if _is_equiv_lattice(rot_lat,orig_lat,eps): 

184 # this operation fixes the lattice and should be recorded 

185 ic += 1 

186 tmp_op_rot.append(this_rot) 

187 tmp_op_shift.append(shift[iRot]) 

188 tv.append(dPerm.v[iRot]) 

189 t_index.append(iRot) 

190 # Added by LN from here 

191 else: 

192 in_list = False 

193 for iDegen in range(c_degen): 

194 if _is_equiv_lattice(degen_lattices[iDegen],rot_lat,eps): 

195 in_list = True 

196 break 

197 

198 if not in_list: 

199 degen_lattices[c_degen] = rot_lat 

200 c_degen += 1 

201 # End of LN additions for degeneracy 

202 # Now we know which rotations fix the lattice and how many 

203 # there are so store them 

204 degeneracy = c_degen 

205 # Allocate the storage for them 

206 fix_op = OpList(tmp_op_rot,tmp_op_shift) # Stuff the rotations into the permanent array 

207 

208 rot_perm = RotPermList(v=tv,RotIndx=t_index) 

209 

210 return(fix_op,rot_perm,degeneracy)

411def _get_dvector_permutations(par_lat,bas_vecs,LatDim,eps): 

412 """This routine applies the symmetry operations of the parent lattice 

413 to the interior points (i.e., the d-set) to see which ones are 

414 equivalent. Labelings of the lattice points that are contain 

415 permutations only of labels on equivalant sites are physically 

416 equivalent and therefore redundant. We use these permutations to 

417 eliminate those duplicate labelings 

418 

419 Args: 

420 par_lat (list): A 2D array containing the parent lattice vectors. 

421 bas_vacs (list): A 2D array containing the basis vectors for the cell. 

422 LatDim (int): 2 if a 2D case 3 if 3D. 

423 eps (float): Finite precesion tolerance. 

424 

425 Returns: 

426 drp_list (list): The basis vector permutations. 

427 """ 

428 

429 from phenum.symmetry import get_spaceGroup, bring_into_cell 

430 from copy import deepcopy 

431 

432 n_d = len(bas_vecs) 

433 a_type = [1]*n_d 

434 

435 bv_copy = deepcopy(bas_vecs) 

436 (rot,shift) = get_spaceGroup(par_lat,a_type,bv_copy,eps = eps) 

437 

438 if LatDim==2: 

439 (rot,shift) = _rm_3D_operations(par_lat,rot,shift,eps) 

440 

441 n_op = len(rot) 

442 inv_par_lat = np.linalg.inv(np.transpose(par_lat)) 

443 nl_temp= n_op # Number of operations that fix the parent 

444 

445 # lattice (but may permute the d-vectors) 

446 v_temp = [] 

447 perm_temp = [] 

448 for iOp in range(n_op): # Try each operation in turn and see how the 

449 # d-vectors are permuted for each 

450 # Rotate each d and add the shift 

451 if len(bas_vecs) == 1: 

452 temp_b = bas_vecs[0] 

453 else: 

454 temp_b = bas_vecs 

455 rd_rot = np.matmul(temp_b,rot[iOp]).tolist() 

456 

457 if n_d > 1: 

458 rd_shift = [shift[iOp]]*n_d 

459 rd = [[rd_rot[i][j] + rd_shift[i][j] for j in range(len(rd_rot[i]))] for i in range(len(rd_rot))] 

460 

461 else: 

462 rd = [rd_rot[i] + shift[iOp][i] for i in range(len(rd_rot))] 

463 

464 trd = list(rd) 

465 if n_d > 1: 

466 for iD in range(n_d): 

467 rd[iD] = bring_into_cell(rd[iD],inv_par_lat,np.transpose(par_lat),eps) 

468 else: 

469 rd = bring_into_cell(rd,inv_par_lat,np.transpose(par_lat),eps) 

470 

471 # The v vector is the vector that must be added (it's a lattice 

472 # vector) to move a rotated d-vector back into the parent cell. 

473 

474 if n_d > 1: 

475 v_temp.append([[rd[i][j] - trd[i][j] for j in range(len(rd[i]))] for i in range(len(rd))]) 

476 else: 

477 v_temp.append([[rd[i] - trd[i] for i in range(len(rd))]]) 

478 if n_d > 1: 

479 perm_temp.append(_map_dvector_permutation(rd,bas_vecs,eps)) 

480 else: 

481 perm_temp.append(_map_dvector_permutation([rd],[bas_vecs],eps)) 

482 

483 drp_list = RotPermList(nL=nl_temp,v=v_temp,perm=perm_temp) 

484 

485 # I don't think we should reduce this list to a unique one. Some 

486 # rotations that don't permute the d's could still permute the 

487 # g's. So we have to keep all the d permutations, even if they 

488 # look redundant here. 

489 return(drp_list)

563 for iD in range(n_d): # Loop over each row in the (d,g) table 

564 temp1 = np.transpose([RPlist[iH].v[iOp][iD]]*n) 

565 temp2 = np.matmul(np.matmul(Op[iH].rot[iOp],T),g) 

566 rag = np.transpose(np.matmul(arrowg,Op[iH].rot[iOp])) 

567 rgp = np.matmul(t_inv,(-temp1+temp2)) 

568 temp_gp = np.round(rgp).astype(int) 

569 temp_ag = np.round(rag).astype(int) 

570 if not np.allclose(rgp,temp_gp,rtol=0,atol=eps): #pragma: no cover 

571 print("Transform left big fractional parts") 

572 exit() 

573 

574 gp = temp_gp 

575 ag = temp_ag 

576 temp_diag = np.transpose([diag]*n) 

577 gp = gp%temp_diag # Mod by each entry of 

578 # the SNF to bring into group Now that the rotated group 

579 # is known, find the mapping of the elements between the 

580 # original group and the permuted group. This is the 

581 # permutation. 

582 

583 skip = [False]*n 

584 for im in range(n): 

585 for jm in range(n): 

586 if skip[jm]: 

587 continue # Skip elements whose mapping is already known 

588 if np.allclose(gp[:,jm],g[:,im]): # these elements 

589 # map to each other The list of operations that fix 

590 # the superlattice are a subset of those that fix 

591 # the parent lattice. RotIndx stores the indicies 

592 # of the parent lattice operations in a list with 

593 # as many entries as supercell fixing operations. 

594 

595 # dperms%perm stores a list of d-vector 

596 # permutations, one permutation (an n_d list) for 

597 # each operation in the parent lattice symmetries 

598 op_indx_in_supercell_list = RPlist[iH].RotIndx[iOp] 

599 row_indx_gtable = dperms.perm.site_perm[op_indx_in_supercell_list][iD] 

600 dgp[im][row_indx_gtable] = jm+iD*n 

601 skip[jm] = True 

602 break 

603 

604 # do the same thing for the arrows 

605 skip = [False]*na_op 

606 for im in range(na_op): 

607 for jm in range(na_op): 

608 if skip[jm]: 

609 continue 

610 if (ag[:,jm] == arrowg[im]).all(): 

611 dap[im] = jm 

612 skip[jm] = True 

613 break 

614 

615 if np.count_nonzero(dgp == 0) > 1 or np.count_nonzero(dap == 0) > 1: #pragma: no cover 

616 print("(d,g)-->(d',g') mapping failed in get_rotation_perm_lists") 

617 exit() 

618 

619 # Now we have the (d',g') table for this rotation. Now 

620 # record the permutation 

621 # permutation in the "long form" 

622 temp_rperms_perm.append(np.transpose(dgp).reshape(n_d*n).tolist()) # store 

623 temp_arrow_perm.append(dap) 

624 

625 # nomenclature: 

626 # N+t = rotation (N) + fractional translation (t) (me bethinks....) 

627 # r = lattice translation 

628 

629 # Now that we have the permutations that are effected by N+t 

630 # type of rotations (for each N in the point group), we need to 

631 # compose them with the permutations effected by lattice 

632 # translations, (parent) lattice translations contained inside the 

633 # supercell (N+t+r, in Rod's nomenclature). Only when we have these 

634 # two lists, and compose them, do we have a guarantee that the 

635 # resulting list is actually a group. For efficiency, we reduce the 

636 # N+t list (remove duplicates). Rod claims that the compositions 

637 # (with the translations) will not have duplicates. 

638 if len(temp_rperms_perm) > 1 and arrows == False: 

639 temp_rperms_perm.sort() 

640 temp_rperms_perm = list(temp_rperms_perm for temp_rperms_perm, _ in itertools.groupby(temp_rperms_perm)) 

641 elif len(temp_rperms_perm) > 1 and arrows == True: 

642 nr = len(temp_rperms_perm[0]) 

643 na = len(temp_arrow_perm[0]) 

644 len_temp_rp = len(temp_rperms_perm) 

645 perms = [] 

646 for i in range(len_temp_rp): 

647 perms.append(list(temp_rperms_perm[i])+list(temp_arrow_perm[i])) 

648 

649 perms = list(perms for perms, _ in itertools.groupby(perms)) 

650 perms.sort() 

651 temp_rperms_perm = [] 

652 temp_arrow_perm = [] 

653 len_perms = len(perms) 

654 for i in range(len_perms): 

655 temp_rperms_perm =[p[:nr] for p in perms] 

656 temp_arrow_perm = [p[-na:] for p in perms] 

657 # The rotations permutations list is now in "alphabetical" 

658 # order and contains no duplicates 

659 

660 # To get the permutations effected by the r's, we don't need 

661 # the r's. We can merely take the member list, the group, and 

662 # add each element of the group to the group itself and see 

663 # what permutation happens. (This part is somewhat redundant 

664 # with make_translation_group in labeling_related module.) 

665 tperms_perm = [] 

666 for ig in range(n): # The number of r's inside the superlattice (the 

667 # translation perms) is the same as the index n 

668 temp_g = np.transpose([g[:,ig]]*n) 

669 tg = g+temp_g#[[g[i][j]+temp_g[i][j] for j in range(len(g[i]))] for i in range(len(g))] # Add the element to the group 

670 temp_diag = np.transpose([diag]*n) 

671 tg = tg%temp_diag#[[tg[i][j]%temp_diag[i][j] for j in range(len(tg[i]))] for i in range(len(tg))] # mod by the SNF entries to 

672 # bring it back to the "primitive" representation 

673 perm = _find_permutation_of_group(np.transpose(g),np.transpose(tg)) 

674 temp_ident = [] 

675 

676 len_ident = len(ident) 

677 for il in range(len_ident): 

678 temp_ident.append([ident[il][i] for i in perm]) 

679 tperms_perm.append(np.reshape(temp_ident,(n*n_d)).tolist()) 

680 

681 rp_list_nl = len(temp_rperms_perm)*n 

682 rp_list_perm_sites = [0]*rp_list_nl 

683 rp_list_perm_arrows = [0]*rp_list_nl 

684 len_t_perm = len(temp_rperms_perm) 

685 for it in range(n): # Loop over unique rotation 

686 for iOp in range(len_t_perm): # Loop over translation perms (r type) 

687 # perms (N+t type) Form the permutation effected by 

688 # composing the iOp-th one with the it-th one 

689 rp_list_temp = [tperms_perm[it][i] for i in temp_rperms_perm[iOp]] 

690 rp_list_perm_sites[iOp*n + it] = rp_list_temp 

691 rp_list_perm_arrows[iOp*n+it] = temp_arrow_perm[iOp] 

692 # ^--- Having gotten both the rotations and the 

693 # translation in the sections above (sort_permutations_list 

694 # etc...), the "operators" of the rotation (one of them is 

695 # R_i) and the translations (one of them is T_j) are now 

696 # "scrambled", i.e., (T_j) o (R_i). Since the *first* 

697 # Rotation R_1 = Id, the *first* entries in the 

698 # RPlist(iH)%perm are equivalent to pure translations only. 

699 # The following entries are a combination of both. 

700 

701 RPlist[iH].perm.site_perm = rp_list_perm_sites 

702 if arrows: 

703 RPlist[iH].perm.arrow_perm = rp_list_perm_arrows 

704 else: 

705 RPlist[iH].perm.arrow_perm = [range(na_op)]*len(rp_list_perm_arrows) 

706 RPlist[iH].n = rp_list_nl 

707 

708 return (RPlist)

711 """Generates the symmetry group for a given lattice and HNF 

712 

713 Args: 

714 

715 par_lat (numpy ndarray): a 2D array that contains the parrent 

716 lattice vectors 

717 

718 bas_vecs (list): A list of the atomic basis vectors for the lattice. 

719 HNF (numpy ndarray): The HNF matrix that defines the supercell. 

720 LatDim (int): 2 if a surface case case 3 if bulk. 

721 arrows (bool, optional): True if the arrow group is to be found as well. 

722 

723 Returns: 

724 sym_group (RotPermList): The symmetry operations represented as permutations 

725 of a list. 

726 """ 

727 

728 from numpy import linalg, allclose 

729 from phenum.symmetry import bring_into_cell, get_spaceGroup 

730 from copy import deepcopy 

731 

732 eps = 1E-10 

733 # map any atoms in the basis that aren't within the cell to be in 

734 # the cell 

735 par_lat_inv = np.transpose(linalg.inv(par_lat)) 

736 temp_basis = deepcopy(bas_vecs) 

737 for i, b_vecs in enumerate(bas_vecs): 

738 # par_lat_inv = linalg.inv(np.transpose(par_lat)) 

739 bas_vecs[i] = bring_into_cell(b_vecs,par_lat_inv,np.transpose(par_lat),eps)

744 .format(" ".join([str(p) for p in temp_basis[i]]),

756 sym_group = _get_rotation_perms_lists(par_lat,[HNF],[L],[SNF],[fixing_ops],[RPList],

820def _group(gen): # pragma: no cover 

821 """This subroutine takes the generators of the group then uses them 

822 to form the entire group. 

823 

824 Args: 

825 gen (list): A 2D array of the generators of the group 

826 where each row is a seperate generator. 

827 

828 Returns: 

829 groupi (list): The full symmetry group. 

830 """ 

831 # q is a counter to track how many group elements we've found 

832 q = 1 

833 # the final output group 

834 groupi = [] 

835 # loop over the generators 

836 for i in gen: 

837 # a second set of generators used to ensure that a group 

838 # action doesn't act an its self 

839 gen2 = [] 

840 # if k is not the ith element of the generators add it to the 

841 # set gen2 

842 for k in gen: 

843 if k != i: 

844 gen2.append(k) 

845 # loop over all except the ith generator 

846 for j in gen2: 

847 # handle the site group and arrow group independently 

848 trans = [j[0][i[0][l]] for l in range(0,len(i[0]))] 

849 rots = [j[1][i[1][l]] for l in range(0,len(i[1]))] 

850 # c is the combination of the trans and rots, if it's not 

851 # in groupi then it should be added. 

852 c = [trans,rots] 

853 if c not in groupi: 

854 groupi.append(c) 

855 

856 # for each new group element, c, we should also see if 

857 # the ith group element acting on it will make another 

858 # new group element. 

859 trans = [c[0][i[0][l]] for l in range(0,len(i[0]))] 

860 rots = [c[1][i[1][l]] for l in range(0,len(i[1]))] 

861 # d is the group action after the ith element has acted on 

862 # c, if it's not in groupi then add it. 

863 d = [trans, rots] 

864 if d not in groupi: 

865 groupi.append(d) 

866 # until d and c are the same again we want to apply the 

867 # ith group element to d so that we've found all possible 

868 # group elements for the combination 

869 while d != c: 

870 trans = [d[0][i[0][l]] for l in range(0,len(i[0]))] 

871 rots = [d[1][i[1][l]] for l in range(0,len(i[1]))] 

872 d = [trans,rots] 

873 # increment q because we found a new group element 

874 q += 1 

875 # every time d is not in group i add it 

876 if d not in groupi: 

877 groupi.append(d) 

878 # new is a tracker that is 0 while we are still finding new group 

879 # elements and 1 when no new ones are found 

880 new = 0 

881 

882 #Here each generator is applied to the group over and over until no new group 

883 #elements are found 

884 while new == 0: 

885 # a second set to store any additional group elements 

886 group2 = [] 

887 for i in gen: 

888 for h in groupi: 

889 # apply the ith generator to the hth element of groupi 

890 # from above to search for new group elements 

891 trans = [h[0][i[0][l]] for l in range(0,len(i[0]))] 

892 rots = [h[1][i[1][l]] for l in range(0,len(i[1]))] 

893 

894 d = [trans,rots] 

895 # if do isn't in groupi or in group2 then add it to 

896 # group2 

897 if d not in groupi and d not in group2: 

898 group2.append(d) 

899 # add anything that is in group2 to groupi 

900 if len(group2) > 0: 

901 groupi.extend(group2) 

902 else: 

903 # if no new group elements were found then we'v found the 

904 # whole group and new is set to 1 

905 new = 1 

906 

907 return(groupi)

928def _rm_3D_operations(aVecs,sgrots,sgshifts,eps): 

929 """This subroutine removes operations that are 3 dimmensional. 

930 

931 Args: 

932 aVecs (list): The 2D array of the primitive real space lattice vectors. 

933 sgrots (list): The 3D array containing the space group rotations. 

934 sgshifts (list): The 2D array containing the space group shifts. 

935 eps (float): Finite precisions tolerance. 

936 

937 Returns: 

938 t_sg_rots (list): The 3D array of the surface rotations. 

939 t_sg_shifts (list): The 2D array of the surface translations. 

940  

941 Raises: 

942 ValueError: if the input vectors aren't primitive. 

943 """ 

944 

945 if not np.allclose( 

946 aVecs[0][1:3],0.0,rtol=0, atol=eps) or not np.allclose(aVecs[2][0:2] 

947 ,0.0,rtol=0,atol=eps): 

948 raise ValueError("Error in rm_3d_operations: only allowed for primitive vectors x00,0xx,0xx")

950 n_rot = len(sgrots) 

951 irot = 0 

952 t_sg_rots = [] 

953 t_sg_shifts = [] 

954 for i in range(n_rot): 

955 if (np.allclose(sgrots[i][0][1:3],0.0,rtol=0,atol=eps) and

959 irot += 1 

960 t_sg_rots.append(sgrots[i]) 

961 t_sg_shifts.append(sgshifts[i]) 

962
