36class Launcher(object): 

37 """Creates a :class:`Launcher` object. This class provides the main 

38 functionality of the Pyautoupdate module. 

39 

40 :param str filepath: Path to file to execute 

41 :param str url: Base URL from which to download new versions 

42 

43 .. note:: 

44 This must be an HTTPS url. HTTP urls are silently changed into HTTPS. 

45 **Use of HTTPS is strictly enforced.** 

46 

47 Parameters, queries, and fragments will be stripped from the URL. 

48 

49 :param str newfiles: Name of ``.zip``, ``tar.gz`` or ``.tar.bz2`` archive 

50 with new versions to download from site 

51 :param int log_level: Logging level for the built in logger 

52 :param tuple args: ``args`` passed to the launched code 

53 :param dict kwargs: ``kwargs`` passed to the launched code 

54 

55 When the code is launched, the following variables are already defined: 

56 

57 +-------------+-------------------------------------------------+ 

58 |Variable Name|Value Description | 

59 +=============+=================================================+ 

60 |``filepath`` |Path to the file that was initially launched | 

61 +-------------+-------------------------------------------------+ 

62 |``url`` |Base url to check and download new versions | 

63 +-------------+-------------------------------------------------+ 

64 |``newfiles`` |Name of the archive to download from the server | 

65 +-------------+-------------------------------------------------+ 

66 |``check_new``|Method to check for updated code | 

67 +-------------+-------------------------------------------------+ 

68 |``pid`` |PID of parent process that spawns the code | 

69 +-------------+-------------------------------------------------+ 

70 |``log`` |Logger for Pyautoupdate and for the executed code| 

71 +-------------+-------------------------------------------------+ 

72 |``args`` |``args`` tuple for the spawned code | 

73 +-------------+-------------------------------------------------+ 

74 |``kwargs`` |``kwargs`` dict for the spawned code | 

75 +-------------+-------------------------------------------------+ 

76 

77 .. warning:: 

78 

79 The :class:`Launcher` uses :class:`multiprocessing.Process` 

80 to run the code. 

81 

82 Please ensure that all arguments passed in as ``args`` and ``kwargs`` 

83 can be pickled. Non-pickleable arguments cannot be passed to the child 

84 process on Windows, and an error will be raised when attempting to run 

85 user code.

314 def _call_code(self, *args, **kwargs): 

315 """Internal function to execute the user code. 

316 

317 This is used as target of a 

318 :class:`multiprocessing.Process` instance. 

319 

320 :param tuple args: ``*args`` tuple from self.args 

321 :param dict kwargs: ``**kwargs`` dict from self.kwargs 

322 

323 .. warning:: 

324 

325 End users should never call this directly. 

326 Please use the :meth:`run` method instead. 

327 """ 

328 # Open code file 

329 # Acquire lock here to avoid TOCTTOU issues with opened code file 

330 # multiprocessing.get_logger again since this is not pickleable 

331 local_log = multiprocessing.get_logger() 

332 local_log.debug("Acquiring code lock to run code") 

333 self.update.acquire() 

334 with open(self.filepath, mode='r') as code_file: 

335 code = code_file.read() 

336 # Set up variables visible to child process 

337 localvar = vars(self).copy() 

338 # Manipulate __dict__ attribute to add handle to check_new 

339 localvar["check_new"] = self.check_new 

340 # Remove handle to process object and lock 

341 # Neither should not be tampered with in child process code 

342 del localvar["_Launcher__process"] 

343 del localvar["_Launcher__process_alive"] 

344 del localvar["update"] 

345 del localvar["past_terminated"] 

346 # Pass in args, kwargs, and logger 

347 localvar["args"] = args 

348 localvar["kwargs"] = kwargs 

349 localvar["log"] = local_log 

350 local_log.debug("Starting process with " 

351 "the following local variables:\n" + 

352 pprint.pformat(localvar)) 

353 # Execute code in file 

354 local_log.info("Starting code from file") 

355 try: 

356 self.__process_alive.set() 

357 exec(code, dict(), localvar) 

358 finally: 

359 local_log.debug("Releasing code lock after running code") 

360 self.update.release() 

361 self.__process_alive.clear() 

362 # Reset past_terminated to False 

363 # (if terminated and rerun, past_terminated should be false) 

364 self.past_terminated = False

366 def run(self, background=False): 

367 """Runs the user code. 

368 

369 If background is ``False``, returns the Process's exitcode. 

370 

371 :param bool background: Whether to run code in background 

372 

373 :return: the exit code if background is ``False`` 

374 :rtype: :class:`int` or :class:`None` 

375 """ 

376 # Find the right error to raise depending on python version 

377 self.log.info("Checking file existence") 

378 try: 

379 error_to_raise = FileNotFoundError 

380 except NameError: 

381 error_to_raise = IOError 

382 if not os.path.isfile(self.filepath): 

383 raise error_to_raise("No file at {0}".format(self.filepath)) 

384 self.log.info("Checking process status") 

385 if self.process_is_alive: 

386 self.log.error("Process is already running") 

387 raise ProcessRunningException 

388 elif self.process_pid is None: 

389 # Process has not run yet 

390 self.log.info("Process has not run yet") 

391 self.log.info("Starting process") 

392 # self.log is not pickleable 

393 # The variable will be reinstantiated inside _call_code 

394 # Temporarily remove here and reinstantiate after start 

395 del self.log 

396 try: 

397 self.__process.start() 

398 finally: 

399 self.log = multiprocessing.get_logger() 

400 self.log.info("Process started") 

401 if not background: 

402 self.process_join() 

403 # Exit code can be used by program that calls the launcher 

404 return self.process_exitcode 

405 elif self.process_exitcode is not None: 

406 # Process has already terminated 

407 # Reinitialize the process instance 

408 self.log.info("Process has already finished") 

409 self.log.info("Reinitializing process object") 

410 self.__process = multiprocessing.Process(target= 

411 self._call_code, 

412 args=self.args, 

413 kwargs=self.kwargs) 

414 # Recursion, since this will reset @property properties 

415 self.run(background) 

416 else: # pragma: no cover 

417 # Should never happen 

418 self.log.error("Process exitcode exists without PID!") 

419 self.log.error("The application is probably in an unstable state.")

423 def check_new(self): 

424 """Retrieves the latest version number from the remote host. 

425 

426 :return: Whether a newer version is available 

427 :rtype: bool 

428 

429 .. note:: 

430 This function internally uses setuptool's ``parse_version`` 

431 to compare versions. 

432 

433 Any versioning scheme conforming to :pep:`440` can be used. 

434 

435 When the server contains an invalid version specification, this 

436 returns ``false``. 

437 

438 .. versionchanged:: 1.0.0 

439 Previously, an invalid server version would cause an exception. 

440 """ 

441 self.log.info("Checking for updates") 

442 request_time = datetime.utcnow() 

443 # If self.queue_update is already present, return false 

444 # TODO: Check again? 

445 if os.path.isfile(self.queue_update): 

446 with open(self.queue_update, 'r') as new_version: 

447 newver = new_version.read() 

448 newver_obj = parse_version(newver) 

449 return isinstance(newver_obj, Version) 

450 else: 

451 versionurl = self.url + self.version_doc 

452 # Get new files 

453 self.log.debug("Retrieving new version from {0}" 

454 .format(versionurl)) 

455 get_new = requests.get(versionurl, allow_redirects=True) 

456 get_new.raise_for_status() 

457 newver = get_new.text 

458 newver_obj = parse_version(newver) 

459 # Read in old version 

460 with open(self.version_doc, 'r') as old_version: 

461 oldver = old_version.read() 

462 oldver = oldver.rstrip("\n") 

463 # Compare old version with new version 

464 invalid = not isinstance(newver_obj, Version) 

465 # Check if new version is valid 

466 if invalid: 

467 self.log.error("Retrieved version number is invalid!\n" 

468 "Please contact the software authors.\n" 

469 "Please include the generated data dump " 

470 "in a bug report.") 

471 # Write notification into log file 

472 version_to_add = "Old {0}|Server Invalid|Time {1}\n"\ 

473 .format(oldver, request_time) 

474 with open(self.version_check_log, "a") as log_file: 

475 log_file.write(version_to_add) 

476 newver_dump = None 

477 # If invalid, dump into dump file 

478 try: 

479 newver_dump = tempfile.NamedTemporaryFile(prefix="newverdump", 

480 delete=False, 

481 mode="wt", 

482 dir=os.getcwd()) 

483 self.log.error("Writing invalid version into {0}" 

484 .format(newver_dump.name)) 

485 newver_dump.write(newver) 

486 # finally runs after return statement 

487 return False 

488 except Exception: 

489 self.log.exception("Unable to write data dump") 

490 return False 

491 finally: 

492 if newver_dump is not None: 

493 newver_dump.close() 

494 # Throw warning after logging 

495 # If version is invalid, upgrade cannot succeed 

496 warnings.warn("Invalid Server version!", CorruptedFileWarning) 

497 

498 # newver_obj will be proper version by this point 

499 has_new = (newver_obj > parse_version(oldver)) 

500 # Add entry to the logfile and update version.txt 

501 if has_new: 

502 version_to_add = "Old {0}|New {1}|Time {2}\n"\ 

503 .format(oldver, newver, request_time) 

504 with open(self.queue_update, 'w') as new_version: 

505 new_version.write(newver) 

506 else: 

507 version_to_add = "Old {0}|Up to date|Time {1}\n"\ 

508 .format(oldver, request_time) 

509 with open(self.version_check_log, "a") as log_file: 

510 log_file.write(version_to_add) 

511 return has_new

561 def _replace_files(self):

576 # Acquiring the lock here prevents this from happening 

577 else:

579 if not self.update.acquire(False): 

580 self.log.warning("Could not acquire lock to update files") 

581 return False

626 CorruptedFileWarning,

628 else:

630 file_rm_temp_dir = os.path.dirname(file_rm_temp)

632 # exist_ok does not exist in Python 2 

633 os.makedirs(file_rm_temp_dir) 

634 if file_rm.split(os.path.sep)[0] not in \ 

635 [self.updatedir, 

636 self.version_doc, 

637 self.version_check_log]: 

638 self.log.debug("Moving {0} to {1}".format(file_rm, 

639 tempdir)) 

640 shutil.move(file_rm, file_rm_temp) 

641 file_rm_dir = os.path.dirname(file_rm) 

642 if os.path.isdir(file_rm_dir): 

643 if not os.listdir(file_rm_dir): 

644 os.rmdir(file_rm_dir) 

645 self.log.debug("Removing directory {0}" 

646 .format(file_rm_dir)) 

647 self.log.info("Removing old filelist") 

648 os.remove(self.file_list) 

649 self.log.info("Creating new filelist") 

650 filelist_new = list() 

651 relpath_start = os.path.join(self.updatedir) 

652 for dirpath, _, filenames in os.walk(self.updatedir): 

653 # _ is dirnames, but it is unused 

654 for filename in filenames: 

655 filepath = os.path.normpath(os.path.join(dirpath, 

656 filename)) 

657 filepath = os.path.relpath(filepath, start=relpath_start) 

658 filepath += "\n" 

659 filelist_new.append(filepath) 

660 self.log.debug("New filelist is:\n" + 

661 pprint.pformat(filelist_new)) 

662 self.log.info("Writing new filelist to {0}" 

663 .format(self.file_list)) 

664 with open(self.file_list, "w") as file_handle: 

665 file_handle.writelines(filelist_new) 

666 self.log.info("Copying downloaded contents to current directory") 

667 copy_glob(os.path.join(self.updatedir, "*"), ".") 

668 self.log.info("Removing backup filelist") 

669 os.remove(filelist_backup.name) 

670 self.log.info("Removing tempdir") 

671 shutil.rmtree(tempdir) 

672 self._reset_update_files() 

673 except Exception: 

674 self.log.exception("An error occured during the update process.") 

675 self.log.error("The temporary directory used is left behind at {}." 

676 .format(tempdir)) 

677 raise 

678 finally: 

679 self.log.debug("Releasing lock after updating files") 

680 self.update.release() 

681 return True
