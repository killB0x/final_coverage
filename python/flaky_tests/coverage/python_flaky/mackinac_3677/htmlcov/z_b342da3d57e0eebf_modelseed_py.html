118def get_modelseed_fba_solutions(model_id): 

119 """ Get the list of fba solutions available for a ModelSEED model. 

120 

121 Parameters 

122 ---------- 

123 model_id : str 

124 ID of model 

125 

126 Returns 

127 ------- 

128 list 

129 List of fba solution data structures 

130 """ 

131 

132 reference = _make_modelseed_reference(model_id) 

133 try: 

134 get_modelseed_model_stats(model_id) # Confirm model exists 

135 solutions = ms_client.call('list_fba_studies', {'model': reference}) 

136 except ServerError as e: 

137 handle_server_error(e, [reference]) 

138 return 

139 

140 # For each solution in the list, get the referenced fba object, and add the 

141 # details on the flux values to the solution. Note ModelSEED stores the 

142 # results of each flux balance analysis separately. 

143 for sol in solutions: 

144 try: 

145 solution_data = get_workspace_object_data(sol['ref']) 

146 except ServerError as e: 

147 handle_server_error(e, sol['ref']) 

148 

149 # A ModelSEED model does not have exchange reactions so instead the results of a flux 

150 # balance analysis reports flux values on metabolites in the extracellular compartment. 

151 # For ModelSEED, a positive flux means the metabolite is consumed and a negative flux 

152 # means the metabolite is produced. 

153 # @todo Should the fluxes be flipped to match COBRA models? 

154 sol['exchanges'] = dict() 

155 for flux in solution_data['FBACompoundVariables']: 

156 exchange_id = flux['modelcompound_ref'].split('/')[-1] 

157 sol['exchanges'][exchange_id] = { 

158 'x': flux['value'], 

159 'lower_bound': flux['lowerBound'], 

160 'upper_bound': flux['upperBound']} 

161 

162 # @todo remove original data? 

163 

164 # Flux values for all of the reactions are reported separately. 

165 sol['reactions'] = dict() 

166 for flux in solution_data['FBAReactionVariables']: 

167 reaction_id = flux['modelreaction_ref'].split('/')[-1] 

168 sol['reactions'][reaction_id] = { 

169 'x': flux['value'], 

170 'lower_bound': flux['lowerBound'], 

171 'upper_bound': flux['upperBound']} 

172 

173 solutions.sort(key=itemgetter('rundate'), reverse=True) # Sort so last completed fba is first in list 

174 return solutions

343def _convert_suffix(modelseed_id, format_type): 

344 """ Convert a string with a compartment suffix from ModelSEED source format to another format. 

345 

346 No conversion is done for unknown format types. 

347 

348 Parameters 

349 ---------- 

350 modelseed_id : str 

351 ID with compartment suffix in ModelSEED source format 

352 format_type : {'modelseed', 'bigg'} 

353 Type of format to convert to 

354 

355 Returns 

356 ------- 

357 str 

358 ID in specified format 

359 """ 

360 

361 # Remove compartment index number for ModelSEED format. For example, rxn00001_c0 becomes rxn00001_c. 

362 # ModelSEED always uses compartment index 0 anyway. 

363 if format_type == 'modelseed': 

364 compartment = re.search(modelseed_suffix_re, modelseed_id).group(1) 

365 return re.sub(modelseed_suffix_re, '', modelseed_id) + '_{0}'.format(compartment) 

366 

367 # Convert to BiGG type format. For example, rxn00001_c0 becomes rxn00001[c]. 

368 elif format_type == 'bigg': 

369 match = re.search(modelseed_suffix_re, modelseed_id) 

370 compartment = match.group(1) 

371 return re.sub(modelseed_suffix_re, '', modelseed_id) + '[{0}]'.format(compartment) 

372 

373 # No conversion is needed or format_type is unknown. 

374 return modelseed_id

377def _add_metabolite(modelseed_compound, model, id_type): 

378 """ Create a COBRApy Metabolite object from a ModelSEED compound dictionary and add it to COBRApy model. 

379 

380 Parameters 

381 ---------- 

382 modelseed_compound : dict 

383 Compound dictionary from ModelSEED model 

384 model : cobra.Model 

385 Model object to add metabolite to 

386 id_type : {'modelseed', 'bigg'} 

387 Type of metabolite ID 

388 

389 Returns 

390 ------- 

391 bool 

392 True when metabolite is a duplicate 

393 """ 

394 

395 # Convert from ModelSEED format to COBRApy format. 

396 cobra_id = _convert_suffix(modelseed_compound['id'], id_type) 

397 cobra_compartment = _convert_compartment(modelseed_compound['modelcompartment_ref'].split('/')[-1], id_type) 

398 cobra_name = _convert_suffix(modelseed_compound['name'], id_type) 

399 

400 # A ModelSEED model usually contains duplicate compounds. Confirm that the duplicate 

401 # compound is an exact duplicate and ignore it. 

402 if model.metabolites.has_id(cobra_id): 

403 metabolite = model.metabolites.get_by_id(cobra_id) 

404 if metabolite.name != cobra_name: 

405 warn('Duplicate ModelSEED compound ID {0} has different name, {1} != {2}' 

406 .format(cobra_id, metabolite.name, cobra_name)) 

407 if metabolite.formula != modelseed_compound['formula']: 

408 warn('Duplicate ModelSEED compound ID {0} has different formula, {1} != {2}' 

409 .format(cobra_id, metabolite.formula, modelseed_compound['formula'])) 

410 if metabolite.charge != modelseed_compound['charge']: 

411 warn('Duplicate ModelSEED compound ID {0} has different charge {1} != {2}' 

412 .format(cobra_id, metabolite.charge, modelseed_compound['charge'])) 

413 if metabolite.compartment != cobra_compartment: 

414 warn('Duplicate ModelSEED compound ID {0} has different compartment {1} != {2}' 

415 .format(cobra_id, metabolite.compartment, cobra_compartment)) 

416 return True 

417 

418 # Create the Metabolite object and add it to the model. 

419 metabolite = Metabolite(id=cobra_id, 

420 formula=modelseed_compound['formula'], 

421 name=cobra_name, 

422 charge=modelseed_compound['charge'], 

423 compartment=cobra_compartment) 

424 model.add_metabolites([metabolite]) 

425 

426 return False

429def _add_reaction(modelseed_reaction, model, id_type, likelihoods): 

430 """ Create a COBRApy Reaction object from a ModelSEED reaction dictionary and add it to COBRApy model. 

431 

432 Parameters 

433 ---------- 

434 modelseed_reaction : dict 

435 Reaction dictionary from ModelSEED model 

436 model : cobra.Model 

437 Model object to add reaction to 

438 id_type : str 

439 Type of reaction ID 

440 likelihoods : dict 

441 Dictionary with reaction likelihoods from ModelSEED model 

442 """ 

443 

444 # Set upper and lower bounds based directionality. Switch reverse reactions to forward reactions (ModelSEED 

445 # does this when exporting to SBML). 

446 reverse = 1.0 

447 if modelseed_reaction['direction'] == '=': 

448 lower_bound = -1000.0 

449 upper_bound = 1000.0 

450 elif modelseed_reaction['direction'] == '>': 

451 lower_bound = 0.0 

452 upper_bound = 1000.0 

453 elif modelseed_reaction['direction'] == '<': 

454 lower_bound = 0.0 

455 upper_bound = 1000.0 

456 reverse = -1.0 

457 else: 

458 warn('Reaction direction {0} assumed to be reversible for reaction {1}' 

459 .format(modelseed_reaction['direction'], modelseed_reaction['id'])) 

460 lower_bound = -1000.0 

461 upper_bound = 1000.0 

462 

463 # Create the Reaction object. 

464 reaction = Reaction(id=_convert_suffix(modelseed_reaction['id'], id_type), 

465 name=re.sub(modelseed_suffix_re, '', modelseed_reaction['name']), 

466 lower_bound=lower_bound, 

467 upper_bound=upper_bound) 

468 

469 # Create dictionary of metabolites and add them to the reaction. 

470 metabolites = dict() 

471 for reagent in modelseed_reaction['modelReactionReagents']: 

472 cobra_metabolite_id = _convert_suffix(reagent['modelcompound_ref'].split('/')[-1], id_type) 

473 metabolite = model.metabolites.get_by_id(cobra_metabolite_id) 

474 metabolites[metabolite] = float(reagent['coefficient']) * reverse 

475 reaction.add_metabolites(metabolites) 

476 

477 # If there are proteins associated with the reaction, build the gene reaction rule and 

478 # add corresponding Gene objects to the model. 

479 if len(modelseed_reaction['modelReactionProteins']) > 0: 

480 # Build a list of proteins associated with the reaction. 

481 protein_list = list() 

482 for protein in modelseed_reaction['modelReactionProteins']: 

483 # Spontaneous and universal reactions can have an entry in the list of proteins 

484 # that does not have any protein subunits. 

485 if len(protein['modelReactionProteinSubunits']) == 0: 

486 continue 

487 

488 # Build a list of protein subunits in this protein. 

489 subunit_list = list() 

490 for subunit in protein['modelReactionProteinSubunits']: 

491 # A protein with multiple subunits can have a subunit that is not linked 

492 # to a feature in the genome of the organism. 

493 if len(subunit['feature_refs']) == 0: 

494 continue 

495 

496 # Build a list of features in this protein subunit. 

497 feature_list = list() 

498 for feature in subunit['feature_refs']: 

499 # Extract the gene ID from the reference to the feature in the genome and 

500 # remove the "fig|" prefix. 

501 gene_id = re.sub(patric_gene_prefix_re, '', feature.split('/')[-1]) 

502 if not model.genes.has_id(gene_id): 

503 gene = Gene(gene_id, subunit['role']) # Use the role name as the Gene name 

504 model.genes.append(gene) 

505 feature_list.append(gene_id) 

506 

507 # Join multiple features using an OR relationship. 

508 if len(feature_list) > 1: 

509 subunit_list.append('( {0} )'.format(' or '.join(feature_list))) 

510 else: 

511 subunit_list.append(feature_list[0]) 

512 

513 # Join multiple protein subunits using an AND relationship. 

514 if len(subunit_list) > 0: 

515 if len(subunit_list) > 1: 

516 protein_list.append('( {0} )'.format(' and '.join(subunit_list))) 

517 else: 

518 protein_list.append(subunit_list[0]) 

519 

520 # If there is an association to a feature, add the rule to the reaction. 

521 if len(protein_list) > 0: 

522 # Join multiple proteins using an OR relationship. 

523 if len(protein_list) > 1: 

524 gpr_rule = '( {0} )'.format(' or '.join(protein_list)) 

525 else: 

526 gpr_rule = protein_list[0] 

527 

528 reaction.gene_reaction_rule = gpr_rule 

529 

530 # Add a note with gap fill details. ModelSEED gap fill data is a dictionary where the key is the 

531 # gap fill solution ID and the value indicates if the reaction was added or reversed and the 

532 # direction of the reaction. For example: {u'gf.0': u'added:>'} 

533 if len(modelseed_reaction['gapfill_data']) > 0: 

534 reaction.notes['gapfill_data'] = modelseed_reaction['gapfill_data'] 

535 

536 # Add a note with likelihood value if available. 

537 if modelseed_reaction['id'] in likelihoods: 

538 reaction.notes['likelihood'] = likelihoods[modelseed_reaction['id']] 

539 reaction.notes['likelihood_str'] = '{0:.6f}'.format(reaction.notes['likelihood']) 

540 else: 

541 reaction.notes['likelihood_str'] = 'unknown' 

542 

543 # Finally, add the reaction to the model. 

544 model.add_reactions([reaction]) 

545 

546 return

549def create_cobra_model_from_modelseed_model(model_id, id_type='modelseed', validate=False): 

550 """ Create a COBRA model from a ModelSEED model. 

551 

552 Parameters 

553 ---------- 

554 model_id : str 

555 ID of model 

556 id_type : {'modelseed', 'bigg'} 

557 Type of IDs ('modelseed' for _c or 'bigg' for '[c]) 

558 validate : bool 

559 When True, check for common problems 

560 

561 Returns 

562 ------- 

563 cobra.Model 

564 Model object 

565 """ 

566 

567 # Validate the id_type parameter. 

568 if id_type == 'modelseed': 

569 cytosol_suffix = '_c' 

570 elif id_type == 'bigg': 

571 cytosol_suffix = '[c]' 

572 else: 

573 raise ValueError('id_type {0} is not supported'.format(id_type)) 

574 

575 # Get the ModelSEED model data. 

576 data = get_modelseed_model_data(model_id) 

577 reference = _make_modelseed_reference(model_id) 

578 

579 # Get the workspace object with the likelihoods and put the likelihood values in a dictionary 

580 # keyed by reaction ID. Calculating likelihoods is optional so the object may not exist. 

581 try: 

582 likelihood_data = get_workspace_object_data(join(reference, 'rxnprobs')) 

583 likelihoods = {r[0]: r[1] for r in likelihood_data['reaction_probabilities']} 

584 except ObjectNotFoundError: 

585 likelihoods = dict() 

586 

587 # Create a new COBRApy Model object. 

588 model = Model(data['id'], name=data['name']) 

589 

590 # Add compartments to the COBRApy model. 

591 for index in range(len(data['modelcompartments'])): 

592 modelseed_compartment = data['modelcompartments'][index] 

593 cobra_id = _convert_compartment(modelseed_compartment['id'], id_type) 

594 model.compartments[cobra_id] = modelseed_compartment['label'][:-2] # Strip _0 suffix from label 

595 

596 # Create Metabolite objects for all of the compounds in the ModelSEED model. 

597 num_duplicates = 0 

598 for index in range(len(data['modelcompounds'])): 

599 duplicate = _add_metabolite(data['modelcompounds'][index], model, id_type=id_type) 

600 if duplicate: 

601 num_duplicates += 1 

602 

603 # Report the number of duplicate metabolites. 

604 if validate and num_duplicates > 0: 

605 warn('{0} duplicate metabolites were removed from model {1} of {2}' 

606 .format(num_duplicates, model.id, model.name)) 

607 

608 # Add all of the reactions to the COBRApy model. 

609 for index in range(len(data['modelreactions'])): 

610 _add_reaction(data['modelreactions'][index], model, id_type, likelihoods) 

611 

612 # Add exchange reactions for metabolites in extracellular compartment. 

613 for index in range(len(model.metabolites)): 

614 metabolite = model.metabolites[index] 

615 if metabolite.compartment.startswith('e'): 

616 # Single reactant metabolite makes a system boundary reaction. 

617 reaction = Reaction(id='EX_' + metabolite.id, 

618 name=metabolite.name + ' exchange', 

619 lower_bound=-1000.0, 

620 upper_bound=1000.0) 

621 reaction.add_metabolites({metabolite: -1.0}) 

622 reaction.notes['likelihood_str'] = 'n/a' 

623 model.add_reactions([reaction]) 

624 

625 # A ModelSEED model must have an exchange reaction for the special biomass metabolite. 

626 metabolite = model.metabolites.get_by_id('cpd11416'+cytosol_suffix) 

627 reaction = Reaction(id='EX_' + metabolite.id, 

628 name=metabolite.name, 

629 lower_bound=-1000.0, 

630 upper_bound=1000.0) 

631 reaction.add_metabolites({metabolite: -1.0}) 

632 model.add_reactions([reaction]) 

633 

634 # Note that when ModelSEED exports to SBML, it includes exchange reactions for 

635 # cpd15302_c0 "Glycogen(n-1)" and cpd08636_c0 "4-5-dihydroxy-2-3-pentanedione". 

636 # No idea why exchange reactions for metabolites in the cytosol compartment are 

637 # added to the SBML file. 

638 # metabolite = model.metabolites.get_by_id('cpd08636'+cytosol_suffix) 

639 # reaction = Reaction(id='EX_' + metabolite.id, 

640 # name=metabolite.name, 

641 # lower_bound=-1000.0, 

642 # upper_bound=1000.0) 

643 # reaction.add_metabolites({metabolite: -1.0}) 

644 # model.add_reactions([reaction]) 

645 # 

646 # metabolite = model.metabolites.get_by_id('cpd15302'+cytosol_suffix) 

647 # reaction = Reaction(id='EX_' + metabolite.id, 

648 # name=metabolite.name, 

649 # lower_bound=-1000.0, 

650 # upper_bound=1000.0) 

651 # reaction.add_metabolites({metabolite: -1.0}) 

652 # model.add_reactions([reaction]) 

653 

654 # Add the biomass reactions to the COBRApy model. ModelSEED models can have more than one biomass reaction 

655 # but the model does not identify which one to use as the objective so always use the first one. 

656 if len(data['biomasses']) > 1: 

657 warn('Found {0} biomass reactions and selected {1} as the objective' 

658 .format(len(data['biomasses']), data['biomasses'][0]['id'])) 

659 for index in range(len(data['biomasses'])): 

660 biomass = data['biomasses'][index] 

661 biomass_metabolites = dict() 

662 for biomass_compound in biomass['biomasscompounds']: 

663 cobra_id = _convert_suffix(biomass_compound['modelcompound_ref'].split('/')[-1], id_type) 

664 metabolite = model.metabolites.get_by_id(cobra_id) 

665 biomass_metabolites[metabolite] = biomass_compound['coefficient'] 

666 reaction = Reaction(id=biomass['id'], 

667 name=biomass['name'], 

668 lower_bound=0.0, 

669 upper_bound=1000.0) 

670 reaction.add_metabolites(biomass_metabolites) 

671 model.add_reactions([reaction]) 

672 if index == 0: 

673 reaction.objective_coefficient = 1. 

674 

675 # If requested, validate the COBRApy model. 

676 if validate: 

677 # See if all of the reactions are mass balanced. 

678 num_unbalanced = 0 

679 for r in model.reactions: 

680 if not r.id.startswith('EX_'): # Skip exchange reactions 

681 unbalanced = r.check_mass_balance() 

682 if len(unbalanced) > 0: 

683 warn('Reaction {0} is unbalanced because {1}\n {2}' 

684 .format(r.id, unbalanced, r.build_reaction_string(use_metabolite_names=True))) 

685 num_unbalanced += 1 

686 if num_unbalanced > 0: 

687 warn('Model {0} has {1} unbalanced reactions'.format(model.id, num_unbalanced)) 

688 

689 return model

692def create_universal_model(template_reference, id_type='modelseed'): 

693 """ Create a universal model from a ModelSEED template model. 

694 

695 A template model has all of the reactions and metabolites that are available for 

696 inclusion in a model. There are different template models for different types 

697 of organisms (e.g. gram negative bacteria). Use an universal model as input to 

698 one of the gap fill functions. 

699 

700 Parameters 

701 ---------- 

702 template_reference : str 

703 Workspace reference to template model 

704 id_type : {'modelseed', 'bigg'}, optional 

705 Type of IDs ('modelseed' for _c or 'bigg' for '[c]) 

706 

707 Returns 

708 ------- 

709 cobra.Model 

710 Model object with all reactions for models of a type of organism 

711 """ 

712 

713 # Get the template model data from the workspace object. 

714 data = get_workspace_object_data(template_reference) 

715 

716 # Create a dict to look up compounds. 

717 compound_index = dict() 

718 for index in range(len(data['compounds'])): 

719 key = '~/compounds/id/{0}'.format(data['compounds'][index]['id']) 

720 compound_index[key] = index 

721 

722 # Create a new COBRApy Model object. 

723 model = Model(data['id'], name=data['name']) 

724 

725 # Add template compartments to the universal model. 

726 for index in range(len(data['compartments'])): 

727 modelseed_compartment = data['compartments'][index] 

728 model.compartments[modelseed_compartment['id']] = modelseed_compartment['name'] 

729 

730 # Create Metabolite objects for all of the compounds in the template model. Metabolite 

731 # data is split between the "compcompounds" (compounds in a compartment) and the 

732 # "compounds" lists. 

733 for compcompound in data['compcompounds']: 

734 compound = data['compounds'][compound_index[compcompound['templatecompound_ref']]] 

735 cobra_id = _convert_suffix(compcompound['id'], id_type) 

736 metabolite = Metabolite(id=cobra_id, 

737 formula=compound['formula'], 

738 name=compound['name'], 

739 charge=compcompound['charge'], 

740 compartment=compcompound['templatecompartment_ref'].split('/')[-1]) 

741 model.add_metabolites([metabolite]) 

742 

743 # Create Reaction objects for all of the reactions in the template model. 

744 for template_reaction in data['reactions']: 

745 # Set upper and lower bounds based directionality. Switch reverse reactions to 

746 # forward reactions. 

747 reverse = 1.0 

748 if template_reaction['direction'] == '=': 

749 lower_bound = -1000.0 

750 upper_bound = 1000.0 

751 elif template_reaction['direction'] == '>': 

752 lower_bound = 0.0 

753 upper_bound = 1000.0 

754 elif template_reaction['direction'] == '<': 

755 lower_bound = 0.0 

756 upper_bound = 1000.0 

757 reverse = -1.0 

758 else: 

759 warn('Reaction direction {0} assumed to be reversible for reaction {1}' 

760 .format(template_reaction['direction'], template_reaction['id'])) 

761 lower_bound = -1000.0 

762 upper_bound = 1000.0 

763 

764 # Create the Reaction object. 

765 reaction = Reaction(id=_convert_suffix(template_reaction['id'], id_type), 

766 name=template_reaction['name'], 

767 lower_bound=lower_bound, 

768 upper_bound=upper_bound) 

769 

770 # Create dictionary of metabolites and add them to the reaction. 

771 metabolites = dict() 

772 for reagent in template_reaction['templateReactionReagents']: 

773 cobra_metabolite_id = _convert_suffix(reagent['templatecompcompound_ref'].split('/')[-1], id_type) 

774 metabolite = model.metabolites.get_by_id(cobra_metabolite_id) 

775 metabolites[metabolite] = float(reagent['coefficient']) * reverse 

776 reaction.add_metabolites(metabolites) 

777 

778 # Add a note with the ModelSEED reaction type (universal, spontaneous, conditional, or gapfilling). 

779 reaction.notes['type'] = template_reaction['type'] 

780 

781 # Finally, add the reaction to the model. 

782 model.add_reactions([reaction]) 

783 

784 return model

787def optimize_modelseed_model(model_id, media_reference=None): 

788 """ Run flux balance analysis on a ModelSEED model. 

789 

790 Parameters 

791 ---------- 

792 model_id : str 

793 ID of model 

794 media_reference : str 

795 Workspace reference to media to optimize on (default is complete media) 

796 

797 Returns 

798 ------- 

799 float 

800 Objective value 

801 """ 

802 

803 # Get the current list of fba solutions which is the only way to tell if this 

804 # optimization is successful because fba_count is not updated in the metadata. 

805 fba_count = len(get_modelseed_fba_solutions(model_id)) 

806 

807 # Set input parameters for method. 

808 reference = _make_modelseed_reference(model_id) 

809 params = dict() 

810 params['model'] = reference 

811 params['predict_essentiality'] = 1 

812 if media_reference is not None: 

813 params['media'] = media_reference 

814 

815 # Run the server method. 

816 try: 

817 job_id = ms_client.call('FluxBalanceAnalysis', params) 

818 _wait_for_job(job_id) 

819 except ServerError as e: 

820 references = [reference] 

821 if media_reference is not None: 

822 references.append(media_reference) 

823 handle_server_error(e, references) 

824 

825 # The completed job does not have the reference to the fba object that 

826 # was just created so get the list of solutions. Last completed 

827 # solution is first in the list. 

828 solutions = get_modelseed_fba_solutions(model_id) 

829 if fba_count == len(solutions): 

830 warn('Optimization for {0} did not return a solution'.format(model_id)) 

831 return 0.0 

832 return float(solutions[0]['objective'])

835def reconstruct_modelseed_model(genome_id, source='patric', template_reference=None, likelihood=False, model_id=None): 

836 """ Reconstruct a draft ModelSEED model for an organism. 

837 

838 Parameters 

839 ---------- 

840 genome_id : str 

841 Genome ID or workspace reference to genome 

842 source : {'patric', 'rast', 'workspace'}, optional 

843 Source of genome 

844 template_reference : str, optional 

845 Workspace reference to template model 

846 likelihood : bool, optional 

847 True to generate reaction likelihoods 

848 model_id : str, optional 

849 ID of output model (default is genome ID) 

850 

851 Returns 

852 ------- 

853 dict 

854 Dictionary of current model statistics 

855 """ 

856 

857 # Set input parameters for method. 

858 params = dict() 

859 if source == 'patric': 

860 params['genome'] = 'PATRIC:' + genome_id 

861 elif source == 'rast': 

862 params['genome'] = 'RAST:' + genome_id 

863 elif source == 'workspace': 

864 params['genome'] = genome_id 

865 else: 

866 raise ValueError('Source type {0} is not supported'.format(source)) 

867 if model_id is None: 

868 model_id = genome_id 

869 params['output_file'] = model_id 

870 if template_reference is not None: 

871 params['template_model'] = template_reference 

872 if likelihood: 

873 params['probanno'] = 1 

874 else: 

875 params['probanno'] = 0 

876 params['gapfill'] = 0 

877 params['predict_essentiality'] = 0 

878 

879 # Workaround for ModelSEED workspace bug. The user's modelseed folder must exist before saving 

880 # the model. Otherwise the type of the folder created for the model is not "modelfolder" and 

881 # subsequent operations on the model will fail. 

882 if ms_client.username is None: 

883 ms_client.set_authentication_token() 

884 folder_reference = '/{0}/{1}'.format(ms_client.username, model_folder) 

885 try: 

886 get_workspace_object_meta(folder_reference) 

887 except ObjectNotFoundError: 

888 put_workspace_object(folder_reference, 'folder') 

889 

890 # Run the server method. 

891 try: 

892 job_id = ms_client.call('ModelReconstruction', params) 

893 except ServerError as e: 

894 references = None 

895 if template_reference is not None: 

896 references = [template_reference] 

897 handle_server_error(e, references) 

898 

899 # The task structure has the workspace where the model is stored but not the name of the model. 

900 _wait_for_job(job_id) 

901 

902 # Get the model statistics for the model. 

903 stats = get_modelseed_model_stats(model_id) 

904 if stats['num_genes'] == 0: # ModelSEED does not return an error if the genome ID is invalid 

905 warn('Model for genome ID {0} has no genes, verify genome ID is valid'.format(genome_id)) 

906 return stats

909def _wait_for_job(jobid): 

910 """ Wait for a job submitted to the ModelSEED app service to end. 

911 

912 Parameters 

913 ---------- 

914 jobid : str 

915 ID of submitted job 

916 

917 Returns 

918 ------- 

919 dict 

920 Task structure with status of job 

921 

922 Raises 

923 ------ 

924 JobError 

925 When a job with the specified ID was not found 

926 """ 

927 

928 task = None 

929 done = False 

930 while not done: 

931 jobs = ms_client.call('CheckJobs', {}) 

932 if jobid in jobs: 

933 task = jobs[jobid] 

934 if task['status'] == 'failed': 

935 if 'error' in task: 

936 raise ServerError(task['error']) 

937 raise ServerError('Job submitted to ModelSEED app service failed, no details provided in response') 

938 elif task['status'] == 'completed': 

939 done = True 

940 else: 

941 sleep(3) 

942 else: 

943 raise JobError('Job {0} was not found'.format(jobid)) 

944 return task
