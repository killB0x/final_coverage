166 def output_sync_map_file(self, container_root_path=None): 

167 """ 

168 Output the sync map file for this task. 

169 

170 If ``container_root_path`` is specified, 

171 the output sync map file will be created 

172 at the path obtained by joining 

173 the ``container_root_path`` and the relative path 

174 of the sync map inside the container. 

175 

176 Otherwise, the sync map file will be created at the path 

177 ``self.sync_map_file_path_absolute``. 

178 

179 Return the the path of the sync map file created, 

180 or ``None`` if an error occurred. 

181 

182 :param string container_root_path: the path to the root directory 

183 for the output container 

184 :rtype: string 

185 """ 

186 if self.sync_map is None: 

187 self.log_exc(u"The sync_map object has not been set", None, True, TypeError) 

188 

189 if (container_root_path is not None) and (self.sync_map_file_path is None): 

190 self.log_exc(u"The (internal) path of the sync map has been set", None, True, TypeError) 

191 

192 self.log([u"container_root_path is %s", container_root_path]) 

193 self.log([u"self.sync_map_file_path is %s", self.sync_map_file_path]) 

194 self.log([u"self.sync_map_file_path_absolute is %s", self.sync_map_file_path_absolute]) 

195 

196 if (container_root_path is not None) and (self.sync_map_file_path is not None): 

197 path = os.path.join(container_root_path, self.sync_map_file_path) 

198 elif self.sync_map_file_path_absolute: 

199 path = self.sync_map_file_path_absolute 

200 gf.ensure_parent_directory(path) 

201 self.log([u"Output sync map to %s", path]) 

202 

203 eaf_audio_ref = self.configuration["o_eaf_audio_ref"] 

204 head_tail_format = self.configuration["o_h_t_format"] 

205 levels = self.configuration["o_levels"] 

206 smil_audio_ref = self.configuration["o_smil_audio_ref"] 

207 smil_page_ref = self.configuration["o_smil_page_ref"] 

208 sync_map_format = self.configuration["o_format"] 

209 

210 self.log([u"eaf_audio_ref is %s", eaf_audio_ref]) 

211 self.log([u"head_tail_format is %s", head_tail_format]) 

212 self.log([u"levels is %s", levels]) 

213 self.log([u"smil_audio_ref is %s", smil_audio_ref]) 

214 self.log([u"smil_page_ref is %s", smil_page_ref]) 

215 self.log([u"sync_map_format is %s", sync_map_format]) 

216 

217 self.log(u"Calling sync_map.write...") 

218 parameters = { 

219 gc.PPN_TASK_OS_FILE_EAF_AUDIO_REF: eaf_audio_ref, 

220 gc.PPN_TASK_OS_FILE_HEAD_TAIL_FORMAT: head_tail_format, 

221 gc.PPN_TASK_OS_FILE_LEVELS: levels, 

222 gc.PPN_TASK_OS_FILE_SMIL_AUDIO_REF: smil_audio_ref, 

223 gc.PPN_TASK_OS_FILE_SMIL_PAGE_REF: smil_page_ref, 

224 } 

225 self.sync_map.write(sync_map_format, path, parameters) 

226 self.log(u"Calling sync_map.write... done") 

227 return path
