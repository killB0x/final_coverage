27import random 

28import copy 

29import bisect

31def mutate(obj, controls): 

32 """ 

33 Create a new object by taking an object and a set of control objects 

34 to change randomly 

35 

36 obj An object to copy. This will be deepcopied, not modified 

37 controls A list of control objects 

38 """ 

39 new_obj = copy.deepcopy(obj) 

40 for control in controls: 

41 # Add 10% variations to controls 

42 new_value = control.get(obj) * (1. + 0.1 * (random.random() - 0.5)) 

43 if abs(new_value) < 1e-10: 

44 # Don't know what scale, so add ~1 random value to it 

45 new_value += random.normal(loc=0.0, scale=1.0) 

46 control.set(new_obj, new_value) 

47 return new_obj

49def pickUnique(N, m, e): 

50 """Pick m random values from the range 0...(N-1), excluding those in list e 

51 The returned values should be in a random order (not sorted) 

52 """ 

53 assert m <= N - len(e) 

54 

55 inds = sorted(e) # Sorted list of indices. Used to avoid clashes 

56 others = [] # The list of three agents 

57 for i in range(m): 

58 newind = random.randint(0, N-1-i-len(e)) 

59 for ind in inds: 

60 if newind == ind: 

61 newind += 1 

62 bisect.insort(inds, newind) 

63 others.append(newind) 

64 return others

66def optimise(obj, controls, measure, maxgen=10, N=10, CR=0.3, F=1.0, monitor=None): 

67 """Use Differential Evolution to optimise an object 

68 https://en.wikipedia.org/wiki/Differential_evolution 

69  

70 obj is an object to be used as starting solution 

71 These objects are deep copied, and passed as arguments 

72 to controls and measure functions 

73  

74 controls A list of control objects. These objects must 

75 have methods: 

76 .set(object, value) which modifies the given object 

77 .get(object) which returns the value from the object 

78 

79 measure(object) is a function which returns a score (value) for a 

80 given object. The optimiser tries to minimise this 

81 value. 

82  

83 maxgen is the maximum number of generations 

84 

85 N >= 4 is the population size 

86 CR [0,1] is the crossover probability 

87 F [0,2] is the differential weight 

88  

89 monitor(generation, best, population)  

90 

91 A function to be called each generation with the best 

92 solution and the whole population  

93 generation = integer 

94 best = (score, object) 

95 population = [(score, object)] 

96 

97 Returns the object with the lowest measure (score). 

98 

99 """ 

100 assert N >= 4 

101 

102 best = (measure(obj), obj) # Highest score, candidate solution (agent) 

103 population = [best] # List of (score, agent) 

104 

105 for i in range(N-1): 

106 agent = mutate(obj, controls) 

107 score_agent = (measure(agent), agent) 

108 population.append(score_agent) 

109 if score_agent[0] > best[0]: 

110 best = score_agent 

111 

112 for generation in range(maxgen): 

113 next_pop = [] # Next generation 

114 for ai, agent in enumerate(population): 

115 # Pick three other random agents, all different 

116 others = [population[index][1] for index in pickUnique(N, 3, [ai])] 

117 

118 new_obj = copy.deepcopy(agent[1]) 

119 R = random.randint(0, len(controls)-1) # Pick a random control to modify 

120 for i, control in enumerate(controls): 

121 if i == R or random.random() < CR: 

122 control.set(new_obj, 

123 control.get(others[0]) + F * (control.get(others[1]) - control.get(others[2]))) 

124 score = measure(new_obj) 

125 if score < agent[0]: 

126 # Better than original 

127 new_agent = (score, new_obj) 

128 next_pop.append(new_agent) 

129 if score < best[0]: 

130 # Now the best candidate 

131 best = new_agent 

132 else: 

133 next_pop.append(agent) 

134 # End of generation. Call monitor with best object, score 

135 if monitor: 

136 monitor(generation, best, next_pop) 

137 # Change to next generation 

138 population = next_pop 

139 # Finished, return best candidate 

140 return best[1]
