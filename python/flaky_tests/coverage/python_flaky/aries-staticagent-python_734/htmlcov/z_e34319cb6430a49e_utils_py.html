68def preprocess(preprocessor: Callable): 

69 """Preprocess a message before handling. 

70 

71 This facility might be used to validate and unpack signatures/attachments, 

72 validate timing or ordering, add data to the message object (i.e. append 

73 protocol state information), etc. 

74 

75 A deep copy of the original message is given to preprocessors to prevent 

76 accidental manipulation. Preprocessors must return the altered message 

77 object if modification is intended. Preprocessors should raise an error if 

78 preprocessing fails. 

79 """ 

80 def _preprocess_decorated(func): 

81 @wraps(func) 

82 def _wrapped(*args, **kwargs): 

83 msg, index = find_message_in_args(args) 

84 preprocessed = preprocessor(copy.deepcopy(msg)) 

85 

86 if preprocessed: 

87 args = list(args) 

88 args[index] = preprocessed 

89 

90 return func(*args, **kwargs) 

91 return _wrapped 

92 return _preprocess_decorated
