1""" 

2The profile contains some defaults for the execution. 

3These defaults are loaded using a double configuration file schema: 

4First, a general file is read and then, the values on it 

5can be updated by a more specific one configured by the user. 

6 

7.. important:: 

8 

9 It is the specific configuration file the one that indicates the 

10 executor to be used. If not provided, a local executor is used. 

11 

12""" 

13from os import path 

14 

15from bgconfig import BGConfig 

16from configobj import ConfigObj 

17 

18from qmap import executor 

19from qmap.globals import QMapError, CONFIGURATION_NAMESPACE 

20from qmap.job.parameters import Parameters 

21 

22 

23class ProfileError(QMapError): 

24 pass 

25 

26 

27def _bgconfig_build_hook(config_template, file_path, home_folder, config_name): 

28 """Trick to raise an error if the config file does not exits""" 

29 raise ProfileError('Profile {} not found'.format(config_name)) 

30 

31 

32class Profile(dict): 

33 

34 """The profile contains the configuration as a dict. 

35 The only special member is the parameters, which is 

36 not part of the dict itself, but and inner object""" 

37 

38 def __init__(self, name=None): 

39 if isinstance(name, dict): # loaded from a file 

40 self.update(name) 

41 else: 

42 builtin_profiles_dir = path.join(path.dirname(__file__), 'executor', 'profiles') 

43 default_configuration = BGConfig(path.join(builtin_profiles_dir, 'default.conf'), 

44 config_spec=path.join(builtin_profiles_dir, 'profile.conf.spec'), 

45 config_namespace=CONFIGURATION_NAMESPACE, 

46 use_bgdata=False, use_env_vars=False) 

47 self.update(default_configuration) 

48 if name is None: # default 

49 profile_configuration = {'executor': 'local', 'editable_params': {}} 

50 elif path.exists(name) and path.isfile(name): # it is a path 

51 profile_configuration = ConfigObj(name) 

52 elif path.exists(path.join(builtin_profiles_dir, '{}.conf'.format(name))): # it is a built-in profile 

53 profile_configuration = BGConfig(path.join(builtin_profiles_dir, '{}.conf'.format(name)), 

54 config_spec=path.join(builtin_profiles_dir, 'profile.conf.spec'), 

55 config_namespace=CONFIGURATION_NAMESPACE, 

56 use_bgdata=False, use_env_vars=False) 

57 else: # search for the name in the config folder 

58 profile_configuration = BGConfig(None, config_name=name, 

59 config_spec=path.join(builtin_profiles_dir, 'profile.conf.spec'), 

60 build_hook=_bgconfig_build_hook, 

61 use_bgdata=False, use_env_vars=False) 

62 self.update(profile_configuration) 

63 self._validate() 

64 executor.load(self['executor'], self.get('show_usage', False)) 

65 self.parameters = Parameters(self.pop('params', {})) # get params out of the profile dict 

66 

67 def _validate(self): 

68 

69 # Validate show_usage 

70 if 'show_usage' in self: 

71 show_usage = self['show_usage'] 

72 if show_usage in [0, False, '0', 'false', 'no']: 

73 self['show_usage'] = False 

74 elif show_usage in [1, True, '1', 'true', 'yes']: 

75 self['show_usage'] = True 

76 else: 

77 raise ProfileError('Invalid value for show_usage {}'.format(show_usage)) 

78 

79 # Validate max_ungrouped 

80 if 'max_ungrouped' in self: 

81 try: 

82 max_ungrouped = int(self['max_ungrouped']) 

83 except ValueError: 

84 raise ProfileError('Invalid value for max_ungrouped {}'.format(max_ungrouped)) 

85 else: 

86 if max_ungrouped < 0: 

87 raise ProfileError('Invalid value for max_ungrouped {}'.format(max_ungrouped)) 

88 elif max_ungrouped == 0: # 0 is the same as not providing a value 

89 del self['max_ungrouped'] 

90 else: 

91 self['max_ungrouped'] = max_ungrouped
