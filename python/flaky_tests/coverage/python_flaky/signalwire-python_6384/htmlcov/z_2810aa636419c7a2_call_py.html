319 def _state_changed(self, params): 

320 self.prev_state = self.state 

321 self.state = params['call_state'] 

322 trigger(self.tag, params, suffix='stateChange') 

323 trigger(self.tag, params, suffix=self.state) 

324 if self.state == CallState.ENDED: 

325 check_id = self.id if self.id else self.tag 

326 trigger(check_id, params, suffix=CallState.ENDED) # terminate components 

327 unregister_all(self.tag) # unregister all external handlers 

328 end_reason = params.get('end_reason', '') 

329 self.failed = end_reason == DisconnectReason.ERROR 

330 self.busy = end_reason == DisconnectReason.BUSY 

331 self.calling.remove_call(self)
