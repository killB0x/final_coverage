11class DynamicEPackage(EObserver): 

12 """A DynamicEPackage gives the ability to directly handle metaclasses 

13 from a metamodel as if it were a statically generated EPackage. 

14 

15 Usage from an existing dynamic EPackage named 'myroot' that defines two 

16 EClass: 'A' and 'B' 

17 >>> from pyecore.utils import DynamicEPackage 

18 >>> MyAPI = DynamicEPackage(myroot) 

19 >>> MyAPI.A 

20 <EClass name="A"> 

21 >>> a = MyAPI.A() 

22 >>> a 

23 <pyecore.ecore.A object at 0x7f118de363c8> 

24 """
