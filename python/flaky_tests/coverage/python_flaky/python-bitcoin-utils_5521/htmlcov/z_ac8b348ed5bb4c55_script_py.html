143class Script: 

144 """Represents any script in Bitcoin 

145 

146 A Script contains just a list of OP_CODES and also knows how to serialize 

147 into bytes 

148 

149 Attributes 

150 ---------- 

151 script : list 

152 the list with all the script OP_CODES and data 

153 

154 Methods 

155 ------- 

156 to_bytes() 

157 returns a serialized byte version of the script 

158 

159 get_script() 

160 returns the list of strings that makes up this script 

161 

162 Raises 

163 ------ 

164 ValueError 

165 If string data is too large or integer is negative 

166 """ 

167 

168 def __init__(self, script): 

169 """See Script description""" 

170 

171 self.script = script 

172 

173 

174 @classmethod 

175 def copy(cls, script): 

176 """Deep copy of Script""" 

177 scripts = copy.deepcopy(script.script) 

178 return cls(scripts)

181 def _op_push_data(self, data): 

182 """Converts data to appropriate OP_PUSHDATA OP code including length 

183 

184 0x01-0x4b -> just length plus data bytes 

185 0x4c-0xff -> OP_PUSHDATA1 plus 1-byte-length plus data bytes 

186 0x0100-0xffff -> OP_PUSHDATA2 plus 2-byte-length plus data bytes 

187 0x010000-0xffffffff -> OP_PUSHDATA4 plus 4-byte-length plus data bytes 

188 

189 Also note that according to standarardness rules (BIP-62) the minimum 

190 possible PUSHDATA operator must be used! 

191 """ 

192 

193 # expects data in hexadecimal characters and converts appropriately 

194 # TODO maybe, for convenience, also accept objects for public keys, 

195 # addresses, etc. and use isinstance and convert manually 

196 data_bytes = unhexlify(data) 

197 

198 if len(data_bytes) < 0x4c: 

199 return chr(len(data_bytes)).encode() + data_bytes 

200 elif len(data_bytes) < 0xff: 

201 return b'\x4c' + chr(len(data_bytes)).encode() + data_bytes 

202 elif len(data_bytes) < 0xffff: 

203 return b'\x4d' + struct.pack('<H', len(data_bytes)) + data_bytes 

204 elif len(data_bytes) < 0xffffffff: 

205 return b'\x4e' + struct.pack('<I', len(data_bytes)) + data_bytes 

206 else: 

207 raise ValueError("Data too large. Cannot push into script")

210 def _segwit_op_push_data(self, data): 

211 # expects data in hexadecimal characters and converts to bytes with 

212 # compact_size (or var_int) length prefix. 

213 # 

214 # TODO maybe, for convenience, also accept objects for public keys, 

215 # addresses, etc. and use isinstance and convert manually 

216 data_bytes = unhexlify(data) 

217 

218 # prepend compact size lenth to data bytes 

219 compact_size_data = prepend_compact_size(data_bytes) 

220 

221 return compact_size_data

225 def _push_integer(self, integer): 

226 """Converts integer to bytes; as signed little-endian integer 

227 

228 Currently supports only positive integers 

229 """ 

230 

231 if integer < 0: 

232 raise ValueError('Integer is currently required to be positive.') 

233 

234 # bytes requires to represent the integer 

235 number_of_bytes = (integer.bit_length() + 7) // 8 

236 

237 # convert to little-endian bytes 

238 integer_bytes = integer.to_bytes(number_of_bytes, byteorder='little') 

239 

240 # if last bit is set then we need to add sign to signify positive 

241 # integer 

242 if integer & (1 << number_of_bytes*8 - 1): 

243 integer_bytes += b'\x00' 

244 

245 return self._op_push_data( hexlify(integer_bytes) )

248 def to_bytes(self, segwit = False): 

249 """Converts the script to bytes 

250 

251 If an OP code the appropriate byte is included according to: 

252 https://en.bitcoin.it/wiki/Script 

253 If not consider it data (signature, public key, public key hash, etc.) and 

254 and include with appropriate OP_PUSHDATA OP code plus length 

255 """ 

256 script_bytes = b'' 

257 for token in self.script: 

258 # add op codes directly 

259 if token in OP_CODES: 

260 script_bytes += OP_CODES[token] 

261 # if integer between 0 and 16 add the appropriate op code 

262 elif type(token) is int and token >= 0 and token <= 16: 

263 script_bytes += OP_CODES['OP_' + str(token)] 

264 # it is data, so add accordingly 

265 else: 

266 if type(token) is int: 

267 script_bytes += self._push_integer(token) 

268 else: 

269 if segwit: 

270 # probably add TxInputWitness which will know how to serialize 

271 script_bytes += self._segwit_op_push_data(token) 

272 else: 

273 script_bytes += self._op_push_data(token) 

274 

275 return script_bytes

278 def to_hex(self): 

279 """Converts the script to hexadecimal""" 

280 

281 b = self.to_bytes() 

282 return hexlify(b).decode('utf-8')

285 def get_script(self): 

286 """Returns script as array of strings""" 

287 return self.script

290 def to_p2sh_script_pub_key(self): 

291 """Converts script to p2sh scriptPubKey (locking script) 

292 

293 Calculates the hash160 (via the address) of the script and uses it to 

294 construct a P2SH script. 

295 """ 

296 

297 address = bitcoinutils.keys.P2shAddress.from_script(self) 

298 return Script(['OP_HASH160', address.to_hash160(), 'OP_EQUAL']) 

299 

300 

301 def to_p2wsh_script_pub_key(self): 

302 """Converts script to p2wsh scriptPubKey (locking script) 

303 

304 Calculates the sha256 of the script and uses it to construct a P2WSH script. 

305 """ 

306 sha256 = hashlib.sha256( self.to_bytes() ).digest() 

307 return Script(['OP_0', hexlify(sha256).decode('utf-8')])
