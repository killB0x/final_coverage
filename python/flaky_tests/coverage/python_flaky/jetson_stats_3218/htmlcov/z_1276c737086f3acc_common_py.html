89def import_os_variables(SOURCE, PATTERN): 

90 if os.path.isfile(SOURCE): 

91 logger.debug("Open source file {source}".format(source=SOURCE)) 

92 proc = sp.Popen(['bash', '-c', 'source {source} && env'.format(source=SOURCE)], stdout=sp.PIPE, stderr=sp.PIPE) 

93 # Load variables 

94 source_env = {} 

95 for tup in map(lambda s: s.decode("utf-8").strip().split('=', 1), proc.stdout): 

96 name = tup[0].strip() 

97 value = tup[1].strip() 

98 if PATTERN in name: 

99 source_env[name] = value 

100 return source_env 

101 else: 

102 logger.error("File does not exist") 

103 return {}

129def status_disk(folder="/var/"): 

130 disk = os.statvfs(folder) 

131 # Evaluate the total space in GB 

132 totalSpace = float(disk.f_bsize * disk.f_blocks) / 1024 / 1024 / 1024 

133 # Evaluate total used space in GB 

134 totalUsedSpace = float(disk.f_bsize * (disk.f_blocks - disk.f_bfree)) / 1024 / 1024 / 1024 

135 # Evaluate total available space in GB 

136 totalAvailSpace = float(disk.f_bsize * disk.f_bfree) / 1024 / 1024 / 1024 

137 # Evaluate total non super-user space in GB 

138 totalAvailSpaceNonRoot = float(disk.f_bsize * disk.f_bavail) / 1024 / 1024 / 1024 

139 return {'total': totalSpace, 

140 'used': totalUsedSpace, 

141 'available': totalAvailSpace, 

142 'available_no_root': totalAvailSpaceNonRoot 

143 }

146def get_local_interfaces(): 

147 """ Returns a dictionary of name:ip key value pairs. 

148 - Reference: 

149 * http://code.activestate.com/recipes/439093/#c1 

150 * https://gist.github.com/pklaus/289646 

151 """ 

152 # Max possible bytes for interface result. Will truncate if more than 4096 characters to describe interfaces. 

153 MAX_BYTES = 4096 

154 # We're going to make a blank byte array to operate on. This is our fill char. 

155 FILL_CHAR = b'\0' 

156 # Command defined in ioctl.h for the system operation for get iface list 

157 # Defined at https://code.woboq.org/qt5/include/bits/ioctls.h.html under 

158 # /* Socket configuration controls. */ section. 

159 SIOCGIFCONF = 0x8912 

160 # Read hostname 

161 hostname = socket.gethostname() 

162 # Make a dgram socket to use as our file descriptor that we'll operate on. 

163 sock = socket.socket(socket.AF_INET, socket.SOCK_DGRAM) 

164 # Make a byte array with our fill character. 

165 names = array.array('B', MAX_BYTES * FILL_CHAR) 

166 # Get the address of our names byte array for use in our struct. 

167 names_address, names_length = names.buffer_info() 

168 # Create a mutable byte buffer to store the data in 

169 mutable_byte_buffer = struct.pack('iL', MAX_BYTES, names_address) 

170 # mutate our mutable_byte_buffer with the results of get_iface_list. 

171 # NOTE: mutated_byte_buffer is just a reference to mutable_byte_buffer - for the sake of clarity we've defined them as 

172 # separate variables, however they are the same address space - that's how fcntl.ioctl() works since the mutate_flag=True 

173 # by default. 

174 mutated_byte_buffer = fcntl.ioctl(sock.fileno(), SIOCGIFCONF, mutable_byte_buffer) 

175 # Get our max_bytes of our mutated byte buffer that points to the names variable address space. 

176 max_bytes_out, names_address_out = struct.unpack('iL', mutated_byte_buffer) 

177 # Convert names to a bytes array - keep in mind we've mutated the names array, so now our bytes out should represent 

178 # the bytes results of the get iface list ioctl command. 

179 namestr = names.tostring() 

180 # Each entry is 40 bytes long. The first 16 bytes are the name string. 

181 # the 20-24th bytes are IP address octet strings in byte form - one for each byte. 

182 # Don't know what 17-19 are, or bytes 25:40. 

183 ip_dict = {} 

184 for i in range(0, max_bytes_out, 40): 

185 name = namestr[i: i + 16].split(FILL_CHAR, 1)[0] 

186 name = name.decode('utf-8') 

187 ip_bytes = namestr[i + 20:i + 24] 

188 full_addr = [] 

189 for netaddr in ip_bytes: 

190 if isinstance(netaddr, int): 

191 full_addr.append(str(netaddr)) 

192 elif isinstance(netaddr, str): 

193 full_addr.append(str(ord(netaddr))) 

194 ip_dict[name] = '.'.join(full_addr) 

195 # Remove loopback interface is in list 

196 if 'lo' in ip_dict: 

197 del ip_dict['lo'] 

198 return {"hostname": hostname, "interfaces": ip_dict}
