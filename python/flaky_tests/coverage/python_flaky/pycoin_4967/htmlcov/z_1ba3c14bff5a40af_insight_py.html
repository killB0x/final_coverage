37 def get_blockheader_with_transaction_hashes(self, block_hash): 

38 URL = "%s/block/%s" % (self.base_url, b2h_rev(block_hash)) 

39 r = json.loads(urlopen(URL).read().decode("utf8")) 

40 version = r.get("version") 

41 previous_block_hash = h2b_rev(r.get("previousblockhash")) 

42 merkle_root = h2b_rev(r.get("merkleroot")) 

43 timestamp = r.get("time") 

44 difficulty = int(r.get("bits"), 16) 

45 nonce = int(r.get("nonce")) 

46 tx_hashes = [h2b_rev(tx_hash) for tx_hash in r.get("tx")] 

47 blockheader = Block(version, previous_block_hash, merkle_root, timestamp, difficulty, nonce) 

48 if blockheader.hash() != block_hash: 

49 return None, None 

50 calculated_hash = merkle(tx_hashes, double_sha256) 

51 if calculated_hash != merkle_root: 

52 return None, None 

53 blockheader.height = r.get("height") 

54 return blockheader, tx_hashes

70 def spendables_for_address(self, address): 

71 """ 

72 Return a list of Spendable objects for the 

73 given bitcoin address. 

74 """ 

75 URL = "%s/addr/%s/utxo" % (self.base_url, address) 

76 r = json.loads(urlopen(URL).read().decode("utf8")) 

77 spendables = [] 

78 for u in r: 

79 coin_value = btc_to_satoshi(str(u.get("amount"))) 

80 script = h2b(u.get("scriptPubKey")) 

81 previous_hash = h2b_rev(u.get("txid")) 

82 previous_index = u.get("vout") 

83 spendables.append(Tx.Spendable(coin_value, script, previous_hash, previous_index)) 

84 return spendables

92 def send_tx(self, tx): 

93 s = io.BytesIO() 

94 tx.stream(s) 

95 tx_as_hex = b2h(s.getvalue()) 

96 data = urlencode(dict(rawtx=tx_as_hex)).encode("utf8") 

97 URL = "%s/tx/send" % self.base_url 

98 try: 

99 d = urlopen(URL, data=data).read() 

100 return d 

101 except request.HTTPError as err:

107def tx_from_json_dict(r): 

108 version = r.get("version") 

109 lock_time = r.get("locktime") 

110 txs_in = [] 

111 for vin in r.get("vin"): 

112 if "coinbase" in vin: 

113 previous_hash = b'\0' * 32 

114 script = h2b(vin.get("coinbase")) 

115 previous_index = 4294967295 

116 else: 

117 previous_hash = h2b_rev(vin.get("txid")) 

118 scriptSig = vin.get("scriptSig") 

119 if "hex" in scriptSig: 

120 script = h2b(scriptSig.get("hex")) 

121 else: 

122 script = BitcoinScriptTools.compile(scriptSig.get("asm")) 

123 previous_index = vin.get("vout") 

124 sequence = vin.get("sequence") 

125 txs_in.append(Tx.TxIn(previous_hash, previous_index, script, sequence)) 

126 txs_out = [] 

127 for vout in r.get("vout"): 

128 coin_value = btc_to_satoshi(decimal.Decimal(vout.get("value"))) 

129 script = BitcoinScriptTools.compile(vout.get("scriptPubKey").get("asm")) 

130 txs_out.append(Tx.TxOut(coin_value, script)) 

131 tx = Tx(version, txs_in, txs_out, lock_time) 

132 bh = r.get("blockhash") 

133 if bh: 

134 bh = h2b_rev(bh) 

135 tx.confirmation_block_hash = bh 

136 return tx
