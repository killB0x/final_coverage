64class ScriptStreamer(object): 

65 """ 

66 This class manages encoding and decoding instructions and data from a script. 

67 

68 Most instructions operate on existing stack data. Then there are some opcodes that 

69 push constant data onto the stack. These are data opcodes. 

70 There are three classes of data opcodes: 

71 * CONSTANT_OPCODE: opcodes that push a constant 

72 * SIZED_OPCODE: opcodes that push a constant number of subsequent bytes 

73 * VARIABLE_OPCODE: opcodes that push an encoding of the number of bytes, followed by those bytes 

74 

75 This class is also aware of "minimal encoding", that is, encoding data using 

76 the shortest data sequence possible, and can verify that scripts use it. 

77 This gives a "canonical" version of solution scripts, which can be used to 

78 help reduce the impact of malleability. 

79 """ 

80 def __init__(self, opcode_const_list, opcode_sized_list, opcode_variable_list, 

81 opcode_lookup, non_minimal_data_handler): 

82 """ 

83 :param opcode_const_list: list of ("OPCODE_NAME", const_value) pairs, where 

84 OPCODE_NAME is the name of an opcode and const_value is the value to be pushed 

85 for that opcode 

86 :param opcode_sized_list: list of ("OPCODE_NAME", data_size) pairs, where 

87 OPCODE_NAME is the name of an opcode and data_size is the size 

88 :param opcode_variable_list: list of ("OPCODE_NAME", max_data_size, enc_f, dec_f) 

89 tuples where OPCODE_NAME is an opcode, max_data_size is the maximum amount of 

90 data that can be pushed by this opcode, enc_f is the encoder for the size of the data, 

91 dec_f is the decoder for the size of the data. 

92 enc_f should have signature (bin_data) and return an integer (length) 

93 dec_f should have signature (bin_data) and return offset, data where 

94 offset is the integer count of bytes consumed and data the decoded result 

95 

96 :param opcode_lookup: dictionary with entries "OPCODE_NAME" => byte 

97 :param non_minimal_data_handler: function called when data encoded non-minimally 

98 """ 

99 

100 # build encoders 

101 const_pairs = [(opcode_lookup.get(opcode), val) for opcode, val in opcode_const_list] 

102 self.const_encoder = {v: int2byte(k) for k, v in const_pairs} 

103 

104 sized_pairs = [(opcode_lookup.get(opcode), size) for opcode, size in opcode_sized_list] 

105 self.sized_encoder = {v: make_sized_encoder(k) for k, v in sized_pairs} 

106 opcode_variable_list = sorted(opcode_variable_list, key=lambda o: o[0]) 

107 self.variable_encoder = list( 

108 (max_size, opcode_lookup.get(opcode), enc_f) 

109 for opcode, max_size, enc_f, dec_f in opcode_variable_list) 

110 

111 # build decoder 

112 

113 self.decoder = {} 

114 

115 # deal with variable data opcodes 

116 

117 min_size = 0 

118 for o, max_size, enc_f, dec_f in opcode_variable_list: 

119 self.decoder[opcode_lookup.get(o)] = make_variable_handler( 

120 dec_f, self.sized_encoder.keys(), min_size, non_minimal_data_handler) 

121 min_size = max_size + 1 

122 

123 # deal with sized data opcodes 

124 

125 self.decoder.update( 

126 {o: make_sized_handler( 

127 v, self.const_encoder.keys(), non_minimal_data_handler) for o, v in sized_pairs}) 

128 

129 # deal with constant data opcodes 

130 

131 self.decoder.update({o: make_const_handler(v) for o, v in const_pairs}) 

132 

133 self.data_opcodes = frozenset(self.decoder.keys())

135 def get_opcode(self, script, pc, verify_minimal_data=False): 

136 """ 

137 Step through the script, returning a tuple with the next opcode, the next 

138 piece of data (if the opcode represents data), the new PC, and a boolean indicated 

139 valid parsing. 

140 """ 

141 opcode = indexbytes(script, pc) 

142 decoder = self.decoder.get(opcode) 

143 # lambda s, p, verify_minimal_data: (p+1, None)) 

144 if decoder:

146 is_ok = (data is not None) 

147 else:

153 def compile_push_data(self, data): 

154 # return bytes that causes the given data to be pushed onto the stack 

155 if data in self.const_encoder:
