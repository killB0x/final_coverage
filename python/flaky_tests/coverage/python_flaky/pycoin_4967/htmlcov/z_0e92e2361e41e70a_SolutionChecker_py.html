99 def _signature_hash(self, tx_out_script, unsigned_txs_out_idx, hash_type): 

100 """ 

101 Return the canonical hash for a transaction. We need to 

102 remove references to the signature, since it's a signature 

103 of the hash before the signature is applied. 

104 

105 :param tx_out_script: the script the coins for unsigned_txs_out_idx are coming from 

106 :param unsigned_txs_out_idx: where to put the tx_out_script 

107 :param hash_type: one of SIGHASH_NONE, SIGHASH_SINGLE, SIGHASH_ALL, 

108 optionally bitwise or'ed with SIGHASH_ANYONECANPAY 

109 """ 

110 

111 # In case concatenating two scripts ends up with two codeseparators, 

112 # or an extra one at the end, this prevents all those possible incompatibilities. 

113 tx_out_script = self.delete_subscript(tx_out_script, self.ScriptTools.compile("OP_CODESEPARATOR")) 

114 

115 # blank out other inputs' signatures 

116 txs_in = [self._tx_in_for_idx(i, tx_in, tx_out_script, unsigned_txs_out_idx) 

117 for i, tx_in in enumerate(self.tx.txs_in)] 

118 txs_out = self.tx.txs_out 

119 

120 # Blank out some of the outputs 

121 if (hash_type & 0x1f) == SIGHASH_NONE: 

122 # Wildcard payee 

123 txs_out = [] 

124 

125 # Let the others update at will 

126 for i in range(len(txs_in)): 

127 if i != unsigned_txs_out_idx: 

128 txs_in[i].sequence = 0 

129 

130 elif (hash_type & 0x1f) == SIGHASH_SINGLE: 

131 # This preserves the ability to validate existing legacy 

132 # transactions which followed a buggy path in Satoshi's 

133 # original code. 

134 if unsigned_txs_out_idx >= len(txs_out): 

135 # This should probably be moved to a constant, but the 

136 # likelihood of ever getting here is already really small 

137 # and getting smaller 

138 return (1 << 248) 

139 

140 # Only lock in the txout payee at same index as txin; delete 

141 # any outputs after this one and set all outputs before this 

142 # one to "null" (where "null" means an empty script and a 

143 # value of -1) 

144 txs_out = [self.tx.TxOut(0xffffffffffffffff, b'')] * unsigned_txs_out_idx 

145 txs_out.append(self.tx.txs_out[unsigned_txs_out_idx]) 

146 

147 # Let the others update at will 

148 for i in range(len(txs_in)): 

149 if i != unsigned_txs_out_idx: 

150 txs_in[i].sequence = 0 

151 

152 # Blank out other inputs completely, not recommended for open transactions 

153 if hash_type & SIGHASH_ANYONECANPAY: 

154 txs_in = [txs_in[unsigned_txs_out_idx]] 

155 

156 tmp_tx = self.tx.__class__(self.tx.version, txs_in, txs_out, self.tx.lock_time) 

157 return from_bytes_32(tmp_tx.hash(hash_type=hash_type))
