153 def executeDialog(self, pD, data): 

154 """ 

155 Executes a dialog. 

156 

157 `pD` points to the dialog. The dialog is executed only if it is valid. 

158 

159 `data` is a data record which is set on the dialog before executing and 

160 read afterwards. If `data` is None no data will be set. 

161 

162 This method calls `execView()` to execute the dialog. The 

163 dialog is destroyed before returning from the function, so a call to 

164 delete is not necessary. `executeDialog()` returns `cmCancel` if the view 

165 is not valid, otherwise it returns the return value of `execView()`. 

166 

167 :param pD: Dialog to execute 

168 :param data: Data to set on dialog 

169 :return: tuple of `executeDate()` call and the data 

170 """ 

171 logger.info('executeDialog(%s)', pD) 

172 c = cmCancel 

173 

174 if self.validView(pD): 

175 if data: 

176 pD.setData(data) 

177 try: 

178 c = Program.desktop.execView(pD) 

179 except: 

180 logger.exception('Dialog failed.') 

181 c = cmCancel 

182 if c != cmCancel: 

183 data = pD.getData() 

184 self.destroy(pD) 

185 

186 return c, data

188 def getEvent(self, event): 

189 """ 

190 Gets an event. 

191 

192 This method collects events from the system like key events, mouse 

193 events and timer events and returns them in the `event` instance. 

194 

195 `getEvent()` first checks if `Program.putEvent()` has generated a 

196 pending event. If so, `getEvent()` returns that event. If there is no 

197 pending event, `getEvent()` calls `Screen.getEvent()`. 

198 

199 If both calls return `evNothing`, indicating that no user input is 

200 available, `getEvent()` calls `Program.idle()` to allow "background" 

201 tasks to be performed while the application is waiting for user input. 

202 

203 Before returning, `getEvent()` passes any `evKeyDown` and 

204 `evMouseDown` events to the `statusLine` for it to map into 

205 associated `evCommand` hot key events. 

206 

207 :param event: Event object to be modified 

208 """ 

209 if Program.pending.qsize(): 

210 event.setFrom(Program.pending.get()) 

211 else: 

212 Screen.getEvent(event) 

213 if event.what == evCommand: 

214 c = event.message.command 

215 if c == cmSysRepaint: 

216 self.redraw() 

217 self.clearEvent(event) 

218 elif c == cmSysResize: 

219 Screen.curX = Screen.curY = 0 

220 self.buffer = Screen.screenBuffer 

221 self.changeBounds(Rect(0, 0, 

222 Screen.screenWidth, 

223 Screen.screenHeight)) 

224 

225 self.setState(sfExposed, False) 

226 self.setState(sfExposed, True) 

227 self.redraw() 

228 self.clearEvent(event) 

229 elif c == cmSysWakeup: 

230 self.idle() 

231 self.clearEvent(event) 

232 

233 if self.statusLine: 

234 if ((event.what & evKeyDown) or 

235 (event.what & evMouseDown) and self.firstThat(self.hasMouse, event) is self.statusLine 

236 ): 

237 self.statusLine.handleEvent(event)

239 def putEvent(self, event): 

240 """ 

241 Sets a pending event. 

242 

243 Puts an event in the pending state, by storing a copy of the `event` 

244 structure in the `pending` variable, a member of `Program`. 

245 

246 Only one event is allowed to be pending. The next call to 

247 `getEvent()` will return this pending event even if there are 

248 other events in the system queue to be handled. 

249 :param event: Event to place in the queue 

250 """ 

251 e = Event(evNothing)

253 Program.pending.put(e)

255 def handleEvent(self, event): 

256 """ 

257 Standard `Program` event handler. 

258 

259 This method first checks for keyboard events. When it catches keys from 

260 Alt-1 to Alt-9 it generates an `evBroadcast` event with the `command` 

261 field equal to `cmSelectWindowNum` and the `infoPtr` field in the range 1 

262 to 9. 

263 

264 Then it calls `Group.handleEvent()`. 

265 

266 Last it checks for a `cmQuit` command in a `evCommand` event. On 

267 success it calls `Group.endModal(cmQuit)` to end the modal state. This 

268 causes the `Program.run()` method to return. In most applications 

269 this will result in program termination. 

270 

271 Method `handleEvent()` is almost always overridden to introduce handling 

272 of commands that are specific to your own application. 

273 

274 :param event: Event to be handled 

275 """ 

276 if event.what == evKeyDown: 

277 c = getAltChar(event.keyDown.keyCode) 

278 if isinstance(c, int): 

279 c = chr(c) 

280 if '1' <= c <= '9': 

281 if self.canMoveFocus(): 

282 if message(Program.desktop, evBroadcast, cmSelectWindowNum, int(c)): 

283 self.clearEvent(event) 

284 else: 

285 self.clearEvent(event) 

286 

287 super().handleEvent(event) 

288 

289 if event.what == evCommand and event.message.command == cmQuit: 

290 self.endModal(cmQuit) 

291 self.clearEvent(event)

299 def idle(self): 

300 """ 

301 Called when in idle state. 

302 

303 This method is called whenever the library is in idle state, i.e. there 

304 is not any event to serve. It allows the application to perform 

305 background tasks while waiting for user input. 

306 

307 The default idle() calls `statusLine.update()` to allow the status line 

308 to update itself according to the current help context. Then, if the 

309 command set has changed since the last call to `idle()`, an 

310 `evBroadcast` with a command value of `cmCommandSetChanged` is 

311 generated to allow views that depend on the command set to enable or 

312 disable themselves. 

313 

314 The user may redefine this method, for example, to update a clock in 

315 like the `vindauga_demo` program does. 

316 """ 

317 if self.statusLine: 

318 self.statusLine.update() 

319 

320 if View.commandSetChanged: 

321 message(self, evBroadcast, cmCommandSetChanged, None) 

322 View.commandSetChanged = False

449 def validView(self, view): 

450 """ 

451 Checks if a view is valid. 

452  

453 Returns `view` if the view pointed by `view` is valid. Otherwise returns None. 

454 

455 First, if `view` is None the call returns None. 

456 

457 Last if a call to `view.valid(cmValid)` returns False the view pointed by 

458 `view` is released and the function returns None. 

459 

460 Otherwise, the view is considered valid, and pointer `view` is returned. 

461 

462 :param view: View to test 

463 :return: View or None 

464 """ 

465 if not view: 

466 return None 

467 

468 if not view.valid(cmValid): 

469 self.destroy(view) 

470 return None 

471 return view
