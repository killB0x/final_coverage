96def _encode_to_jis_x_0201(value, errors='strict'): 

97 """Convert a unicode string into JIS X 0201 byte string using shift_jis 

98 encodings. 

99 shift_jis is a superset of jis_x_0201. So we can regard the encoded value 

100 as jis_x_0201 if it is single byte character. 

101 

102 Parameters 

103 ---------- 

104 value : text type 

105 The unicode string as presented to the user. 

106 errors : str 

107 The behavior of a character which could not be encoded. If 'strict' is 

108 passed, raise an UnicodeEncodeError. If any other value is passed, 

109 non ISO IR 14 characters are replaced by the ASCII '?'. 

110 

111 Returns 

112 ------- 

113 byte string 

114 The encoded string. If some characters in value could not be encoded to 

115 JIS X 0201, and `errors` is not set to 'strict', they are replaced to 

116 '?'. 

117 

118 Raises 

119 ------ 

120 UnicodeEncodeError 

121 If errors is set to 'strict' and `value` could not be encoded with 

122 JIS X 0201. 

123 """ 

124 

125 Encoder = codecs.getincrementalencoder('shift_jis') 

126 encoder = Encoder() 

127 

128 # If errors is not strict, this function is used as fallback. 

129 # In this case, we use only ISO IR 14 to encode given value 

130 # without escape sequence. 

131 if errors != 'strict' or value == '': 

132 encoded = b'' 

133 for c in value: 

134 try: 

135 b = encoder.encode(c) 

136 except UnicodeEncodeError as e: 

137 b = b'?' 

138 

139 if len(b) != 1 or 0x80 <= ord(b): 

140 b = b'?' 

141 encoded += b 

142 return encoded 

143 

144 encoded = encoder.encode(value[0]) 

145 if len(encoded) != 1: 

146 raise UnicodeEncodeError( 

147 'shift_jis', value, 0, len(value), 'illegal multibyte sequence') 

148 

149 msb = ord(encoded) & 0x80 # msb is 1 for ISO IR 13, 0 for ISO IR 14 

150 for i, c in enumerate(value[1:], 1): 

151 try: 

152 b = encoder.encode(c) 

153 except UnicodeEncodeError as e: 

154 e.start = i 

155 e.end = len(value) 

156 raise e 

157 if len(b) != 1 or ((ord(b) & 0x80) ^ msb) != 0: 

158 character_set = 'ISO IR 14' if msb == 0 else 'ISO IR 13' 

159 msg = 'Given character is out of {}'.format(character_set) 

160 raise UnicodeEncodeError('shift_jis', value, i, len(value), msg) 

161 encoded += b 

162 

163 return encoded

175def _encode_to_given_charset(value, character_set, errors='strict'): 

176 """Convert a unicode string into given character set. 

177 The escape sequence which is located at the end of the encoded value has 

178 to vary depending on the value 1 of SpecificCharacterSet. So we have to 

179 trim it and append the correct escape sequence manually. 

180 

181 Parameters 

182 ---------- 

183 value : text type 

184 The unicode string as presented to the user. 

185 character_set: str: 

186 Character set for result. 

187 errors : str 

188 The behavior of a character which could not be encoded. This value 

189 is passed to errors argument of str.encode(). 

190 

191 Returns 

192 ------- 

193 byte string 

194 The encoded string. If some characters in value could not be encoded to 

195 given character_set, it depends on the behavior of corresponding python 

196 encoder. 

197 

198 Raises 

199 ------ 

200 UnicodeEncodeError 

201 If errors is set to 'strict' and `value` could not be encoded with 

202 given character_set. 

203 """ 

204 

205 encoding = python_encoding[character_set] 

206 # If errors is not strict, this function is used as fallback. 

207 # So keep the tail escape sequence of encoded for backward compatibility. 

208 if errors != 'strict': 

209 return value.encode(encoding, errors=errors) 

210 

211 Encoder = codecs.getincrementalencoder(encoding) 

212 encoder = Encoder() 

213 

214 encoded = encoder.encode(value[0]) 

215 if not encoded.startswith(ENCODINGS_TO_CODES[encoding]): 

216 raise UnicodeEncodeError( 

217 encoding, value, 0, len(value), 

218 'Given character is out of {}'.format(character_set)) 

219 

220 for i, c in enumerate(value[1:], 1): 

221 try: 

222 b = encoder.encode(c) 

223 except UnicodeEncodeError as e: 

224 e.start = i 

225 e.end = len(value) 

226 raise e 

227 if b[:1] == ESC: 

228 raise UnicodeEncodeError( 

229 encoding, value, i, len(value), 

230 'Given character is out of {}'.format(character_set)) 

231 encoded += b 

232 return encoded

235def _get_escape_sequence_for_encoding(encoding, encoded=None): 

236 """ Return an escape sequence corresponding to the given encoding. If 

237 encoding is 'shift_jis', return 'ESC)I' or 'ESC(J' depending on the first 

238 byte of encoded. 

239 

240 Parameters 

241 ---------- 

242 encoding : str 

243 An encoding is used to specify an escape sequence. 

244 encoded : bytes 

245 The encoded value is used to choose an escape sequence if encoding is 

246 'shift_jis'. 

247 

248 Returns 

249 ------- 

250 string 

251 Escape sequence for encoded value. 

252 """ 

253 

254 ESC_ISO_IR_14 = ESC + b'(J' 

255 ESC_ISO_IR_13 = ESC + b')I' 

256 

257 if encoding == 'shift_jis': 

258 if encoded is None: 

259 return ESC_ISO_IR_14 

260 

261 first_byte = encoded[0] 

262 if 0x80 <= first_byte: 

263 return ESC_ISO_IR_13 

264 

265 return ESC_ISO_IR_14 

266 return ENCODINGS_TO_CODES.get(encoding, b'')

280def decode_string(value, encodings, delimiters): 

281 """Decode an encoded byte `value` into a unicode string using `encodings`. 

282 

283 .. versionadded:: 1.2 

284 

285 Parameters 

286 ---------- 

287 value : bytes 

288 The encoded byte string in the DICOM element value. Should be 

289 :class:`bytes` 

290 encodings : list of str 

291 The encodings needed to decode the string as a list of Python 

292 encodings, converted from the encodings in (0008,0005) *Specific 

293 Character Set*. 

294 delimiters : set of int 

295 A set of characters or character codes, each of which resets the 

296 encoding in `value`. 

297 

298 Returns 

299 ------- 

300 str 

301 The decoded unicode string. If the value could not be decoded, 

302 and :func:`enforce_valid_values<pydicom.config.enforce_valid_values>` 

303 is ``False``, a warning is issued, and `value` is decoded using the 

304 first encoding with replacement characters, resulting in data loss. 

305 

306 Raises 

307 ------ 

308 UnicodeDecodeError 

309 If :func:`enforce_valid_values<pydicom.config.enforce_valid_values>` 

310 is ``True`` and `value` could not be decoded with the given encodings. 

311 """ 

312 # shortcut for the common case - no escape sequences present 

313 if ESC not in value: 

314 first_encoding = encodings[0] 

315 try: 

316 return value.decode(first_encoding) 

317 except LookupError: 

318 if config.enforce_valid_values: 

319 raise 

320 warnings.warn("Unknown encoding '{}' - " 

321 "using default encoding instead" 

322 .format(first_encoding)) 

323 first_encoding = default_encoding 

324 return value.decode(first_encoding) 

325 except UnicodeError: 

326 if config.enforce_valid_values: 

327 raise 

328 warnings.warn("Failed to decode byte string with encoding '{}' - " 

329 "using replacement characters in decoded " 

330 "string".format(first_encoding)) 

331 return value.decode(first_encoding, errors='replace') 

332 

333 # Each part of the value that starts with an escape sequence is decoded 

334 # separately. If it starts with an escape sequence, the 

335 # corresponding encoding is used, otherwise (e.g. the first part if it 

336 # does not start with an escape sequence) the first encoding. 

337 # See PS3.5, 6.1.2.4 and 6.1.2.5 for the use of code extensions. 

338 # 

339 # The following regex splits the value into these parts, by matching 

340 # the substring until the first escape character, and subsequent 

341 # substrings starting with an escape character. 

342 regex = b'(^[^\x1b]+|[\x1b][^\x1b]*)' 

343 fragments = re.findall(regex, value) 

344 

345 # decode each byte string fragment with it's corresponding encoding 

346 # and join them all together 

347 return ''.join([_decode_fragment(fragment, encodings, delimiters) 

348 for fragment in fragments])

351def _decode_fragment(byte_str, encodings, delimiters): 

352 """Decode a byte string encoded with a single encoding. 

353 If `byte_str` starts with an escape sequence, the encoding corresponding 

354 to this sequence is used for decoding if present in `encodings`, 

355 otherwise the first value in encodings. 

356 If a delimiter occurs inside the string, it resets the encoding to the 

357 first encoding in case of single-byte encodings. 

358 

359 Parameters 

360 ---------- 

361 byte_str : bytes 

362 The raw string to be decoded. 

363 encodings: list of str 

364 The list of Python encodings as converted from the values in the 

365 Specific Character Set tag. 

366 delimiters: set of int 

367 A set of characters or character codes, each of which resets the 

368 encoding in `byte_str`. 

369 

370 Returns 

371 ------- 

372 text type 

373 The decoded unicode string. If the value could not be decoded, 

374 and `config.enforce_valid_values` is not set, a warning is issued, 

375 and the value is decoded using the first encoding with replacement 

376 characters, resulting in data loss. 

377 

378 Raises 

379 ------ 

380 UnicodeDecodeError 

381 If `config.enforce_valid_values` is set and `value` could not be 

382 decoded with the given encodings. 

383 

384 References 

385 ---------- 

386 * DICOM Standard, Part 5, 

387 :dcm:`Sections 6.1.2.4<part05/chapter_6.html#sect_6.1.2.4>` and 

388 :dcm:`6.1.2.5<part05/chapter_6.html#sect_6.1.2.5>` 

389 * DICOM Standard, Part 3, 

390 :dcm:`Annex C.12.1.1.2<part03/sect_C.12.html#sect_C.12.1.1.2>` 

391 """ 

392 try: 

393 if byte_str.startswith(ESC): 

394 return _decode_escaped_fragment(byte_str, encodings, delimiters) 

395 # no escape sequence - use first encoding 

396 return byte_str.decode(encodings[0]) 

397 except UnicodeError: 

398 if config.enforce_valid_values: 

399 raise 

400 warnings.warn("Failed to decode byte string with encodings: {} - " 

401 "using replacement characters in decoded " 

402 "string".format(', '.join(encodings))) 

403 return byte_str.decode(encodings[0], errors='replace')

406def _decode_escaped_fragment(byte_str, encodings, delimiters): 

407 """Decodes a byte string starting with an escape sequence. 

408 See `_decode_fragment` for parameter description and more information. 

409 """ 

410 # all 4-character escape codes start with one of two character sets 

411 seq_length = 4 if byte_str.startswith((b'\x1b$(', b'\x1b$)')) else 3 

412 encoding = CODES_TO_ENCODINGS.get(byte_str[:seq_length], '') 

413 if encoding in encodings or encoding == default_encoding: 

414 if encoding in handled_encodings: 

415 # Python strips the escape sequences for this encoding. 

416 # Any delimiters must be handled correctly by `byte_str`. 

417 return byte_str.decode(encoding) 

418 else: 

419 # Python doesn't know about the escape sequence - 

420 # we have to strip it before decoding 

421 byte_str = byte_str[seq_length:] 

422 

423 # If a delimiter occurs in the string, it resets the encoding. 

424 # The following returns the first occurrence of a delimiter in 

425 # the byte string, or None if it does not contain any. 

426 index = next((index for index, ch in enumerate(byte_str) 

427 if ch in delimiters), None) 

428 if index is not None: 

429 # the part of the string after the first delimiter 

430 # is decoded with the first encoding 

431 return (byte_str[:index].decode(encoding) + 

432 byte_str[index:].decode(encodings[0])) 

433 # No delimiter - use the encoding defined by the escape code 

434 return byte_str.decode(encoding) 

435 

436 # unknown escape code - use first encoding 

437 msg = "Found unknown escape sequence in encoded string value" 

438 if config.enforce_valid_values: 

439 raise ValueError(msg) 

440 warnings.warn(msg + " - using encoding {}".format(encodings[0])) 

441 return byte_str.decode(encodings[0], errors='replace')

444def encode_string(value, encodings): 

445 """Encode a unicode string `value` into :class:`bytes` using `encodings`. 

446 

447 .. versionadded:: 1.2 

448 

449 Parameters 

450 ---------- 

451 value : str 

452 The unicode string as presented to the user. 

453 encodings : list of str 

454 The encodings needed to encode the string as a list of Python 

455 encodings, converted from the encodings in (0008,0005) *Specific 

456 Character Set*. 

457 

458 Returns 

459 ------- 

460 bytes 

461 The encoded string. If `value` could not be encoded with any of 

462 the given encodings, and 

463 :func:`enforce_valid_values<pydicom.config.enforce_valid_values>` is 

464 ``False``, a warning is issued, and `value` is encoded using the first 

465 encoding with replacement characters, resulting in data loss. 

466 

467 Raises 

468 ------ 

469 UnicodeEncodeError 

470 If :func:`enforce_valid_values<pydicom.config.enforce_valid_values>` 

471 is ``True`` and `value` could not be encoded with the supplied 

472 encodings. 

473 """ 

474 for i, encoding in enumerate(encodings): 

475 try: 

476 encoded = _encode_string_impl(value, encoding) 

477 

478 if i > 0 and encoding not in handled_encodings: 

479 escape_sequence = _get_escape_sequence_for_encoding( 

480 encoding, encoded=encoded) 

481 encoded = escape_sequence + encoded 

482 if encoding in need_tail_escape_sequence_encodings: 

483 encoded += _get_escape_sequence_for_encoding(encodings[0]) 

484 return encoded 

485 except UnicodeError: 

486 continue 

487 else: 

488 # if we have more than one encoding, we retry encoding by splitting 

489 # `value` into chunks that can be encoded with one of the encodings 

490 if len(encodings) > 1: 

491 try: 

492 return _encode_string_parts(value, encodings) 

493 except ValueError: 

494 pass 

495 # all attempts failed - raise or warn and encode with replacement 

496 # characters 

497 if config.enforce_valid_values: 

498 # force raising a valid UnicodeEncodeError 

499 value.encode(encodings[0]) 

500 

501 warnings.warn("Failed to encode value with encodings: {} - using " 

502 "replacement characters in encoded string" 

503 .format(', '.join(encodings))) 

504 return _encode_string_impl(value, encodings[0], errors='replace')

507def _encode_string_parts(value, encodings): 

508 """Convert a unicode string into a byte string using the given 

509 list of encodings. 

510 This is invoked if `encode_string` failed to encode `value` with a single 

511 encoding. We try instead to use different encodings for different parts 

512 of the string, using the encoding that can encode the longest part of 

513 the rest of the string as we go along. 

514 

515 Parameters 

516 ---------- 

517 value : text type 

518 The unicode string as presented to the user. 

519 encodings : list 

520 The encodings needed to encode the string as a list of Python 

521 encodings, converted from the encodings in Specific Character Set. 

522 

523 Returns 

524 ------- 

525 byte string 

526 The encoded string, including the escape sequences needed to switch 

527 between different encodings. 

528 

529 Raises 

530 ------ 

531 ValueError 

532 If `value` could not be encoded with the given encodings. 

533 

534 """ 

535 encoded = bytearray() 

536 unencoded_part = value 

537 best_encoding = None 

538 while unencoded_part: 

539 # find the encoding that can encode the longest part of the rest 

540 # of the string still to be encoded 

541 max_index = 0 

542 for encoding in encodings: 

543 try: 

544 _encode_string_impl(unencoded_part, encoding) 

545 # if we get here, the whole rest of the value can be encoded 

546 best_encoding = encoding 

547 max_index = len(unencoded_part) 

548 break 

549 except UnicodeError as e: 

550 if e.start > max_index: 

551 # e.start is the index of first character failed to encode 

552 max_index = e.start 

553 best_encoding = encoding 

554 # none of the given encodings can encode the first character - give up 

555 if max_index == 0: 

556 raise ValueError("None of the given encodings can encode the " 

557 "first character") 

558 

559 # encode the part that can be encoded with the found encoding 

560 encoded_part = _encode_string_impl(unencoded_part[:max_index], 

561 best_encoding) 

562 if best_encoding not in handled_encodings: 

563 encoded += _get_escape_sequence_for_encoding( 

564 best_encoding, encoded=encoded_part) 

565 encoded += encoded_part 

566 # set remaining unencoded part of the string and handle that 

567 unencoded_part = unencoded_part[max_index:] 

568 # unencoded_part is empty - we are done, return the encoded string 

569 if best_encoding in need_tail_escape_sequence_encodings: 

570 encoded += _get_escape_sequence_for_encoding(encodings[0]) 

571 return bytes(encoded)

574def _encode_string_impl(value, encoding, errors='strict'): 

575 """Convert a unicode string into a byte string. If given encoding is in 

576 custom_encoders, use a corresponding custom_encoder. If given encoding 

577 is not in custom_encoders, use a corresponding python handled encoder. 

578 """ 

579 if encoding in custom_encoders: 

580 return custom_encoders[encoding](value, errors=errors) 

581 else: 

582 return value.encode(encoding, errors=errors)

586# default is ISO-IR 6 G0, equiv to common chr set of ISO 8859 (PS3.5 6.1.2.1) 

587# (0008,0005) value 1 can *replace* the default encoding... 

588# for VRs of SH, LO, ST, LT, PN and UT (PS3.5 6.1.2.3)... 

589# with a single-byte character encoding 

590# if (0008,0005) is multi-valued, then value 1 (or default if blank)... 

591# is used until code extension escape sequence is hit, 

592# which can be at start of string, or after CR/LF, FF, or 

593# in Person Name PN, after ^ or = 

594# NOTE also that 7.5.3 SEQUENCE INHERITANCE states that if (0008,0005) 

595# is not present in a sequence item then it is inherited from its parent.

598def convert_encodings(encodings): 

599 """Convert DICOM `encodings` into corresponding Python encodings. 

600 

601 Handles some common spelling mistakes and issues a warning in this case. 

602 

603 Handles stand-alone encodings: if they are the first encodings, 

604 additional encodings are ignored, if they are not the first encoding, 

605 they are ignored. In both cases, a warning is issued. 

606 

607 Invalid encodings are replaced with the default encoding with a 

608 respective warning issued, if 

609 :func:`enforce_valid_values<pydicom.config.enforce_valid_values>` is 

610 ``False``, otherwise an exception is raised. 

611 

612 Parameters 

613 ---------- 

614 encodings : list of str 

615 The list of encodings as read from (0008,0005) *Specific Character 

616 Set*. 

617 

618 Returns 

619 ------- 

620 list of str 

621 A :class:`list` of Python encodings corresponding to the DICOM 

622 encodings. If an encoding is already a Python encoding, it is returned 

623 unchanged. Encodings with common spelling errors are replaced by the 

624 correct encoding, and invalid encodings are replaced with the default 

625 encoding if 

626 :func:`enforce_valid_values<pydicom.config.enforce_valid_values>` is 

627 ``False``. 

628 

629 Raises 

630 ------ 

631 LookupError 

632 If `encodings` contains a value that could not be converted and 

633 :func:`enforce_valid_values<pydicom.config.enforce_valid_values>` is 

634 ``True``. 

635 """ 

636 

637 # If a list if passed, we don't want to modify the list in place so copy it 

638 encodings = encodings[:] 

639 

640 if isinstance(encodings, str): 

641 encodings = [encodings] 

642 elif not encodings[0]: 

643 encodings[0] = 'ISO_IR 6' 

644 

645 py_encodings = [] 

646 for encoding in encodings: 

647 try: 

648 py_encodings.append(python_encoding[encoding]) 

649 except KeyError: 

650 py_encodings.append( 

651 _python_encoding_for_corrected_encoding(encoding)) 

652 

653 if len(encodings) > 1: 

654 py_encodings = _handle_illegal_standalone_encodings(encodings, 

655 py_encodings) 

656 return py_encodings

659def _python_encoding_for_corrected_encoding(encoding): 

660 """Try to replace the given invalid encoding with a valid encoding by 

661 checking for common spelling errors, and return the correct Python 

662 encoding for that encoding. Otherwise check if the 

663 encoding is already a valid Python encoding, and return that. If both 

664 attempts fail, return the default encoding. 

665 Issue a warning for the invalid encoding except for the case where it is 

666 already converted. 

667 """ 

668 # standard encodings 

669 patched = None 

670 if re.match('^ISO[^_]IR', encoding) is not None: 

671 patched = 'ISO_IR' + encoding[6:] 

672 # encodings with code extensions 

673 elif re.match('^(?=ISO.2022.IR.)(?!ISO 2022 IR )', 

674 encoding) is not None: 

675 patched = 'ISO 2022 IR ' + encoding[12:] 

676 

677 if patched: 

678 # handle encoding patched for common spelling errors 

679 try: 

680 py_encoding = python_encoding[patched] 

681 _warn_about_invalid_encoding(encoding, patched) 

682 return py_encoding 

683 except KeyError: 

684 _warn_about_invalid_encoding(encoding) 

685 return default_encoding 

686 

687 # fallback: assume that it is already a python encoding 

688 try: 

689 codecs.lookup(encoding) 

690 return encoding 

691 except LookupError: 

692 _warn_about_invalid_encoding(encoding) 

693 return default_encoding

696def _warn_about_invalid_encoding(encoding, patched_encoding=None): 

697 """Issue a warning for the given invalid encoding. 

698 If patched_encoding is given, it is mentioned as the 

699 replacement encoding, other the default encoding. 

700 If no replacement encoding is given, and config.enforce_valid_values 

701 is set, LookupError is raised. 

702 """ 

703 if patched_encoding is None: 

704 if config.enforce_valid_values: 

705 raise LookupError( 

706 "Unknown encoding '{}'".format(encoding)) 

707 msg = ("Unknown encoding '{}' - using default encoding " 

708 "instead".format(encoding)) 

709 else: 

710 msg = ("Incorrect value for Specific Character Set " 

711 "'{}' - assuming '{}'".format(encoding, patched_encoding)) 

712 warnings.warn(msg, stacklevel=2)

715def _handle_illegal_standalone_encodings(encodings, py_encodings): 

716 """Check for stand-alone encodings in multi-valued encodings. 

717 If the first encoding is a stand-alone encoding, the rest of the 

718 encodings is removed. If any other encoding is a stand-alone encoding, 

719 it is removed from the encodings. 

720 """ 

721 if encodings[0] in STAND_ALONE_ENCODINGS: 

722 warnings.warn("Value '{}' for Specific Character Set does not " 

723 "allow code extensions, ignoring: {}" 

724 .format(encodings[0], ', '.join(encodings[1:])), 

725 stacklevel=2) 

726 py_encodings = py_encodings[:1] 

727 else: 

728 for i, encoding in reversed(list(enumerate(encodings[1:]))): 

729 if encoding in STAND_ALONE_ENCODINGS: 

730 warnings.warn( 

731 "Value '{}' cannot be used as code extension, " 

732 "ignoring it".format(encoding), 

733 stacklevel=2) 

734 del py_encodings[i + 1] 

735 return py_encodings

738def decode_element(data_element, dicom_character_set): 

739 """Apply the DICOM character encoding to a data element 

740 

741 Parameters 

742 ---------- 

743 data_element : dataelem.DataElement 

744 The :class:`DataElement<pydicom.dataelem.DataElement>` instance 

745 containing an encoded byte string value to decode. 

746 dicom_character_set : str or list of str or None 

747 The value of (0008,0005) *Specific Character Set*, which may be a 

748 single value, a multiple value (code extension), or may also be ``''`` 

749 or ``None``, in which case ``'ISO_IR 6'`` will be used. 

750 """ 

751 if data_element.is_empty: 

752 return data_element.empty_value 

753 if not dicom_character_set: 

754 dicom_character_set = ['ISO_IR 6'] 

755 

756 encodings = convert_encodings(dicom_character_set) 

757 

758 # decode the string value to unicode 

759 # PN is special case as may have 3 components with different chr sets 

760 if data_element.VR == "PN": 

761 if data_element.VM <= 1: 

762 data_element.value = data_element.value.decode(encodings) 

763 else: 

764 data_element.value = [ 

765 val.decode(encodings) for val in data_element.value 

766 ] 

767 

768 if data_element.VR in text_VRs: 

769 # You can't re-decode unicode (string literals in py3) 

770 if data_element.VM == 1: 

771 if isinstance(data_element.value, str): 

772 return 

773 data_element.value = decode_string(data_element.value, encodings, 

774 TEXT_VR_DELIMS) 

775 else: 

776 

777 output = list() 

778 

779 for value in data_element.value: 

780 if isinstance(value, str): 

781 output.append(value) 

782 else: 

783 output.append(decode_string(value, encodings, 

784 TEXT_VR_DELIMS)) 

785 

786 data_element.value = output
