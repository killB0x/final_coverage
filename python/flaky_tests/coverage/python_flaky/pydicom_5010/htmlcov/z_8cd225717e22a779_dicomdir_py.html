88 def parse_records(self): 

89 """Build the hierarchy of given directory records, and structure

92 This is intended for initial read of file only,

96 # Define a helper function for organizing the records 

97 def get_siblings(record, map_offset_to_record): 

98 """Return a list of all siblings of the given directory record, 

99 including itself. 

100 """ 

101 sibling_list = [record] 

102 current_record = record 

103 while ('OffsetOfTheNextDirectoryRecord' in current_record and 

104 current_record.OffsetOfTheNextDirectoryRecord): 

105 offset_of_next = current_record.OffsetOfTheNextDirectoryRecord 

106 sibling = map_offset_to_record[offset_of_next] 

107 sibling_list.append(sibling) 

108 current_record = sibling 

109 return sibling_list

111 # Build the mapping from file offsets to records 

112 records = self.DirectoryRecordSequence 

113 if not records: 

114 return

116 map_offset_to_record = {} 

117 for record in records: 

118 offset = record.seq_item_tell 

119 map_offset_to_record[offset] = record 

120 # logging.debug("Record offsets: " + map_offset_to_record.keys()) 

121 

122 # Find the children of each record 

123 for record in records: 

124 record.children = []

126 child_offset = (record.

129 child = map_offset_to_record[child_offset] 

130 record.children = get_siblings(child, map_offset_to_record)
