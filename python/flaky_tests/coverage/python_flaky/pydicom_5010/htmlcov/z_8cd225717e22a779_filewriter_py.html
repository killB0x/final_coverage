30def _correct_ambiguous_vr_element(elem, ds, is_little_endian): 

31 """Implementation for `correct_ambiguous_vr_element`. 

32 See `correct_ambiguous_vr_element` for description. 

33 """ 

34 # 'OB or OW': 7fe0,0010 PixelData 

35 if elem.tag == 0x7fe00010: 

36 # Compressed Pixel Data 

37 # PS3.5 Annex A.4 

38 # If encapsulated, VR is OB and length is undefined 

39 if elem.is_undefined_length: 

40 elem.VR = 'OB' 

41 elif ds.is_implicit_VR: 

42 # Non-compressed Pixel Data - Implicit Little Endian 

43 # PS3.5 Annex A1: VR is always OW 

44 elem.VR = 'OW' 

45 else: 

46 # Non-compressed Pixel Data - Explicit VR 

47 # PS3.5 Annex A.2: 

48 # If BitsAllocated is > 8 then VR shall be OW, 

49 # else may be OB or OW. 

50 # If we get here, the data has not been written before 

51 # or has been converted from Implicit Little Endian, 

52 # so we default to OB for BitsAllocated 1 or 8 

53 elem.VR = 'OW' if ds.BitsAllocated > 8 else 'OB' 

54 

55 # 'US or SS' and dependent on PixelRepresentation 

56 # (0018,9810) Zero Velocity Pixel Value 

57 # (0022,1452) Mapped Pixel Value 

58 # (0028,0104)/(0028,0105) Smallest/Largest Valid Pixel Value 

59 # (0028,0106)/(0028,0107) Smallest/Largest Image Pixel Value 

60 # (0028,0108)/(0028,0109) Smallest/Largest Pixel Value in Series 

61 # (0028,0110)/(0028,0111) Smallest/Largest Image Pixel Value in Plane 

62 # (0028,0120) Pixel Padding Value 

63 # (0028,0121) Pixel Padding Range Limit 

64 # (0028,1101-1103) Red/Green/Blue Palette Color Lookup Table Descriptor 

65 # (0028,3002) LUT Descriptor 

66 # (0040,9216)/(0040,9211) Real World Value First/Last Value Mapped 

67 # (0060,3004)/(0060,3006) Histogram First/Last Bin Value 

68 elif elem.tag in [ 

69 0x00189810, 0x00221452, 0x00280104, 0x00280105, 0x00280106, 

70 0x00280107, 0x00280108, 0x00280109, 0x00280110, 0x00280111, 

71 0x00280120, 0x00280121, 0x00281101, 0x00281102, 0x00281103, 

72 0x00283002, 0x00409211, 0x00409216, 0x00603004, 0x00603006 

73 ]: 

74 # US if PixelRepresentation value is 0x0000, else SS 

75 # For references, see the list at 

76 # https://github.com/darcymason/pydicom/pull/298 

77 # PixelRepresentation is usually set in the root dataset 

78 while 'PixelRepresentation' not in ds and ds.parent and ds.parent(): 

79 ds = ds.parent() 

80 # if no pixel data is present, none if these tags is used, 

81 # so we can just ignore a missing PixelRepresentation in this case 

82 if ('PixelRepresentation' not in ds and 'PixelData' not in ds or 

83 ds.PixelRepresentation == 0): 

84 elem.VR = 'US' 

85 byte_type = 'H' 

86 else: 

87 elem.VR = 'SS' 

88 byte_type = 'h' 

89 

90 # Need to handle type check for elements with VM > 1 

91 elem_value = elem.value if elem.VM == 1 else elem.value[0] 

92 if not isinstance(elem_value, int): 

93 elem.value = convert_numbers( 

94 elem.value, is_little_endian, byte_type 

95 ) 

96 

97 # 'OB or OW' and dependent on WaveformBitsAllocated 

98 # (5400, 0110) Channel Minimum Value 

99 # (5400, 0112) Channel Maximum Value 

100 # (5400, 100A) Waveform Padding Data 

101 # (5400, 1010) Waveform Data 

102 elif elem.tag in [0x54000110, 0x54000112, 0x5400100A, 0x54001010]: 

103 # If WaveformBitsAllocated is > 8 then OW, otherwise may be 

104 # OB or OW. 

105 # See PS3.3 C.10.9.1. 

106 if ds.is_implicit_VR: 

107 elem.VR = 'OW' 

108 else: 

109 elem.VR = 'OW' if ds.WaveformBitsAllocated > 8 else 'OB' 

110 

111 # 'US or OW': 0028,3006 LUTData 

112 elif elem.tag == 0x00283006: 

113 # First value in LUT Descriptor is how many values in 

114 # LUTData, if there's only one value then must be US 

115 # As per PS3.3 C.11.1.1.1 

116 if ds.LUTDescriptor[0] == 1: 

117 elem.VR = 'US' 

118 elem_value = elem.value if elem.VM == 1 else elem.value[0] 

119 if not isinstance(elem_value, int): 

120 elem.value = convert_numbers(elem.value, is_little_endian, 'H') 

121 else: 

122 elem.VR = 'OW' 

123 

124 # 'OB or OW': 60xx,3000 OverlayData and dependent on Transfer Syntax 

125 elif (elem.tag.group in range(0x6000, 0x601F, 2) 

126 and elem.tag.elem == 0x3000): 

127 # Implicit VR must be OW, explicit VR may be OB or OW 

128 # as per PS3.5 Section 8.1.2 and Annex A 

129 elem.VR = 'OW' 

130 

131 return elem

134def correct_ambiguous_vr_element(elem, ds, is_little_endian): 

135 """Attempt to correct the ambiguous VR element `elem`. 

136 

137 When it's not possible to correct the VR, the element will be returned 

138 unchanged. Currently the only ambiguous VR elements not corrected for are 

139 all retired or part of DICONDE. 

140 

141 If the VR is corrected and is 'US' or 'SS' then the value will be updated 

142 using the :func:`~pydicom.values.convert_numbers` function. 

143 

144 Parameters 

145 ---------- 

146 elem : dataelem.DataElement 

147 The element with an ambiguous VR. 

148 ds : dataset.Dataset 

149 The dataset containing `elem`. 

150 is_little_endian : bool 

151 The byte ordering of the values in the dataset. 

152 

153 Returns 

154 ------- 

155 dataelem.DataElement 

156 The corrected element 

157 """ 

158 if 'or' in elem.VR: 

159 # convert raw data elements before handling them 

160 if elem.is_raw: 

161 elem = DataElement_from_raw(elem) 

162 ds.__setitem__(elem.tag, elem) 

163 

164 try: 

165 _correct_ambiguous_vr_element(elem, ds, is_little_endian) 

166 except AttributeError as e: 

167 raise AttributeError( 

168 f"Failed to resolve ambiguous VR for tag {elem.tag}: " + str(e) 

169 ) 

170 

171 return elem

174def correct_ambiguous_vr(ds, is_little_endian): 

175 """Iterate through `ds` correcting ambiguous VR elements (if possible). 

176 

177 When it's not possible to correct the VR, the element will be returned 

178 unchanged. Currently the only ambiguous VR elements not corrected for are 

179 all retired or part of DICONDE. 

180 

181 If the VR is corrected and is 'US' or 'SS' then the value will be updated 

182 using the :func:`~pydicom.values.convert_numbers` function. 

183 

184 Parameters 

185 ---------- 

186 ds : pydicom.dataset.Dataset 

187 The dataset containing ambiguous VR elements. 

188 is_little_endian : bool 

189 The byte ordering of the values in the dataset. 

190 

191 Returns 

192 ------- 

193 ds : dataset.Dataset 

194 The corrected dataset 

195 

196 Raises 

197 ------ 

198 AttributeError 

199 If a tag is missing in `ds` that is required to resolve the ambiguity. 

200 """ 

201 # Iterate through the elements 

202 for elem in ds: 

203 # raw data element sequences can be written as they are, because we 

204 # have ensured that the transfer syntax has not changed at this point 

205 if elem.VR == 'SQ': 

206 for item in elem: 

207 correct_ambiguous_vr(item, is_little_endian) 

208 elif 'or' in elem.VR: 

209 correct_ambiguous_vr_element(elem, ds, is_little_endian) 

210 return ds

213def write_numbers(fp, data_element, struct_format): 

214 """Write a "value" of type struct_format from the dicom file. 

215 

216 "Value" can be more than one number. 

217 

218 Parameters 

219 ---------- 

220 fp : file-like 

221 The file-like to write the encoded data to. 

222 data_element : dataelem.DataElement 

223 The element to encode. 

224 struct_format : str 

225 The character format as used by the struct module. 

226 """ 

227 endianChar = '><' [fp.is_little_endian] 

228 value = data_element.value 

229 if value == "": 

230 return # don't need to write anything for empty string 

231 

232 format_string = endianChar + struct_format 

233 try: 

234 try: 

235 value.append # works only if list, not if string or number 

236 except AttributeError: # is a single value - the usual case 

237 fp.write(pack(format_string, value)) 

238 else: 

239 for val in value: 

240 fp.write(pack(format_string, val)) 

241 except Exception as e: 

242 raise IOError( 

243 f"{str(e)}\nfor data_element:\n{str(data_element)}" 

244 )

450def write_data_element(fp, data_element, encodings=None): 

451 """Write the data_element to file fp according to 

452 dicom media storage rules. 

453 """ 

454 # Write element's tag 

455 fp.write_tag(data_element.tag) 

456 

457 # write into a buffer to avoid seeking back which can be expansive 

458 buffer = DicomBytesIO() 

459 buffer.is_little_endian = fp.is_little_endian 

460 buffer.is_implicit_VR = fp.is_implicit_VR 

461 

462 VR = data_element.VR 

463 if not fp.is_implicit_VR and len(VR) != 2: 

464 msg = ( 

465 f"Cannot write ambiguous VR of '{VR}' for data element with " 

466 f"tag {repr(data_element.tag)}.\nSet the correct VR before " 

467 f"writing, or use an implicit VR transfer syntax" 

468 ) 

469 raise ValueError(msg) 

470 

471 if data_element.is_raw: 

472 # raw data element values can be written as they are 

473 buffer.write(data_element.value) 

474 is_undefined_length = data_element.length == 0xFFFFFFFF 

475 else: 

476 if VR not in writers: 

477 raise NotImplementedError( 

478 f"write_data_element: unknown Value Representation '{VR}'" 

479 ) 

480 

481 encodings = encodings or [default_encoding] 

482 encodings = convert_encodings(encodings) 

483 writer_function, writer_param = writers[VR] 

484 is_undefined_length = data_element.is_undefined_length 

485 if not data_element.is_empty: 

486 if VR in text_VRs or VR in ('PN', 'SQ'): 

487 writer_function(buffer, data_element, encodings=encodings) 

488 else: 

489 # Many numeric types use the same writer but with 

490 # numeric format parameter 

491 if writer_param is not None: 

492 writer_function(buffer, data_element, writer_param) 

493 else: 

494 writer_function(buffer, data_element) 

495 

496 # valid pixel data with undefined length shall contain encapsulated 

497 # data, e.g. sequence items - raise ValueError otherwise (see #238) 

498 if is_undefined_length and data_element.tag == 0x7fe00010: 

499 encap_item = b'\xfe\xff\x00\xe0' 

500 if not fp.is_little_endian: 

501 # Non-conformant endianness 

502 encap_item = b'\xff\xfe\xe0\x00' 

503 if not data_element.value.startswith(encap_item): 

504 raise ValueError( 

505 "(7FE0,0010) Pixel Data has an undefined length indicating " 

506 "that it's compressed, but the data isn't encapsulated as " 

507 "required. See pydicom.encaps.encapsulate() for more " 

508 "information" 

509 ) 

510 

511 value_length = buffer.tell() 

512 if (not fp.is_implicit_VR and VR not in extra_length_VRs and 

513 not is_undefined_length and value_length > 0xffff): 

514 # see PS 3.5, section 6.2.2 for handling of this case 

515 msg = ( 

516 f"The value for the data element {data_element.tag} exceeds the " 

517 f"size of 64 kByte and cannot be written in an explicit transfer " 

518 f"syntax. The data element VR is changed from '{VR}' to 'UN' " 

519 f"to allow saving the data." 

520 ) 

521 warnings.warn(msg) 

522 VR = 'UN' 

523 

524 # write the VR for explicit transfer syntax 

525 if not fp.is_implicit_VR: 

526 fp.write(bytes(VR, default_encoding)) 

527 

528 if VR in extra_length_VRs: 

529 fp.write_US(0) # reserved 2 bytes 

530 

531 if (not fp.is_implicit_VR and VR not in extra_length_VRs and 

532 not is_undefined_length): 

533 fp.write_US(value_length) # Explicit VR length field is 2 bytes 

534 else: 

535 # write the proper length of the data_element in the length slot, 

536 # unless is SQ with undefined length. 

537 fp.write_UL(0xFFFFFFFF if is_undefined_length else value_length) 

538 

539 fp.write(buffer.getvalue()) 

540 if is_undefined_length: 

541 fp.write_tag(SequenceDelimiterTag) 

542 fp.write_UL(0) # 4-byte 'length' of delimiter data item 

543

545def write_dataset(fp, dataset, parent_encoding=default_encoding): 

546 """Write a Dataset dictionary to the file. Return the total length written. 

547 """ 

548 _harmonize_properties(dataset, fp) 

549 

550 if None in (dataset.is_little_endian, dataset.is_implicit_VR): 

551 name = dataset.__class__.__name__ 

552 raise AttributeError( 

553 f"'{name}.is_little_endian' and '{name}.is_implicit_VR' must " 

554 f"be set appropriately before saving" 

555 ) 

556 

557 if not dataset.is_original_encoding: 

558 dataset = correct_ambiguous_vr(dataset, fp.is_little_endian) 

559 

560 dataset_encoding = dataset.get('SpecificCharacterSet', parent_encoding) 

561 

562 fpStart = fp.tell() 

563 # data_elements must be written in tag order 

564 tags = sorted(dataset.keys()) 

565 

566 for tag in tags: 

567 # do not write retired Group Length (see PS3.5, 7.2) 

568 if tag.element == 0 and tag.group > 6: 

569 continue 

570 with tag_in_exception(tag): 

571 write_data_element(fp, dataset.get_item(tag), dataset_encoding) 

572 

573 return fp.tell() - fpStart

576def _harmonize_properties(dataset, fp): 

577 """Make sure the properties in the dataset and the file pointer are 

578 consistent, so the user can set both with the same effect. 

579 Properties set on the destination file object always have preference. 

580 """ 

581 # ensure preference of fp over dataset 

582 if hasattr(fp, 'is_little_endian'): 

583 dataset.is_little_endian = fp.is_little_endian 

584 if hasattr(fp, 'is_implicit_VR'): 

585 dataset.is_implicit_VR = fp.is_implicit_VR 

586 

587 # write the properties back to have a consistent state 

588 fp.is_implicit_VR = dataset.is_implicit_VR 

589 fp.is_little_endian = dataset.is_little_endian

611def write_sequence_item(fp, dataset, encodings): 

612 """Write a `dataset` in a sequence to the file-like `fp`. 

613 

614 This is similar to writing a data_element, but with a specific tag for 

615 Sequence Item. 

616 

617 See DICOM Standard, Part 5, :dcm:`Section 7.5<sect_7.5.html>`. 

618 

619 Parameters 

620 ---------- 

621 fp : file-like 

622 The file-like to write the encoded data to. 

623 dataset : Dataset 

624 The :class:`Dataset<pydicom.dataset.Dataset>` to write to `fp`. 

625 encodings : list of str 

626 The character encodings to use on text values. 

627 """ 

628 fp.write_tag(ItemTag) # marker for start of Sequence Item 

629 length_location = fp.tell() # save location for later. 

630 # will fill in real value later if not undefined length 

631 fp.write_UL(0xffffffff) 

632 write_dataset(fp, dataset, parent_encoding=encodings) 

633 if getattr(dataset, "is_undefined_length_sequence_item", False): 

634 fp.write_tag(ItemDelimiterTag) 

635 fp.write_UL(0) # 4-bytes 'length' field for delimiter item 

636 else: # we will be nice and set the lengths for the reader of this file 

637 location = fp.tell() 

638 fp.seek(length_location) 

639 fp.write_UL(location - length_location - 4) # 4 is length of UL

664def write_file_meta_info(fp, file_meta, enforce_standard=True): 

665 """Write the File Meta Information elements in `file_meta` to `fp`. 

666 

667 If `enforce_standard` is ``True`` then the file-like `fp` should be 

668 positioned past the 128 byte preamble + 4 byte prefix (which should 

669 already have been written). 

670 

671 **DICOM File Meta Information Group Elements** 

672 

673 From the DICOM standard, Part 10, 

674 :dcm:`Section 7.1<part10/chapter_7.html#sect_7.1>`, any DICOM file shall 

675 contain a 128-byte preamble, a 4-byte DICOM prefix 'DICM' and (at a 

676 minimum) the following Type 1 DICOM Elements (from 

677 :dcm:`Table 7.1-1<part10/chapter_7.html#table_7.1-1>`): 

678 

679 * (0002,0000) *File Meta Information Group Length*, UL, 4 

680 * (0002,0001) *File Meta Information Version*, OB, 2 

681 * (0002,0002) *Media Storage SOP Class UID*, UI, N 

682 * (0002,0003) *Media Storage SOP Instance UID*, UI, N 

683 * (0002,0010) *Transfer Syntax UID*, UI, N 

684 * (0002,0012) *Implementation Class UID*, UI, N 

685 

686 If `enforce_standard` is ``True`` then (0002,0000) will be added/updated, 

687 (0002,0001) and (0002,0012) will be added if not already present and the 

688 other required elements will be checked to see if they exist. If 

689 `enforce_standard` is ``False`` then `file_meta` will be written as is 

690 after minimal validation checking. 

691 

692 The following Type 3/1C Elements may also be present: 

693 

694 * (0002,0013) *Implementation Version Name*, SH, N 

695 * (0002,0016) *Source Application Entity Title*, AE, N 

696 * (0002,0017) *Sending Application Entity Title*, AE, N 

697 * (0002,0018) *Receiving Application Entity Title*, AE, N 

698 * (0002,0102) *Private Information*, OB, N 

699 * (0002,0100) *Private Information Creator UID*, UI, N 

700 

701 If `enforce_standard` is ``True`` then (0002,0013) will be added/updated. 

702 

703 *Encoding* 

704 

705 The encoding of the *File Meta Information* shall be *Explicit VR Little 

706 Endian*. 

707 

708 Parameters 

709 ---------- 

710 fp : file-like 

711 The file-like to write the File Meta Information to. 

712 file_meta : pydicom.dataset.Dataset 

713 The File Meta Information elements. 

714 enforce_standard : bool 

715 If ``False``, then only the *File Meta Information* elements already in 

716 `file_meta` will be written to `fp`. If ``True`` (default) then a DICOM 

717 Standards conformant File Meta will be written to `fp`. 

718 

719 Raises 

720 ------ 

721 ValueError 

722 If `enforce_standard` is ``True`` and any of the required *File Meta 

723 Information* elements are missing from `file_meta`, with the 

724 exception of (0002,0000), (0002,0001) and (0002,0012). 

725 ValueError 

726 If any non-Group 2 Elements are present in `file_meta`. 

727 """ 

728 validate_file_meta(file_meta, enforce_standard) 

729 

730 if enforce_standard and 'FileMetaInformationGroupLength' not in file_meta: 

731 # Will be updated with the actual length later 

732 file_meta.FileMetaInformationGroupLength = 0 

733 

734 # Write the File Meta Information Group elements 

735 # first write into a buffer to avoid seeking back, that can be 

736 # expansive and is not allowed if writing into a zip file 

737 buffer = DicomBytesIO() 

738 buffer.is_little_endian = True 

739 buffer.is_implicit_VR = False 

740 write_dataset(buffer, file_meta) 

741 

742 # If FileMetaInformationGroupLength is present it will be the first written 

743 # element and we must update its value to the correct length. 

744 if 'FileMetaInformationGroupLength' in file_meta: 

745 # Update the FileMetaInformationGroupLength value, which is the number 

746 # of bytes from the end of the FileMetaInformationGroupLength element 

747 # to the end of all the File Meta Information elements. 

748 # FileMetaInformationGroupLength has a VR of 'UL' and so has a value 

749 # that is 4 bytes fixed. The total length of when encoded as 

750 # Explicit VR must therefore be 12 bytes. 

751 file_meta.FileMetaInformationGroupLength = buffer.tell() - 12 

752 buffer.seek(0) 

753 write_data_element(buffer, file_meta[0x00020000]) 

754 

755 fp.write(buffer.getvalue())

758def _write_dataset(fp, dataset, write_like_original): 

759 """Write the Data Set to a file-like. Assumes the file meta information, 

760 if any, has been written. 

761 """ 

762 

763 # if we want to write with the same endianess and VR handling as 

764 # the read dataset we want to preserve raw data elements for 

765 # performance reasons (which is done by get_item); 

766 # otherwise we use the default converting item getter 

767 if dataset.is_original_encoding: 

768 get_item = Dataset.get_item 

769 else: 

770 get_item = Dataset.__getitem__ 

771 

772 # WRITE DATASET 

773 # The transfer syntax used to encode the dataset can't be changed 

774 # within the dataset. 

775 # Write any Command Set elements now as elements must be in tag order 

776 # Mixing Command Set with other elements is non-conformant so we 

777 # require `write_like_original` to be True 

778 command_set = get_item(dataset, slice(0x00000000, 0x00010000)) 

779 if command_set and write_like_original: 

780 fp.is_implicit_VR = True 

781 fp.is_little_endian = True 

782 write_dataset(fp, command_set) 

783 

784 # Set file VR and endianness. MUST BE AFTER writing META INFO (which 

785 # requires Explicit VR Little Endian) and COMMAND SET (which requires 

786 # Implicit VR Little Endian) 

787 fp.is_implicit_VR = dataset.is_implicit_VR 

788 fp.is_little_endian = dataset.is_little_endian 

789 

790 # Write non-Command Set elements now 

791 write_dataset(fp, get_item(dataset, slice(0x00010000, None)))

794def dcmwrite(filename, dataset, write_like_original=True): 

795 """Write `dataset` to the `filename` specified. 

796 

797 If `write_like_original` is ``True`` then `dataset` will be written as is 

798 (after minimal validation checking) and may or may not contain all or parts 

799 of the File Meta Information (and hence may or may not be conformant with 

800 the DICOM File Format). 

801 

802 If `write_like_original` is ``False``, `dataset` will be stored in the 

803 :dcm:`DICOM File Format <part10/chapter_7.html>`. To do 

804 so requires that the ``Dataset.file_meta`` attribute 

805 exists and contains a :class:`Dataset` with the required (Type 1) *File 

806 Meta Information Group* elements. The byte stream of the `dataset` will be 

807 placed into the file after the DICOM *File Meta Information*. 

808 

809 If `write_like_original` is ``True`` then the :class:`Dataset` will be 

810 written as is (after minimal validation checking) and may or may not 

811 contain all or parts of the *File Meta Information* (and hence may or 

812 may not be conformant with the DICOM File Format). 

813 

814 **File Meta Information** 

815 

816 The *File Meta Information* consists of a 128-byte preamble, followed by 

817 a 4 byte ``b'DICM'`` prefix, followed by the *File Meta Information Group* 

818 elements. 

819 

820 **Preamble and Prefix** 

821 

822 The ``dataset.preamble`` attribute shall be 128-bytes long or ``None`` and 

823 is available for use as defined by the Application Profile or specific 

824 implementations. If the preamble is not used by an Application Profile or 

825 specific implementation then all 128 bytes should be set to ``0x00``. The 

826 actual preamble written depends on `write_like_original` and 

827 ``dataset.preamble`` (see the table below). 

828 

829 +------------------+------------------------------+ 

830 | | write_like_original | 

831 +------------------+-------------+----------------+ 

832 | dataset.preamble | True | False | 

833 +==================+=============+================+ 

834 | None | no preamble | 128 0x00 bytes | 

835 +------------------+-------------+----------------+ 

836 | 128 bytes | dataset.preamble | 

837 +------------------+------------------------------+ 

838 

839 The prefix shall be the bytestring ``b'DICM'`` and will be written if and 

840 only if the preamble is present. 

841 

842 **File Meta Information Group Elements** 

843 

844 The preamble and prefix are followed by a set of DICOM elements from the 

845 (0002,eeee) group. Some of these elements are required (Type 1) while 

846 others are optional (Type 3/1C). If `write_like_original` is ``True`` 

847 then the *File Meta Information Group* elements are all optional. See 

848 :func:`~pydicom.filewriter.write_file_meta_info` for more information on 

849 which elements are required. 

850 

851 The *File Meta Information Group* elements should be included within their 

852 own :class:`~pydicom.dataset.Dataset` in the ``dataset.file_meta`` 

853 attribute. 

854 

855 If (0002,0010) *Transfer Syntax UID* is included then the user must ensure 

856 its value is compatible with the values for the 

857 ``dataset.is_little_endian`` and ``dataset.is_implicit_VR`` attributes. 

858 For example, if ``is_little_endian`` and ``is_implicit_VR`` are both 

859 ``True`` then the Transfer Syntax UID must be 1.2.840.10008.1.2 *Implicit 

860 VR Little Endian*. See the DICOM Standard, Part 5, 

861 :dcm:`Section 10<part05/chapter_10.html>` for more information on Transfer 

862 Syntaxes. 

863 

864 *Encoding* 

865 

866 The preamble and prefix are encoding independent. The File Meta elements 

867 are encoded as *Explicit VR Little Endian* as required by the DICOM 

868 Standard. 

869 

870 **Dataset** 

871 

872 A DICOM Dataset representing a SOP Instance related to a DICOM Information 

873 Object Definition. It is up to the user to ensure the `dataset` conforms 

874 to the DICOM Standard. 

875 

876 *Encoding* 

877 

878 The `dataset` is encoded as specified by the ``dataset.is_little_endian`` 

879 and ``dataset.is_implicit_VR`` attributes. It's up to the user to ensure 

880 these attributes are set correctly (as well as setting an appropriate 

881 value for ``dataset.file_meta.TransferSyntaxUID`` if present). 

882 

883 Parameters 

884 ---------- 

885 filename : str or PathLike or file-like 

886 Name of file or the file-like to write the new DICOM file to. 

887 dataset : pydicom.dataset.FileDataset 

888 Dataset holding the DICOM information; e.g. an object read with 

889 :func:`~pydicom.filereader.dcmread`. 

890 write_like_original : bool, optional 

891 If ``True`` (default), preserves the following information from 

892 the Dataset (and may result in a non-conformant file): 

893 

894 - preamble -- if the original file has no preamble then none will be 

895 written. 

896 - file_meta -- if the original file was missing any required *File 

897 Meta Information Group* elements then they will not be added or 

898 written. 

899 If (0002,0000) *File Meta Information Group Length* is present then 

900 it may have its value updated. 

901 - seq.is_undefined_length -- if original had delimiters, write them now 

902 too, instead of the more sensible length characters 

903 - is_undefined_length_sequence_item -- for datasets that belong to a 

904 sequence, write the undefined length delimiters if that is 

905 what the original had. 

906 

907 If ``False``, produces a file conformant with the DICOM File Format, 

908 with explicit lengths for all elements. 

909 

910 Raises 

911 ------ 

912 AttributeError 

913 If either ``dataset.is_implicit_VR`` or ``dataset.is_little_endian`` 

914 have not been set. 

915 ValueError 

916 If group 2 elements are in ``dataset`` rather than 

917 ``dataset.file_meta``, or if a preamble is given but is not 128 bytes 

918 long, or if Transfer Syntax is a compressed type and pixel data is not 

919 compressed. 

920 

921 See Also 

922 -------- 

923 pydicom.dataset.Dataset 

924 Dataset class with relevant attributes and information. 

925 pydicom.dataset.Dataset.save_as 

926 Write a DICOM file from a dataset that was read in with ``dcmread()``. 

927 ``save_as()`` wraps ``dcmwrite()``. 

928 """ 

929 

930 # Ensure is_little_endian and is_implicit_VR are set 

931 if None in (dataset.is_little_endian, dataset.is_implicit_VR): 

932 has_tsyntax = False 

933 try: 

934 tsyntax = dataset.file_meta.TransferSyntaxUID 

935 if not tsyntax.is_private: 

936 dataset.is_little_endian = tsyntax.is_little_endian 

937 dataset.is_implicit_VR = tsyntax.is_implicit_VR 

938 has_tsyntax = True 

939 except AttributeError: 

940 pass 

941 

942 if not has_tsyntax: 

943 name = dataset.__class__.__name__ 

944 raise AttributeError( 

945 f"'{name}.is_little_endian' and '{name}.is_implicit_VR' must " 

946 f"be set appropriately before saving" 

947 ) 

948 

949 # Try and ensure that `is_undefined_length` is set correctly 

950 try: 

951 tsyntax = dataset.file_meta.TransferSyntaxUID 

952 if not tsyntax.is_private: 

953 dataset['PixelData'].is_undefined_length = tsyntax.is_compressed 

954 except (AttributeError, KeyError): 

955 pass 

956 

957 # Check that dataset's group 0x0002 elements are only present in the 

958 # `dataset.file_meta` Dataset - user may have added them to the wrong 

959 # place 

960 if dataset.group_dataset(0x0002) != Dataset(): 

961 raise ValueError( 

962 f"File Meta Information Group Elements (0002,eeee) should be in " 

963 f"their own Dataset object in the " 

964 f"'{dataset.__class__.__name__}.file_meta' attribute." 

965 ) 

966 

967 # A preamble is required under the DICOM standard, however if 

968 # `write_like_original` is True we treat it as optional 

969 preamble = getattr(dataset, 'preamble', None) 

970 if preamble and len(preamble) != 128: 

971 raise ValueError( 

972 f"'{dataset.__class__.__name__}.preamble' must be 128-bytes long." 

973 ) 

974 if not preamble and not write_like_original: 

975 # The default preamble is 128 0x00 bytes. 

976 preamble = b'\x00' * 128 

977 

978 # File Meta Information is required under the DICOM standard, however if 

979 # `write_like_original` is True we treat it as optional 

980 if not write_like_original: 

981 # the checks will be done in write_file_meta_info() 

982 dataset.fix_meta_info(enforce_standard=False) 

983 else: 

984 dataset.ensure_file_meta() 

985 

986 # Check for decompression, give warnings if inconsistencies 

987 # If decompressed, then pixel_array is now used instead of PixelData 

988 if dataset.is_decompressed: 

989 if dataset.file_meta.TransferSyntaxUID.is_compressed: 

990 raise ValueError( 

991 f"The Transfer Syntax UID element in " 

992 f"'{dataset.__class__.__name__}.file_meta' is compressed " 

993 f"but the pixel data has been decompressed" 

994 ) 

995 

996 # Force PixelData to the decompressed version 

997 dataset.PixelData = dataset.pixel_array.tobytes() 

998 

999 caller_owns_file = True 

1000 # Open file if not already a file object 

1001 filename = path_from_pathlike(filename) 

1002 if isinstance(filename, str): 

1003 fp = DicomFile(filename, 'wb') 

1004 # caller provided a file name; we own the file handle 

1005 caller_owns_file = False 

1006 else: 

1007 fp = DicomFileLike(filename) 

1008 

1009 try: 

1010 # WRITE FILE META INFORMATION 

1011 if preamble: 

1012 # Write the 'DICM' prefix if and only if we write the preamble 

1013 fp.write(preamble) 

1014 fp.write(b'DICM') 

1015 

1016 tsyntax = None 

1017 if dataset.file_meta: # May be an empty Dataset 

1018 # If we want to `write_like_original`, don't enforce_standard 

1019 write_file_meta_info( 

1020 fp, dataset.file_meta, enforce_standard=not write_like_original 

1021 ) 

1022 tsyntax = getattr(dataset.file_meta, "TransferSyntaxUID", None) 

1023 

1024 if (tsyntax == DeflatedExplicitVRLittleEndian): 

1025 # See PS3.5 section A.5 

1026 # when writing, the entire dataset following 

1027 # the file metadata is prepared the normal way, 

1028 # then "deflate" compression applied. 

1029 buffer = DicomBytesIO() 

1030 _write_dataset(buffer, dataset, write_like_original) 

1031 

1032 # Compress the encoded data and write to file 

1033 compressor = zlib.compressobj(wbits=-zlib.MAX_WBITS) 

1034 deflated = compressor.compress(buffer.parent.getvalue()) 

1035 deflated += compressor.flush() 

1036 if len(deflated) % 2: 

1037 deflated += b'\x00' 

1038 

1039 fp.write(deflated) 

1040 else: 

1041 _write_dataset(fp, dataset, write_like_original) 

1042 

1043 finally: 

1044 if not caller_owns_file: 

1045 fp.close()
