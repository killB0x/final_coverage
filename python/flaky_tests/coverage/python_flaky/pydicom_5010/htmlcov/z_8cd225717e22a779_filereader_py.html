31def data_element_generator(fp, 

32 is_implicit_VR, 

33 is_little_endian, 

34 stop_when=None, 

35 defer_size=None, 

36 encoding=default_encoding, 

37 specific_tags=None): 

38 

39 """Create a generator to efficiently return the raw data elements. 

40 

41 .. note:: 

42 

43 This function is used internally - usually there is no need to call it 

44 from user code. To read data from a DICOM file, :func:`dcmread` 

45 shall be used instead. 

46 

47 Parameters 

48 ---------- 

49 fp : file-like 

50 The file-like to read from. 

51 is_implicit_VR : bool 

52 ``True`` if the data is encoded as implicit VR, ``False`` otherwise. 

53 is_little_endian : bool 

54 ``True`` if the data is encoded as little endian, ``False`` otherwise. 

55 stop_when : None, callable, optional 

56 If ``None`` (default), then the whole file is read. A callable which 

57 takes tag, VR, length, and returns ``True`` or ``False``. If it 

58 returns ``True``, ``read_data_element`` will just return. 

59 defer_size : int, str, None, optional 

60 See :func:`dcmread` for parameter info. 

61 encoding : 

62 Encoding scheme 

63 specific_tags : list or None 

64 See :func:`dcmread` for parameter info. 

65 

66 Returns 

67 ------- 

68 VR : str or None 

69 ``None`` if implicit VR, otherwise the VR read from the file. 

70 length : int 

71 The length of the DICOM data element (could be DICOM "undefined 

72 length" ``0xFFFFFFFFL``) 

73 value_bytes : bytes or str 

74 The raw bytes from the DICOM file (not parsed into Python types) 

75 is_little_endian : bool 

76 ``True`` if transfer syntax is little endian; else ``False``. 

77 """ 

78 # Summary of DICOM standard PS3.5-2008 chapter 7: 

79 # If Implicit VR, data element is: 

80 # tag, 4-byte length, value. 

81 # The 4-byte length can be FFFFFFFF (undefined length)* 

82 # 

83 # If Explicit VR: 

84 # if OB, OW, OF, SQ, UN, or UT: 

85 # tag, VR, 2-bytes reserved (both zero), 4-byte length, value 

86 # For all but UT, the length can be FFFFFFFF (undefined length)* 

87 # else: (any other VR) 

88 # tag, VR, (2 byte length), value 

89 # * for undefined length, a Sequence Delimitation Item marks the end 

90 # of the Value Field. 

91 # Note, except for the special_VRs, both impl and expl VR use 8 bytes; 

92 # the special VRs follow the 8 bytes with a 4-byte length 

93 

94 # With a generator, state is stored, so we can break down 

95 # into the individual cases, and not have to check them again for each 

96 # data element 

97 

98 if is_little_endian: 

99 endian_chr = "<" 

100 else: 

101 endian_chr = ">" 

102 if is_implicit_VR: 

103 element_struct = Struct(endian_chr + "HHL") 

104 else: # Explicit VR 

105 # tag, VR, 2-byte length (or 0 if special VRs) 

106 element_struct = Struct(endian_chr + "HH2sH") 

107 extra_length_struct = Struct(endian_chr + "L") # for special VRs 

108 extra_length_unpack = extra_length_struct.unpack # for lookup speed 

109 

110 # Make local variables so have faster lookup 

111 fp_read = fp.read 

112 fp_tell = fp.tell 

113 logger_debug = logger.debug 

114 debugging = config.debugging 

115 element_struct_unpack = element_struct.unpack 

116 defer_size = size_in_bytes(defer_size) 

117 

118 tag_set = set() 

119 if specific_tags is not None: 

120 for tag in specific_tags: 

121 if isinstance(tag, str): 

122 tag = Tag(tag_for_keyword(tag)) 

123 if isinstance(tag, BaseTag): 

124 tag_set.add(tag) 

125 tag_set.add(Tag(0x08, 0x05)) 

126 has_tag_set = len(tag_set) > 0 

127 

128 while True: 

129 # Read tag, VR, length, get ready to read value 

130 bytes_read = fp_read(8) 

131 if len(bytes_read) < 8: 

132 return # at end of file 

133 if debugging: 

134 debug_msg = "{0:08x}: {1}".format(fp.tell() - 8, 

135 bytes2hex(bytes_read)) 

136 

137 if is_implicit_VR: 

138 # must reset VR each time; could have set last iteration (e.g. SQ) 

139 VR = None 

140 group, elem, length = element_struct_unpack(bytes_read) 

141 else: # explicit VR 

142 group, elem, VR, length = element_struct_unpack(bytes_read) 

143 VR = VR.decode(default_encoding) 

144 if VR in extra_length_VRs: 

145 bytes_read = fp_read(4) 

146 length = extra_length_unpack(bytes_read)[0] 

147 if debugging: 

148 debug_msg += " " + bytes2hex(bytes_read) 

149 if debugging: 

150 debug_msg = "%-47s (%04x, %04x)" % (debug_msg, group, elem) 

151 if not is_implicit_VR: 

152 debug_msg += " %s " % VR 

153 if length != 0xFFFFFFFF: 

154 debug_msg += "Length: %d" % length 

155 else: 

156 debug_msg += "Length: Undefined length (FFFFFFFF)" 

157 logger_debug(debug_msg) 

158 

159 # Positioned to read the value, but may not want to -- check stop_when 

160 value_tell = fp_tell() 

161 tag = TupleTag((group, elem)) 

162 if stop_when is not None: 

163 # XXX VR may be None here!! Should stop_when just take tag? 

164 if stop_when(tag, VR, length): 

165 if debugging: 

166 logger_debug("Reading ended by stop_when callback. " 

167 "Rewinding to start of data element.") 

168 rewind_length = 8 

169 if not is_implicit_VR and VR in extra_length_VRs: 

170 rewind_length += 4 

171 fp.seek(value_tell - rewind_length) 

172 return 

173 

174 # Reading the value 

175 # First case (most common): reading a value with a defined length 

176 if length != 0xFFFFFFFF: 

177 # don't defer loading of Specific Character Set value as it is 

178 # needed immediately to get the character encoding for other tags 

179 if has_tag_set and tag not in tag_set: 

180 # skip the tag if not in specific tags 

181 fp.seek(fp_tell() + length) 

182 continue 

183 

184 if (defer_size is not None and length > defer_size and 

185 tag != BaseTag(0x00080005)): 

186 # Flag as deferred by setting value to None, and skip bytes 

187 value = None 

188 logger_debug("Defer size exceeded. " 

189 "Skipping forward to next data element.") 

190 fp.seek(fp_tell() + length) 

191 else: 

192 value = (fp_read(length) if length > 0 

193 else empty_value_for_VR(VR, raw=True)) 

194 if debugging: 

195 dotdot = "..." if length > 12 else " " 

196 displayed_value = value[:12] if value else b'' 

197 logger_debug("%08x: %-34s %s %r %s" % 

198 (value_tell, bytes2hex(displayed_value), 

199 dotdot, displayed_value, dotdot)) 

200 

201 # If the tag is (0008,0005) Specific Character Set, then store it 

202 if tag == BaseTag(0x00080005): 

203 from pydicom.values import convert_string 

204 encoding = convert_string(value or b'', is_little_endian) 

205 # Store the encoding value in the generator 

206 # for use with future elements (SQs) 

207 encoding = convert_encodings(encoding) 

208 

209 yield RawDataElement(tag, VR, length, value, value_tell, 

210 is_implicit_VR, is_little_endian) 

211 

212 # Second case: undefined length - must seek to delimiter, 

213 # unless is SQ type, in which case is easier to parse it, because 

214 # undefined length SQs and items of undefined lengths can be nested 

215 # and it would be error-prone to read to the correct outer delimiter 

216 else: 

217 # Try to look up type to see if is a SQ 

218 # if private tag, won't be able to look it up in dictionary, 

219 # in which case just ignore it and read the bytes unless it is 

220 # identified as a Sequence 

221 if VR is None: 

222 try: 

223 VR = dictionary_VR(tag) 

224 except KeyError: 

225 # Look ahead to see if it consists of items 

226 # and is thus a SQ 

227 next_tag = TupleTag(unpack(endian_chr + "HH", fp_read(4))) 

228 # Rewind the file 

229 fp.seek(fp_tell() - 4) 

230 if next_tag == ItemTag: 

231 VR = 'SQ' 

232 

233 if VR == 'SQ': 

234 if debugging: 

235 msg = "{0:08x}: Reading/parsing undefined length sequence" 

236 logger_debug(msg.format(fp_tell())) 

237 seq = read_sequence(fp, is_implicit_VR, 

238 is_little_endian, length, encoding) 

239 if has_tag_set and tag not in tag_set: 

240 continue 

241 yield DataElement(tag, VR, seq, value_tell, 

242 is_undefined_length=True) 

243 else: 

244 delimiter = SequenceDelimiterTag 

245 if debugging: 

246 logger_debug("Reading undefined length data element") 

247 value = read_undefined_length_value(fp, is_little_endian, 

248 delimiter, defer_size) 

249 

250 # tags with undefined length are skipped after read 

251 if has_tag_set and tag not in tag_set: 

252 continue 

253 yield RawDataElement(tag, VR, length, value, value_tell, 

254 is_implicit_VR, is_little_endian)

257def _is_implicit_vr(fp, implicit_vr_is_assumed, is_little_endian, stop_when): 

258 """Check if the real VR is explicit or implicit. 

259 

260 Parameters 

261 ---------- 

262 fp : an opened file object 

263 implicit_vr_is_assumed : bool 

264 True if implicit VR is assumed. 

265 If this does not match with the real transfer syntax, a user warning 

266 will be issued. 

267 is_little_endian : bool 

268 True if file has little endian transfer syntax. 

269 Needed to interpret the first tag. 

270 stop_when : None, optional 

271 Optional call_back function which can terminate reading. 

272 Needed to check if the next tag still belongs to the read dataset. 

273 

274 Returns 

275 ------- 

276 True if implicit VR is used, False otherwise. 

277 """ 

278 tag_bytes = fp.read(4) 

279 vr = fp.read(2) 

280 if len(vr) < 2: 

281 return implicit_vr_is_assumed 

282 

283 # it is sufficient to check if the VR is in valid ASCII range, as it is 

284 # extremely unlikely that the tag length accidentally has such a 

285 # representation - this would need the first tag to be longer than 16kB 

286 # (e.g. it should be > 0x4141 = 16705 bytes) 

287 found_implicit = not (0x40 < vr[0] < 0x5B and 0x40 < vr[1] < 0x5B) 

288 

289 if found_implicit != implicit_vr_is_assumed: 

290 # first check if the tag still belongs to the dataset if stop_when 

291 # is given - if not, the dataset is empty and we just return 

292 endian_chr = "<" if is_little_endian else ">" 

293 tag = TupleTag(unpack(endian_chr + "HH", tag_bytes)) 

294 if stop_when is not None and stop_when(tag, vr, 0): 

295 return found_implicit 

296 

297 # got to the real problem - warn or raise depending on config 

298 found_vr = 'implicit' if found_implicit else 'explicit' 

299 expected_vr = 'implicit' if not found_implicit else 'explicit' 

300 message = ('Expected {0} VR, but found {1} VR - using {1} VR for ' 

301 'reading'.format(expected_vr, found_vr)) 

302 if config.enforce_valid_values: 

303 raise InvalidDicomError(message) 

304 warnings.warn(message, UserWarning) 

305 return found_implicit

308def read_dataset(fp, is_implicit_VR, is_little_endian, bytelength=None, 

309 stop_when=None, defer_size=None, 

310 parent_encoding=default_encoding, specific_tags=None, 

311 at_top_level=True): 

312 """Return a :class:`~pydicom.dataset.Dataset` instance containing the next 

313 dataset in the file. 

314 

315 Parameters 

316 ---------- 

317 fp : file-like 

318 An opened file-like object. 

319 is_implicit_VR : bool 

320 ``True`` if file transfer syntax is implicit VR. 

321 is_little_endian : bool 

322 ``True`` if file has little endian transfer syntax. 

323 bytelength : int, None, optional 

324 ``None`` to read until end of file or ItemDeliterTag, else a fixed 

325 number of bytes to read 

326 stop_when : None, optional 

327 Optional call_back function which can terminate reading. See help for 

328 :func:`data_element_generator` for details 

329 defer_size : int, None, optional 

330 Size to avoid loading large elements in memory. See :func:`dcmread` for 

331 more parameter info. 

332 parent_encoding : 

333 Optional encoding to use as a default in case (0008,0005) *Specific 

334 Character Set* isn't specified. 

335 specific_tags : list or None 

336 See :func:`dcmread` for parameter info. 

337 at_top_level: bool 

338 If dataset is top level (not within a sequence). 

339 Used to turn off explicit VR heuristic within sequences 

340 

341 Returns 

342 ------- 

343 dataset.Dataset 

344 A Dataset instance. 

345 

346 See Also 

347 -------- 

348 :class:`~pydicom.dataset.Dataset` 

349 A collection (dictionary) of DICOM 

350 :class:`~pydicom.dataelem.DataElement` instances. 

351 """ 

352 raw_data_elements = dict() 

353 fp_start = fp.tell() 

354 if at_top_level: 

355 is_implicit_VR = _is_implicit_vr( 

356 fp, is_implicit_VR, is_little_endian, stop_when) 

357 fp.seek(fp_start) 

358 de_gen = data_element_generator(fp, is_implicit_VR, is_little_endian, 

359 stop_when, defer_size, parent_encoding, 

360 specific_tags) 

361 try: 

362 while (bytelength is None) or (fp.tell() - fp_start < bytelength): 

363 raw_data_element = next(de_gen) 

364 # Read data elements. Stop on some errors, but return what was read 

365 tag = raw_data_element.tag 

366 # Check for ItemDelimiterTag --dataset is an item in a sequence 

367 if tag == BaseTag(0xFFFEE00D): 

368 break 

369 raw_data_elements[tag] = raw_data_element 

370 except StopIteration: 

371 pass 

372 except EOFError as details: 

373 if config.enforce_valid_values: 

374 raise 

375 msg = str(details) + " in file " + getattr(fp, "name", "<no filename>") 

376 warnings.warn(msg, UserWarning) 

377 except NotImplementedError as details: 

378 logger.error(details) 

379 

380 ds = Dataset(raw_data_elements) 

381 if 0x00080005 in raw_data_elements: 

382 char_set = DataElement_from_raw(raw_data_elements[0x00080005]).value 

383 encoding = convert_encodings(char_set) 

384 else: 

385 encoding = parent_encoding 

386 ds.set_original_encoding(is_implicit_VR, is_little_endian, encoding) 

387 return ds

390def read_sequence(fp, is_implicit_VR, is_little_endian, bytelength, encoding, 

391 offset=0): 

392 """Read and return a :class:`~pydicom.sequence.Sequence` -- i.e. a 

393 :class:`list` of :class:`Datasets<pydicom.dataset.Dataset>`. 

394 """ 

395 

396 seq = [] # use builtin list to start for speed, convert to Sequence at end 

397 is_undefined_length = False 

398 if bytelength != 0: # SQ of length 0 possible (PS 3.5-2008 7.5.1a (p.40) 

399 if bytelength == 0xffffffff: 

400 is_undefined_length = True 

401 bytelength = None 

402 fp_tell = fp.tell # for speed in loop 

403 fpStart = fp_tell() 

404 while (not bytelength) or (fp_tell() - fpStart < bytelength): 

405 file_tell = fp.tell() 

406 dataset = read_sequence_item(fp, is_implicit_VR, is_little_endian, 

407 encoding, offset) 

408 if dataset is None: # None is returned if hit Sequence Delimiter 

409 break 

410 dataset.file_tell = file_tell + offset 

411 seq.append(dataset) 

412 seq = Sequence(seq) 

413 seq.is_undefined_length = is_undefined_length 

414 return seq

417def read_sequence_item(fp, is_implicit_VR, is_little_endian, encoding, 

418 offset=0): 

419 """Read and return a single :class:`~pydicom.sequence.Sequence` item, i.e. 

420 a :class:`~pydicom.dataset.Dataset`. 

421 """ 

422 seq_item_tell = fp.tell() + offset 

423 if is_little_endian: 

424 tag_length_format = "<HHL" 

425 else: 

426 tag_length_format = ">HHL" 

427 try: 

428 bytes_read = fp.read(8) 

429 group, element, length = unpack(tag_length_format, bytes_read) 

430 except BaseException: 

431 raise IOError("No tag to read at file position " 

432 "{0:05x}".format(fp.tell() + offset)) 

433 tag = (group, element) 

434 if tag == SequenceDelimiterTag: # No more items, time to stop reading 

435 logger.debug( 

436 "{0:08x}: {1}".format(fp.tell() - 8 + offset, "End of Sequence")) 

437 if length != 0: 

438 logger.warning("Expected 0x00000000 after delimiter, found 0x%x, " 

439 "at position 0x%x" % ( 

440 length, fp.tell() - 4 + offset)) 

441 return None 

442 if tag != ItemTag: 

443 logger.warning("Expected sequence item with tag %s at file position " 

444 "0x%x" % (ItemTag, fp.tell() - 4 + offset)) 

445 else: 

446 logger.debug("{0:08x}: {1} Found Item tag (start of item)".format( 

447 fp.tell() - 4 + offset, bytes2hex(bytes_read))) 

448 if length == 0xFFFFFFFF: 

449 ds = read_dataset(fp, is_implicit_VR, is_little_endian, 

450 bytelength=None, parent_encoding=encoding, 

451 at_top_level=False) 

452 ds.is_undefined_length_sequence_item = True 

453 else: 

454 ds = read_dataset(fp, is_implicit_VR, is_little_endian, length, 

455 parent_encoding=encoding, 

456 at_top_level=False) 

457 ds.is_undefined_length_sequence_item = False 

458 logger.debug("%08x: Finished sequence item" % (fp.tell() + offset,)) 

459 ds.seq_item_tell = seq_item_tell 

460 return ds

493def _read_file_meta_info(fp): 

494 """Return a Dataset containing any File Meta (0002,eeee) elements in `fp`. 

495 

496 File Meta elements are always Explicit VR Little Endian (DICOM Standard, 

497 Part 10, :dcm:`Section 7<part10/chapter_7.html>`). Once any File Meta 

498 elements are read `fp` will be positioned at the start of the next group 

499 of elements. 

500 

501 Parameters 

502 ---------- 

503 fp : file-like 

504 The file-like positioned at the start of any File Meta Information 

505 group elements. 

506 

507 Returns 

508 ------- 

509 dataset.Dataset 

510 The File Meta elements as a Dataset instance. May be empty if no 

511 File Meta are present. 

512 """ 

513 

514 def _not_group_0002(tag, VR, length): 

515 """Return True if the tag is not in group 0x0002, False otherwise.""" 

516 return tag.group != 2 

517 

518 start_file_meta = fp.tell() 

519 file_meta = FileMetaDataset( 

520 read_dataset( 

521 fp, is_implicit_VR=False, is_little_endian=True, 

522 stop_when=_not_group_0002 

523 ) 

524 ) 

525 if not file_meta._dict: 

526 return file_meta 

527 

528 # Test the file meta for correct interpretation by requesting the first 

529 # data element: if it fails, retry loading the file meta with an 

530 # implicit VR (issue #503) 

531 try: 

532 file_meta[list(file_meta.elements())[0].tag] 

533 except NotImplementedError: 

534 fp.seek(start_file_meta) 

535 file_meta = FileMetaDataset( 

536 read_dataset( 

537 fp, is_implicit_VR=True, is_little_endian=True, 

538 stop_when=_not_group_0002 

539 ) 

540 ) 

541 

542 # Log if the Group Length doesn't match actual length 

543 if 'FileMetaInformationGroupLength' in file_meta: 

544 # FileMetaInformationGroupLength must be 12 bytes long and its value 

545 # counts from the beginning of the next element to the end of the 

546 # file meta elements 

547 length_file_meta = fp.tell() - (start_file_meta + 12) 

548 if file_meta.FileMetaInformationGroupLength != length_file_meta: 

549 logger.info("_read_file_meta_info: (0002,0000) 'File Meta " 

550 "Information Group Length' value doesn't match the " 

551 "actual File Meta Information length ({0} vs {1} " 

552 "bytes)." 

553 .format(file_meta.FileMetaInformationGroupLength, 

554 length_file_meta)) 

555 

556 return file_meta

571def read_preamble(fp, force): 

572 """Return the 128-byte DICOM preamble in `fp` if present. 

573 

574 `fp` should be positioned at the start of the file-like. If the preamble 

575 and prefix are found then after reading `fp` will be positioned at the 

576 first byte after the prefix (byte offset 133). If either the preamble or 

577 prefix are missing and `force` is ``True`` then after reading `fp` will be 

578 positioned at the start of the file-like. 

579 

580 Parameters 

581 ---------- 

582 fp : file-like object 

583 The file-like to read the preamble from. 

584 force : bool 

585 Flag to force reading of a file even if no header is found. 

586 

587 Returns 

588 ------- 

589 preamble : str/bytes or None 

590 The 128-byte DICOM preamble will be returned if the appropriate prefix 

591 ('DICM') is found at byte offset 128. Returns ``None`` if the 'DICM' 

592 prefix is not found and `force` is ``True``. 

593 

594 Raises 

595 ------ 

596 InvalidDicomError 

597 If `force` is ``False`` and no appropriate header information found. 

598 

599 Notes 

600 ----- 

601 Also reads past the 'DICM' marker. Rewinds file to the beginning if 

602 no header found. 

603 """ 

604 logger.debug("Reading File Meta Information preamble...") 

605 preamble = fp.read(128) 

606 if config.debugging: 

607 sample = bytes2hex(preamble[:8]) + "..." + bytes2hex(preamble[-8:]) 

608 logger.debug("{0:08x}: {1}".format(fp.tell() - 128, sample)) 

609 

610 logger.debug("Reading File Meta Information prefix...") 

611 magic = fp.read(4) 

612 if magic != b"DICM" and force: 

613 logger.info( 

614 "File is not conformant with the DICOM File Format: 'DICM' " 

615 "prefix is missing from the File Meta Information header " 

616 "or the header itself is missing. Assuming no header and " 

617 "continuing.") 

618 preamble = None 

619 fp.seek(0) 

620 elif magic != b"DICM" and not force: 

621 raise InvalidDicomError("File is missing DICOM File Meta Information " 

622 "header or the 'DICM' prefix is missing from " 

623 "the header. Use force=True to force reading.") 

624 else: 

625 logger.debug("{0:08x}: 'DICM' prefix found".format(fp.tell() - 4)) 

626 return preamble

633def read_partial(fileobj, stop_when=None, defer_size=None, 

634 force=False, specific_tags=None): 

635 """Parse a DICOM file until a condition is met. 

636 

637 Parameters 

638 ---------- 

639 fileobj : a file-like object 

640 Note that the file will not close when the function returns. 

641 stop_when : 

642 Stop condition. See :func:`read_dataset` for more info. 

643 defer_size : int, str, None, optional 

644 See :func:`dcmread` for parameter info. 

645 force : bool 

646 See :func:`dcmread` for parameter info. 

647 specific_tags : list or None 

648 See :func:`dcmread` for parameter info. 

649 

650 Notes 

651 ----- 

652 Use :func:`dcmread` unless you need to stop on some condition other than 

653 reaching pixel data. 

654 

655 Returns 

656 ------- 

657 dataset.FileDataset or dicomdir.DicomDir 

658 The read dataset. 

659 

660 See Also 

661 -------- 

662 dcmread 

663 More generic file reading function. 

664 """ 

665 # Read File Meta Information 

666 

667 # Read preamble (if present) 

668 preamble = read_preamble(fileobj, force) 

669 # Read any File Meta Information group (0002,eeee) elements (if present) 

670 file_meta_dataset = _read_file_meta_info(fileobj) 

671 

672 # Read Dataset 

673 

674 # Read any Command Set group (0000,eeee) elements (if present) 

675 command_set = _read_command_set_elements(fileobj) 

676 

677 # Check to see if there's anything left to read 

678 peek = fileobj.read(1) 

679 if peek != b'': 

680 fileobj.seek(-1, 1) 

681 

682 # `filobj` should be positioned at the start of the dataset by this point. 

683 # Ensure we have appropriate values for `is_implicit_VR` and 

684 # `is_little_endian` before we try decoding. We assume an initial 

685 # transfer syntax of implicit VR little endian and correct it as necessary 

686 is_implicit_VR = True 

687 is_little_endian = True 

688 transfer_syntax = file_meta_dataset.get("TransferSyntaxUID") 

689 if peek == b'': # EOF 

690 pass 

691 elif transfer_syntax is None: # issue 258 

692 # If no TransferSyntaxUID element then we have to try and figure out 

693 # the correct values for `is_little_endian` and `is_implicit_VR`. 

694 # Peek at the first 6 bytes to get the first element's tag group and 

695 # (possibly) VR 

696 group, _, VR = unpack("<HH2s", fileobj.read(6)) 

697 fileobj.seek(-6, 1) 

698 

699 # Test the VR to see if it's valid, and if so then assume explicit VR 

700 from pydicom.values import converters 

701 VR = VR.decode(default_encoding) 

702 if VR in converters.keys(): 

703 is_implicit_VR = False 

704 # Big endian encoding can only be explicit VR 

705 # Big endian 0x0004 decoded as little endian will be 1024 

706 # Big endian 0x0100 decoded as little endian will be 1 

707 # Therefore works for big endian tag groups up to 0x00FF after 

708 # which it will fail, in which case we leave it as little endian 

709 # and hope for the best (big endian is retired anyway) 

710 if group >= 1024: 

711 is_little_endian = False 

712 elif transfer_syntax == pydicom.uid.ImplicitVRLittleEndian: 

713 pass 

714 elif transfer_syntax == pydicom.uid.ExplicitVRLittleEndian: 

715 is_implicit_VR = False 

716 elif transfer_syntax == pydicom.uid.ExplicitVRBigEndian: 

717 is_implicit_VR = False 

718 is_little_endian = False 

719 elif transfer_syntax == pydicom.uid.DeflatedExplicitVRLittleEndian: 

720 # See PS3.5 section A.5 

721 # when written, the entire dataset following 

722 # the file metadata was prepared the normal way, 

723 # then "deflate" compression applied. 

724 # All that is needed here is to decompress and then 

725 # use as normal in a file-like object 

726 zipped = fileobj.read() 

727 # -MAX_WBITS part is from comp.lang.python answer: 

728 # groups.google.com/group/comp.lang.python/msg/e95b3b38a71e6799 

729 unzipped = zlib.decompress(zipped, -zlib.MAX_WBITS) 

730 fileobj = BytesIO(unzipped) # a file-like object 

731 is_implicit_VR = False 

732 else: 

733 # Any other syntax should be Explicit VR Little Endian, 

734 # e.g. all Encapsulated (JPEG etc) are ExplVR-LE 

735 # by Standard PS 3.5-2008 A.4 (p63) 

736 is_implicit_VR = False 

737 

738 # Try and decode the dataset 

739 # By this point we should be at the start of the dataset and have 

740 # the transfer syntax (whether read from the file meta or guessed at) 

741 try: 

742 dataset = read_dataset(fileobj, is_implicit_VR, is_little_endian, 

743 stop_when=stop_when, defer_size=defer_size, 

744 specific_tags=specific_tags) 

745 except EOFError: 

746 if config.enforce_valid_values: 

747 raise 

748 # warning already logged in read_dataset 

749 

750 # Add the command set elements to the dataset (if any) 

751 dataset.update(command_set._dict) 

752 

753 class_uid = file_meta_dataset.get("MediaStorageSOPClassUID", None) 

754 if class_uid and class_uid.name == "Media Storage Directory Storage": 

755 dataset_class = DicomDir 

756 else: 

757 dataset_class = FileDataset 

758 new_dataset = dataset_class(fileobj, dataset, preamble, file_meta_dataset, 

759 is_implicit_VR, is_little_endian) 

760 # save the originally read transfer syntax properties in the dataset 

761 new_dataset.set_original_encoding(is_implicit_VR, is_little_endian, 

762 dataset._character_set) 

763 return new_dataset

766def dcmread(fp, defer_size=None, stop_before_pixels=False, 

767 force=False, specific_tags=None): 

768 """Read and parse a DICOM dataset stored in the DICOM File Format. 

769 

770 Read a DICOM dataset stored in accordance with the :dcm:`DICOM File 

771 Format <part10/chapter_7.html>`. If the dataset is not stored in 

772 accordance with the File Format (i.e. the preamble and prefix are missing, 

773 there are missing required Type 1 *File Meta Information Group* elements 

774 or the entire *File Meta Information* is missing) then you will have to 

775 set `force` to ``True``. 

776 

777 Parameters 

778 ---------- 

779 fp : str or PathLike or file-like 

780 Either a file-like object, or a string containing the file name. If a 

781 file-like object, the caller is responsible for closing it. 

782 defer_size : int or str or None, optional 

783 If ``None`` (default), all elements are read into memory. If specified, 

784 then if a data element's stored value is larger than `defer_size`, the 

785 value is not read into memory until it is accessed in code. Specify an 

786 integer (bytes), or a string value with units, e.g. "512 KB", "2 MB". 

787 stop_before_pixels : bool, optional 

788 If ``False`` (default), the full file will be read and parsed. Set 

789 ``True`` to stop before reading (7FE0,0010) *Pixel Data* (and all 

790 subsequent elements). 

791 force : bool, optional 

792 If ``False`` (default), raises an 

793 :class:`~pydicom.errors.InvalidDicomError` if the file is 

794 missing the *File Meta Information* header. Set to ``True`` to force 

795 reading even if no *File Meta Information* header is found. 

796 specific_tags : list or None, optional 

797 If not ``None``, only the tags in the list are returned. The list 

798 elements can be tags or tag names. Note that the element (0008,0005) 

799 *Specific Character Set* is always returned if present - this ensures 

800 correct decoding of returned text values. 

801 

802 Returns 

803 ------- 

804 FileDataset 

805 An instance of :class:`~pydicom.dataset.FileDataset` that represents 

806 a parsed DICOM file. 

807 

808 Raises 

809 ------ 

810 InvalidDicomError 

811 If `force` is ``True`` and the file is not a valid DICOM file. 

812 

813 See Also 

814 -------- 

815 pydicom.dataset.FileDataset 

816 Data class that is returned. 

817 pydicom.filereader.read_partial 

818 Only read part of a DICOM file, stopping on given conditions. 

819 

820 Examples 

821 -------- 

822 Read and return a dataset stored in accordance with the DICOM File Format: 

823 

824 >>> ds = pydicom.dcmread("rtplan.dcm") 

825 >>> ds.PatientName 

826 

827 Read and return a dataset not in accordance with the DICOM File Format: 

828 

829 >>> ds = pydicom.dcmread("rtplan.dcm", force=True) 

830 >>> ds.PatientName 

831 

832 Use within a context manager: 

833 

834 >>> with pydicom.dcmread("rtplan.dcm") as ds: 

835 >>> ds.PatientName 

836 """ 

837 # Open file if not already a file object 

838 caller_owns_file = True 

839 fp = path_from_pathlike(fp) 

840 if isinstance(fp, str): 

841 # caller provided a file name; we own the file handle 

842 caller_owns_file = False 

843 logger.debug("Reading file '{0}'".format(fp)) 

844 fp = open(fp, 'rb') 

845 

846 if config.debugging: 

847 logger.debug("\n" + "-" * 80) 

848 logger.debug("Call to dcmread()") 

849 msg = ("filename:'%s', defer_size='%s', " 

850 "stop_before_pixels=%s, force=%s, specific_tags=%s") 

851 logger.debug(msg % (fp.name, defer_size, stop_before_pixels, 

852 force, specific_tags)) 

853 if caller_owns_file: 

854 logger.debug("Caller passed file object") 

855 else: 

856 logger.debug("Caller passed file name") 

857 logger.debug("-" * 80) 

858 

859 # Convert size to defer reading into bytes 

860 defer_size = size_in_bytes(defer_size) 

861 

862 # Iterate through all items and store them --include file meta if present 

863 stop_when = None 

864 if stop_before_pixels: 

865 stop_when = _at_pixel_data 

866 try: 

867 dataset = read_partial(fp, stop_when, defer_size=defer_size, 

868 force=force, specific_tags=specific_tags) 

869 finally: 

870 if not caller_owns_file: 

871 fp.close() 

872 # XXX need to store transfer syntax etc. 

873 return dataset

910def data_element_offset_to_value(is_implicit_VR, VR): 

911 """Return number of bytes from start of data element to start of value""" 

912 if is_implicit_VR: 

913 offset = 8 # tag of 4 plus 4-byte length 

914 else: 

915 if VR in extra_length_VRs: 

916 offset = 12 # tag 4 + 2 VR + 2 reserved + 4 length 

917 else:

919 return offset

922def read_deferred_data_element(fileobj_type, filename_or_obj, timestamp, 

923 raw_data_elem): 

924 """Read the previously deferred value from the file into memory 

925 and return a raw data element. 

926 

927 .. note: 

928 

929 This is called internally by pydicom and will normally not be 

930 needed in user code. 

931 

932 Parameters 

933 ---------- 

934 fileobj_type : type 

935 The type of the original file object. 

936 filename_or_obj : str or file-like 

937 The filename of the original file if one exists, or the file-like 

938 object where the data element persists. 

939 timestamp : time or None 

940 The time the original file has been read, if not a file-like. 

941 raw_data_elem : dataelem.RawDataElement 

942 The raw data element with no value set. 

943 

944 Returns 

945 ------- 

946 dataelem.RawDataElement 

947 The data element with the value set. 

948 

949 Raises 

950 ------ 

951 IOError 

952 If `filename_or_obj` is ``None``. 

953 IOError 

954 If `filename_or_obj` is a filename and the corresponding file does 

955 not exist. 

956 ValueError 

957 If the VR or tag of `raw_data_elem` does not match the read value. 

958 """ 

959 logger.debug("Reading deferred element %r" % str(raw_data_elem.tag)) 

960 # If it wasn't read from a file, then return an error 

961 if filename_or_obj is None: 

962 raise IOError("Deferred read -- original filename not stored. " 

963 "Cannot re-open") 

964 is_filename = isinstance(filename_or_obj, str) 

965 

966 # Check that the file is the same as when originally read 

967 if is_filename and not os.path.exists(filename_or_obj): 

968 raise IOError("Deferred read -- original file " 

969 "{0:s} is missing".format(filename_or_obj)) 

970 if timestamp is not None: 

971 statinfo = os.stat(filename_or_obj) 

972 if statinfo.st_mtime != timestamp: 

973 warnings.warn("Deferred read warning -- file modification time " 

974 "has changed.") 

975 

976 # Open the file, position to the right place 

977 fp = (fileobj_type(filename_or_obj, 'rb') 

978 if is_filename else filename_or_obj) 

979 is_implicit_VR = raw_data_elem.is_implicit_VR 

980 is_little_endian = raw_data_elem.is_little_endian 

981 offset = data_element_offset_to_value(is_implicit_VR, raw_data_elem.VR) 

982 fp.seek(raw_data_elem.value_tell - offset) 

983 elem_gen = data_element_generator(fp, is_implicit_VR, is_little_endian, 

984 defer_size=None) 

985 

986 # Read the data element and check matches what was stored before 

987 data_elem = next(elem_gen) 

988 fp.close() 

989 if data_elem.VR != raw_data_elem.VR: 

990 raise ValueError("Deferred read VR {0:s} does not match " 

991 "original {1:s}".format(data_elem.VR, 

992 raw_data_elem.VR)) 

993 if data_elem.tag != raw_data_elem.tag: 

994 raise ValueError("Deferred read tag {0!r} does not match " 

995 "original {1!r}".format(data_elem.tag, 

996 raw_data_elem.tag)) 

997 

998 # Everything is ok, now this object should act like usual DataElement 

999 return data_elem
