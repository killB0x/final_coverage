28# Precompiled search patterns for camel_to_underscore() 

29first_cap_re = re.compile('(.)([A-Z][a-z]+)') 

30all_cap_re = re.compile('([a-z0-9])([A-Z])')

32byte_VRs = [

38def camel_to_underscore(name): 

39 """Convert name from CamelCase to lower_case_with_underscores""" 

40 # From http://stackoverflow.com/questions/1175208 

41 s1 = first_cap_re.sub(r'\1_\2', name) 

42 return all_cap_re.sub(r'\1_\2', s1).lower()

45def tag_repr(tag): 

46 """String of tag value as (0xgggg, 0xeeee)""" 

47 return "(0x{group:04x}, 0x{elem:04x})".format( 

48 group=tag.group, elem=tag.element)

51def default_name_filter(name): 

52 """Callable to reduce some names in code to more readable short form 

53 

54 :arg name: a sequence variable name or sequence item name 

55 :return: a shorter version of name if a known conversion, 

56 else return original name 

57 

58 """ 

59 name = camel_to_underscore(name) 

60 name = name.replace("control_point", "cp") 

61 name = name.replace("reference", "ref") 

62 name = name.replace("fraction_group", "frxn_gp") 

63 return name

79def code_dataelem(dataelem, 

80 dataset_name="ds", 

81 exclude_size=None, 

82 include_private=False): 

83 """Code lines for a single DICOM data element 

84 

85 :arg dataelem: the DataElement instance to turn into code 

86 :arg dataset_name: variable name of the Dataset containing dataelem 

87 :arg exclude_size: if specified, values longer than this (in bytes) 

88 will only have a commented string for a value, 

89 causing a syntax error when the code is run, 

90 and thus prompting the user to remove or fix that line. 

91 :return: a string containing code to recreate the data element 

92 If the data element is a sequence, calls code_sequence 

93 

94 """ 

95 

96 if dataelem.VR == "SQ": 

97 return code_sequence(dataelem, dataset_name, exclude_size, 

98 include_private) 

99 

100 # If in DICOM dictionary, set using the keyword 

101 # If not (e.g. is private element), set using add_new method 

102 have_keyword = True 

103 try: 

104 keyword = dictionary_keyword(dataelem.tag) 

105 except KeyError: 

106 have_keyword = False 

107 

108 valuerep = repr(dataelem.value) 

109 

110 if exclude_size: 

111 if (dataelem.VR in byte_VRs and 

112 len(dataelem.value) > exclude_size): 

113 valuerep = ( 

114 "# XXX Array of %d bytes excluded" % len(dataelem.value)) 

115 

116 if have_keyword: 

117 format_str = "{ds_name}.{keyword} = {valuerep}" 

118 line = format_str.format( 

119 ds_name=dataset_name, keyword=keyword, valuerep=valuerep) 

120 else: 

121 format_str = "{ds_name}.add_new({tag}, '{VR}', {valuerep})" 

122 line = format_str.format( 

123 ds_name=dataset_name, 

124 tag=tag_repr(dataelem.tag), 

125 VR=dataelem.VR, 

126 valuerep=valuerep) 

127 return line

130def code_sequence(dataelem, 

131 dataset_name="ds", 

132 exclude_size=None, 

133 include_private=False, 

134 name_filter=default_name_filter): 

135 """Code lines for recreating a Sequence data element 

136 

137 :arg dataelem: the DataElement instance of the Sequence 

138 :arg dataset_name: variable name of the dataset containing the Sequence 

139 :arg exclude_size: if specified, values longer than this (in bytes) 

140 will only have a commented string for a value, 

141 causing a syntax error when the code is run, 

142 and thus prompting the user to remove or fix that line. 

143 :arg include_private: If True, private data elements will be coded. 

144 If False, private elements are skipped 

145 :arg name_filter: a callable taking a sequence name or sequence item name, 

146 and returning a shorter name for easier code reading 

147 :return: a string containing code lines to recreate a DICOM sequence 

148 

149 """ 

150 lines = [] 

151 seq = dataelem.value 

152 seq_name = dataelem.name 

153 seq_item_name = seq_name.replace(' Sequence', '') 

154 seq_keyword = dictionary_keyword(dataelem.tag) 

155 

156 # Create comment line to document the start of Sequence 

157 lines.append('') 

158 lines.append("# " + seq_name) 

159 

160 # Code line to create a new Sequence object 

161 if name_filter: 

162 seq_var = name_filter(seq_keyword) 

163 lines.append(seq_var + " = Sequence()") 

164 

165 # Code line to add the sequence to its parent 

166 lines.append(dataset_name + "." + seq_keyword + " = " + seq_var) 

167 

168 # Code lines to add sequence items to the Sequence 

169 for i, ds in enumerate(seq): 

170 # Determine index to use. If seq item has a data element with 'Index', 

171 # use that; if one with 'Number', use that, else start at 1 

172 index_keyword = seq_keyword.replace("Sequence", "") + "Index" 

173 number_keyword = seq_keyword.replace("Sequence", "") + "Number" 

174 if index_keyword in ds: 

175 index_str = str(getattr(ds, index_keyword)) 

176 elif number_keyword in ds: 

177 index_str = str(getattr(ds, number_keyword)) 

178 else: 

179 index_str = str(i + 1) 

180 

181 # Code comment line to mark start of sequence item 

182 lines.append('') 

183 lines.append("# " + seq_name + ": " + seq_item_name + " " + index_str) 

184 

185 # Determine the variable name to use for the sequence item (dataset) 

186 ds_name = seq_var.replace("_sequence", "") + index_str 

187 

188 # Code the sequence item 

189 code_item = code_dataset(ds, ds_name, exclude_size, include_private) 

190 lines.append(code_item) 

191 

192 # Code the line to append the item to its parent sequence 

193 lines.append(seq_var + ".append(" + ds_name + ")") 

194 

195 # Join the lines and return a single string 

196 return line_term.join(lines)

199def code_dataset(ds, 

200 dataset_name="ds", 

201 exclude_size=None, 

202 include_private=False, 

203 is_file_meta=False): 

204 """Return python code lines for import statements needed by other code 

205 

206 :arg exclude_size: if specified, values longer than this (in bytes) 

207 will only have a commented string for a value, 

208 causing a syntax error when the code is run, 

209 and thus prompting the user to remove or fix that line. 

210 :arg include_private: If True, private data elements will be coded. 

211 If False, private elements are skipped 

212 :return: a list of code lines containing import statements 

213 

214 """ 

215 lines = [] 

216 ds_class = " = FileMetaDataset()" if is_file_meta else " = Dataset()" 

217 lines.append(dataset_name + ds_class) 

218 for dataelem in ds: 

219 # If a private data element and flag says so, skip it and go to next 

220 if not include_private and dataelem.tag.is_private: 

221 continue 

222 # Otherwise code the line and add it to the lines list 

223 code_line = code_dataelem(dataelem, dataset_name, exclude_size, 

224 include_private) 

225 lines.append(code_line) 

226 # Add blank line if just coded a sequence 

227 if dataelem.VR == "SQ": 

228 lines.append('') 

229 # If sequence was end of this dataset, remove the extra blank line 

230 if len(lines) and lines[-1] == '': 

231 lines.pop() 

232 # Join all the code lines and return them 

233 return line_term.join(lines)

236def code_file(filename, exclude_size=None, include_private=False): 

237 """Write a complete source code file to recreate a DICOM file 

238 

239 :arg filename: complete path and filename of a DICOM file to convert 

240 :arg exclude_size: if specified, values longer than this (in bytes) 

241 will only have a commented string for a value, 

242 causing a syntax error when the code is run, 

243 and thus prompting the user to remove or fix that line. 

244 :arg include_private: If True, private data elements will be coded. 

245 If False, private elements are skipped 

246 :return: a string containing code lines to recreate entire file 

247 

248 """ 

249 lines = [] 

250 

251 ds = pydicom.dcmread(filename, force=True) 

252 

253 # Code a nice header for the python file 

254 lines.append("# Coded version of DICOM file '{0}'".format(filename)) 

255 lines.append("# Produced by pydicom codify utility script") 

256 

257 # Code the necessary imports 

258 lines.append(code_imports()) 

259 lines.append('') 

260 

261 # Code the file_meta information 

262 lines.append("# File meta info data elements") 

263 code_meta = code_dataset(ds.file_meta, "file_meta", exclude_size, 

264 include_private, is_file_meta=True) 

265 lines.append(code_meta) 

266 lines.append('') 

267 

268 # Code the main dataset 

269 lines.append("# Main data elements") 

270 code_ds = code_dataset( 

271 ds, exclude_size=exclude_size, include_private=include_private) 

272 lines.append(code_ds) 

273 lines.append('') 

274 

275 # Add the file meta to the dataset, and set transfer syntax 

276 lines.append("ds.file_meta = file_meta") 

277 lines.append("ds.is_implicit_VR = " + str(ds.is_implicit_VR)) 

278 lines.append("ds.is_little_endian = " + str(ds.is_little_endian)) 

279 

280 # Return the complete code string 

281 return line_term.join(lines)

284def main(default_exclude_size, args=None): 

285 """Create python code according to user options 

286 

287 Parameters: 

288 ----------- 

289 default_exclude_size: int 

290 Values longer than this will be coded as a commented syntax error 

291 

292 args: list 

293 Command-line arguments to parse. If None, then sys.argv is used 

294 """ 

295 

296 try: 

297 import argparse 

298 except ImportError: 

299 print("The argparse module is required to run this script") 

300 print("argparse is standard in python >= 2.7,") 

301 print(" or can be installed with 'pip install argparse'") 

302 sys.exit(-1) 

303 

304 parser = argparse.ArgumentParser( 

305 description="Produce python/pydicom code from a DICOM file", 

306 epilog="Binary data (e.g. pixels) larger than --exclude-size " 

307 "(default %d bytes) is not included. A dummy line " 

308 "with a syntax error is produced. " 

309 "Private data elements are not included " 

310 "by default." % default_exclude_size) 

311 parser.add_argument( 

312 'infile', help="DICOM file from which to produce code lines") 

313 parser.add_argument( 

314 'outfile', 

315 nargs='?', 

316 type=argparse.FileType('w'), 

317 help=("Filename to write python code to. " 

318 "If not specified, code is written to stdout"), 

319 default=sys.stdout) 

320 help_exclude_size = 'Exclude binary data larger than specified (bytes). ' 

321 help_exclude_size += 'Default is %d bytes' % default_exclude_size 

322 parser.add_argument( 

323 '-e', 

324 '--exclude-size', 

325 type=int, 

326 default=default_exclude_size, 

327 help=help_exclude_size) 

328 parser.add_argument( 

329 '-p', 

330 '--include-private', 

331 action="store_true", 

332 help='Include private data elements ' 

333 '(default is to exclude them)') 

334 parser.add_argument( 

335 '-s', 

336 '--save-as', 

337 help=("Specify the filename for ds.save_as(save_filename); " 

338 "otherwise the input name + '_from_codify' will be used")) 

339 

340 args = parser.parse_args(args) 

341 

342 # Read the requested file and convert to python/pydicom code lines 

343 filename = args.infile # name 

344 code_lines = code_file(filename, args.exclude_size, args.include_private) 

345 

346 # If requested, write a code line to save the dataset 

347 if args.save_as: 

348 save_as_filename = args.save_as 

349 else: 

350 base, ext = os.path.splitext(filename) 

351 save_as_filename = base + "_from_codify" + ".dcm" 

352 line = "\nds.save_as(r'{filename}', write_like_original=False)" 

353 save_line = line.format(filename=save_as_filename) 

354 code_lines += save_line 

355 

356 # Write the code lines to specified file or to standard output 

357 # For test_util, captured output .name throws error, ignore it: 

358 try: 

359 if args.outfile.name != "<stdout>": 

360 print("Writing code to file '%s'" % args.outfile.name) 

361 except AttributeError: 

362 pass 

363 args.outfile.write(code_lines) 

364 

365 

366if __name__ == "__main__": 

367 main(default_exclude_size=100)
