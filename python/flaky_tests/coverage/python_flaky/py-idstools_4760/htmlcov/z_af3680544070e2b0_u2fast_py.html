38import time 

39import logging

41try: 

42 import argparse

44 from idstools.compat.argparse import argparse

58def print_time(sec, usec): 

59 tt = time.localtime(sec) 

60 return "%04d/%02d/%02d-%02d:%02d:%02d.%06d" % ( 

61 tt.tm_year, tt.tm_mon, tt.tm_mday, tt.tm_hour, tt.tm_min, tt.tm_sec, 

62 usec)

64def print_event(event, msgmap, classmap): 

65 msg_entry = msgmap.get(event["generator-id"], event["signature-id"]) 

66 if msg_entry: 

67 msg = msg_entry["msg"] 

68 else: 

69 msg = "Snort Event" 

70 

71 class_entry = classmap.get(event["classification-id"]) 

72 if class_entry: 

73 class_description = class_entry["description"] 

74 else: 

75 class_description = str(event["classification-id"]) 

76 

77 proto = proto_map.get(event["protocol"], str(event["protocol"])) 

78 

79 print("%s [**] [%d:%d:%d] %s [**] [Classification: %s] [Priority: %d] {%s} %s:%d -> %s:%d" % ( 

80 print_time(event["event-second"], event["event-microsecond"]), 

81 event["generator-id"], 

82 event["signature-id"], 

83 event["signature-revision"], 

84 msg, 

85 class_description, 

86 event["priority"], 

87 proto, 

88 event["source-ip"], 

89 event["sport-itype"], 

90 event["destination-ip"], 

91 event["dport-icode"], 

92 ))

94def load_from_snort_conf(snort_conf, classmap, msgmap): 

95 snort_etc = os.path.dirname(snort_conf) 

96 

97 classification_config = os.path.join(snort_etc, "classification.config") 

98 if os.path.exists(classification_config): 

99 LOG.debug("Loading %s.", classification_config) 

100 classmap.load_from_file(open(classification_config)) 

101 

102 genmsg_map = os.path.join(snort_etc, "gen-msg.map") 

103 if os.path.exists(genmsg_map): 

104 LOG.debug("Loading %s.", genmsg_map) 

105 msgmap.load_generator_map(open(genmsg_map)) 

106 

107 sidmsg_map = os.path.join(snort_etc, "sid-msg.map") 

108 if os.path.exists(sidmsg_map): 

109 LOG.debug("Loading %s.", sidmsg_map) 

110 msgmap.load_signature_map(open(sidmsg_map))

112def main(): 

113 

114 msgmap = maps.SignatureMap() 

115 classmap = maps.ClassificationMap() 

116 

117 parser = argparse.ArgumentParser( 

118 fromfile_prefix_chars='@') 

119 parser.add_argument( 

120 "-C", dest="classification_path", metavar="<classification.config>", 

121 help="path to classification config") 

122 parser.add_argument( 

123 "-S", dest="sidmsgmap_path", metavar="<msg-msg.map>", 

124 help="path to sid-msg.map") 

125 parser.add_argument( 

126 "-G", dest="genmsgmap_path", metavar="<gen-msg.map>", 

127 help="path to gen-msg.map") 

128 parser.add_argument( 

129 "--snort-conf", dest="snort_conf", metavar="<snort.conf>", 

130 help="attempt to load classifications and map files based on the " 

131 "location of the snort.conf") 

132 parser.add_argument( 

133 "--directory", metavar="<spool directory>", 

134 help="spool directory (eg: /var/log/snort)") 

135 parser.add_argument( 

136 "--prefix", metavar="<spool file prefix>", 

137 help="spool filename prefix (eg: unified2.log)") 

138 parser.add_argument( 

139 "--bookmark", action="store_true", default=False, 

140 help="enable bookmarking") 

141 parser.add_argument( 

142 "--follow", action="store_true", default=False, 

143 help="follow files/continuous mode (spool mode only)") 

144 parser.add_argument( 

145 "filenames", nargs="*") 

146 args = parser.parse_args() 

147 

148 if args.snort_conf: 

149 load_from_snort_conf(args.snort_conf, classmap, msgmap) 

150 

151 if args.classification_path: 

152 classmap.load_from_file( 

153 open(os.path.expanduser(args.classification_path))) 

154 if args.genmsgmap_path: 

155 msgmap.load_generator_map(open(os.path.expanduser(args.genmsgmap_path))) 

156 if args.sidmsgmap_path: 

157 msgmap.load_signature_map(open(os.path.expanduser(args.sidmsgmap_path))) 

158 

159 if msgmap.size() == 0: 

160 LOG.warn("WARNING: No alert message map entries loaded.") 

161 else: 

162 LOG.info("Loaded %s rule message map entries.", msgmap.size()) 

163 

164 if classmap.size() == 0: 

165 LOG.warn("WARNING: No classifications loaded.") 

166 else: 

167 LOG.info("Loaded %s classifications.", classmap.size()) 

168 

169 if args.directory and args.prefix: 

170 reader = unified2.SpoolEventReader( 

171 directory=args.directory, 

172 prefix=args.prefix, 

173 follow=args.follow, 

174 bookmark=args.bookmark) 

175 

176 for event in reader: 

177 print_event(event, msgmap, classmap) 

178 

179 elif args.filenames: 

180 reader = unified2.FileEventReader(*args.filenames) 

181 for event in reader: 

182 print_event(event, msgmap, classmap) 

183 

184 else: 

185 parser.print_help() 

186 return 1 

187 

188if __name__ == "__main__": 

189 sys.exit(main())
