142 # typedef struct { 

143 # UINT32 Version; 

144 # 

145 # /// 

146 # /// The number of drivers included in the capsule and the number of corresponding 

147 # /// offsets stored in ItemOffsetList array. 

148 # /// 

149 # UINT16 EmbeddedDriverCount; 

150 # 

151 # /// 

152 # /// The number of payload items included in the capsule and the number of 

153 # /// corresponding offsets stored in the ItemOffsetList array. 

154 # /// 

155 # UINT16 PayloadItemCount; 

156 # 

157 # /// 

158 # /// Variable length array of dimension [EmbeddedDriverCount + PayloadItemCount] 

159 # /// containing offsets of each of the drivers and payload items contained within the capsule 

160 # /// 

161 # // UINT64 ItemOffsetList[]; 

162 # } EFI_FIRMWARE_MANAGEMENT_CAPSULE_HEADER;

206 offset_list = [] 

207 Offset = self._StructSize + (self.EmbeddedDriverCount + self.PayloadItemCount) * self._ItemOffsetSize

209 FmpCapsuleData = FmpCapsuleData + EmbeddedDriver

211 Offset = Offset + len(EmbeddedDriver) 

212 for FmpCapsuleImageHeader in self._FmpCapsuleImageHeaderList: 

213 FmpCapsuleImage = FmpCapsuleImageHeader.Encode()

223 def Decode(self, Buffer): 

224 if len(Buffer) < self._StructSize: 

225 raise ValueError 

226 (Version, EmbeddedDriverCount, PayloadItemCount) = struct.unpack( 

227 self._StructFormat,

234 self.EmbeddedDriverCount = EmbeddedDriverCount 

235 self.PayloadItemCount = PayloadItemCount 

236 self._EmbeddedDriverList = [] 

237 self._FmpCapsuleImageHeaderList = []

239 offset_list = [] 

240 

241 # 

242 # Parse the ItemOffsetList values 

243 # 

244 Offset = self._StructSize 

245 for Index in range(0, EmbeddedDriverCount + PayloadItemCount): 

246 ItemOffset = struct.unpack(self._ItemOffsetFormat, Buffer[Offset:Offset + self._ItemOffsetSize])[0] 

247 if ItemOffset >= len(Buffer): 

248 raise ValueError 

249 offset_list.append(ItemOffset) 

250 Offset = Offset + self._ItemOffsetSize

253 # 

254 # Parse the EmbeddedDrivers 

255 # 

256 for Index in range(0, EmbeddedDriverCount): 

257 Offset = offset_list[Index] 

258 if Index < (len(offset_list) - 1): 

259 Length = offset_list[Index + 1] - Offset 

260 else: 

261 Length = len(Buffer) - Offset 

262 self.AddEmbeddedDriver(Buffer[Offset:Offset + Length]) 

263 

264 # 

265 # Parse the Payloads that are FMP Capsule Images 

266 # 

267 for Index in range(EmbeddedDriverCount, EmbeddedDriverCount + PayloadItemCount): 

268 Offset = offset_list[Index]

271 else:

274 FmpCapsuleImageHeader.Decode(Buffer[Offset:Offset + Length]) 

275 self.AddFmpCapsuleImageHeader(FmpCapsuleImageHeader) 

276 

277 return Result
