33class DiskInfo(object): 

34 @staticmethod 

35 def invert_dict_nesting(origin): 

36 """ 

37 origin = { 

38 'x': {'xx': 1, 'yy': 2}, 

39 'y': {'xx': 3, 'yy': 4}, 

40 } 

41 result = { 

42 'xx': {'x': 1, 'y': 3}, 

43 'yy': {'x': 2, 'y': 4}, 

44 } 

45 """ 

46 result = defaultdict() 

47 for out_key, out_value in origin.items(): 

48 for in_key, in_value in out_value.items(): 

49 result.setdefault(in_key, dict()) 

50 result[in_key][out_key] = in_value 

51 return dict(result)
