12def online_kmeans(X, metric, n_clusters, n_repetitions=20, 

13 tolerance=1e-5, max_iter=5e4): 

14 """Perform online K-means clustering. 

15 

16 Perform online version of k-means algorithm on data contained in X. 

17 The data points are treated sequentially and the cluster centers are 

18 updated one at a time. This version of k-means avoids computing the 

19 mean of each cluster at each iteration and is therefore less 

20 computationally intensive than the offline version. 

21 

22 In the setting of quantization of probability distributions, this 

23 algorithm is also known as Competitive Learning Riemannian Quantization. 

24 It computes the closest approximation of the empirical distribution of 

25 data by a discrete distribution supported by a smaller number of points 

26 with respect to the Wasserstein distance. This smaller number of points 

27 is n_clusters. 

28 

29 Parameters 

30 ---------- 

31 X : array-like, shape=[..., n_features] 

32 Input data. It is treated sequentially by the algorithm, i.e. 

33 one datum is chosen randomly at each iteration. 

34 metric : object 

35 Metric of the space in which the data lives. At each iteration, 

36 one of the cluster centers is moved in the direction of the new 

37 datum, according the exponential map of the underlying space, which 

38 is a method of metric. 

39 n_clusters : int 

40 Number of clusters of the k-means clustering, or number of desired 

41 atoms of the quantized distribution. 

42 n_repetitions : int, default=20 

43 The cluster centers are updated using decreasing step sizes, each 

44 of which stays constant for n_repetitions iterations to allow a better 

45 exploration of the data points. 

46 max_iter : int, default=5e4 

47 Maximum number of iterations. If it is reached, the 

48 quantization may be inacurate. 

49 

50 Returns 

51 ------- 

52 cluster_centers : array, shape=[n_clusters, n_features] 

53 Coordinates of cluster centers. 

54 labels : array, shape=[n_samples] 

55 Cluster labels for each point. 

56 """ 

57 n_samples = X.shape[0] 

58 

59 random_indices = gs.random.randint(low=0, high=n_samples, 

60 size=(n_clusters,)) 

61 cluster_centers = gs.get_slice(X, gs.cast(random_indices, gs.int32)) 

62 

63 gap = 1.0 

64 iteration = 0 

65 

66 while iteration < max_iter: 

67 iteration += 1 

68 step_size = gs.floor(gs.array(iteration / n_repetitions)) + 1 

69 

70 random_index = gs.random.randint(low=0, high=n_samples, size=(1,)) 

71 point = gs.get_slice(X, gs.cast(random_index, gs.int32)) 

72 

73 index_to_update = metric.closest_neighbor_index(point, cluster_centers) 

74 center_to_update = gs.copy( 

75 gs.get_slice(cluster_centers, index_to_update)) 

76 

77 tangent_vec_update = metric.log( 

78 point=point, base_point=center_to_update 

79 ) / (step_size + 1) 

80 new_center = metric.exp( 

81 tangent_vec=tangent_vec_update, 

82 base_point=center_to_update 

83 ) 

84 gap = metric.dist(center_to_update, new_center) 

85 if gap == 0 and iteration == 1: 

86 gap = gs.array(1.0) 

87 

88 cluster_centers[index_to_update, :] = new_center 

89 

90 if gs.isclose(gap, 0, atol=tolerance): 

91 break 

92 

93 if iteration == max_iter - 1: 

94 logging.warning( 

95 'Maximum number of iterations {} reached. The' 

96 'clustering may be inaccurate'.format(max_iter)) 

97 

98 labels = gs.zeros(n_samples) 

99 for i in range(n_samples): 

100 labels[i] = int(metric.closest_neighbor_index(X[i], cluster_centers)) 

101 

102 return cluster_centers, labels

105class OnlineKMeans(BaseEstimator, ClusterMixin): 

106 """Online k-means clustering. 

107 

108 Online k-means clustering seeks to divide a set of data points into 

109 a specified number of classes, while minimizing intra-class variance. 

110 It is closely linked to discrete quantization, which computes the closest 

111 approximation of the empirical distribution of the dataset by a discrete 

112 distribution supported by a smaller number of points with respect to the 

113 Wasserstein distance. The algorithm used can either be seen as an online 

114 version of the k-means algorithm or as Competitive Learning Riemannian 

115 Quantization (see [LBP2019]_). 

116 

117 Parameters 

118 ---------- 

119 metric : object 

120 Metric of the space in which the data lives. At each iteration, 

121 one of the cluster centers is moved in the direction of the new 

122 datum, according the exponential map of the underlying space, which 

123 is a method of metric. 

124 n_clusters : int 

125 Number of clusters of the k-means clustering, or number of desired 

126 atoms of the quantized distribution. 

127 n_repetitions : int, default=20 

128 The cluster centers are updated using decreasing step sizes, each 

129 of which stays constant for n_repetitions iterations to allow a better 

130 exploration of the data points. 

131 max_iter : int, default=5e4 

132 Maximum number of iterations. If it is reached, the 

133 quantization may be inacurate. 

134 

135 Attributes 

136 ---------- 

137 cluster_centers_ : array, [n_clusters, n_features] 

138 Coordinates of cluster centers. 

139 labels_ : 

140 Labels of each point. 

141 

142 Example 

143 ------- 

144 >>> from geomstats.geometry.hypersphere import Hypersphere 

145 >>> from geomstats.learning.onlinekmeans import OnlineKmeans 

146 >>> sphere = Hypersphere(dim=2) 

147 >>> metric = sphere.metric 

148 >>> X = sphere.random_von_mises_fisher(kappa=10, n_samples=50) 

149 >>> clustering = OnlineKmeans(metric=metric,n_clusters=4).fit(X) 

150 >>> clustering.cluster_centers_ 

151 >>> clustering.labels_ 

152 

153 References 

154 ---------- 

155 .. [LBP2019] A. Le Brigant and S. Puechmorel, Optimal Riemannian 

156 quantization with an application to air traffic analysis. J. Multivar. 

157 Anal. 173 (2019), 685 - 703. 

158 """ 

159
