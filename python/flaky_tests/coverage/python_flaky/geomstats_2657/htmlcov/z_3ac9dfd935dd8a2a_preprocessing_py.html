17class ToTangentSpace(BaseEstimator, TransformerMixin): 

18 """Lift data to a tangent space. 

19 

20 Compute the logs of all data points and reshape them to 

21 1d vectors if necessary. This means that all the data points, that belong 

22 to a possibly non-linear manifold are lifted to one of the tangent space of 

23 the manifold, which is a vector space. By default, the mean of the data 

24 is computed (with the FrechetMean or the ExponentialBarycenter estimator, 

25 as appropriate) and the tangent space at the mean is used. Any other base 

26 point can be passed. The data points are then represented by the initial 

27 velocities of the geodesics that lead from base_point to each data point. 

28 Any machine learning algorithm can then be used with the output array. 

29 

30 Parameters 

31 ---------- 

32 geometry : {Manifold ,LieGroup or RiemannianMetric} 

33 Metric or Lie group to use to compute the log and exp. If a Lie group 

34 is passed, its group exp/log will be used, which don't necessarily 

35 correspond to a Riemannian metric. To use a `metric` on the Lie group, 

36 explicitly pass `geometry=metric` 

37 **kwargs : key-word arguments for the FrechetMean/ExponentialBarycenter 

38 estimator. 

39 """ 

40 

41 def __init__(self, geometry, **kwargs): 

42 

43 if isinstance(geometry, LieGroup): 

44 self._used_geometry = geometry 

45 self.estimator = ExponentialBarycenter( 

46 group=self._used_geometry, **kwargs) 

47 else: 

48 if hasattr(geometry, 'metric'): 

49 self._used_geometry = geometry.metric 

50 elif isinstance(geometry, RiemannianMetric): 

51 self._used_geometry = geometry 

52 else: 

53 raise ValueError('The input geometry must be either a ' 

54 'Manifold equipped with a ' 

55 'RiemannianMetric, or a RiemannianMetric or a' 

56 ' LieGroup') 

57 self.estimator = FrechetMean(metric=self._used_geometry, **kwargs) 

58 self.point_type = geometry.default_point_type 

59 self.geometry = geometry

61 def fit(self, X, y=None, weights=None, base_point=None): 

62 """Compute the central point at which to take the log. 

63 

64 This method is only used if `base_point=None` to compute the mean of 

65 the input data. 

66 

67 Parameters 

68 ---------- 

69 X : array-like, shape=[..., {dim, [n, n]}] 

70 The training input samples. 

71 y : array-like, shape (n_samples,) or (n_samples, n_outputs) 

72 Ignored 

73 weights : array-like, shape=[..., 1] 

74 Weights associated to the points. 

75 Optional, default: None 

76 base_point : array-like, shape=[{dim, [n, n]}] 

77 Point similar to the input data from which to compute the logs. 

78 Optional, default: None. 

79 

80 Returns 

81 ------- 

82 self : object 

83 Returns self. 

84 """ 

85 if base_point is None: 

86 self.estimator.fit(X, y, weights) 

87 return self

89 def transform(self, X, base_point=None): 

90 """Lift data to a tangent space. 

91 

92 Compute the logs of all data point and reshapes them to 

93 1d vectors if necessary. By default the logs are taken at the mean 

94 but any other base point can be passed. Any machine learning 

95 algorithm can then be used with the output array. 

96 

97 Parameters 

98 ---------- 

99 X : array-like, shape=[..., {dim, [n, n]}] 

100 Data to transform. 

101 y : Ignored (Compliance with scikit-learn interface) 

102 base_point : array-like, shape={dim, [n,n]}, optional (mean) 

103 Point on the manifold, the returned samples will be tangent 

104 vectors at the base point. 

105 

106 Returns 

107 ------- 

108 X_new : array-like, shape=[..., dim] 

109 Lifted data. 

110 """ 

111 if base_point is None: 

112 base_point = self.estimator.estimate_ 

113 

114 if self.estimator.estimate_ is None: 

115 raise RuntimeError('fit needs to be called first or a ' 

116 'base_point passed.') 

117 

118 tangent_vecs = self._used_geometry.log(X, base_point=base_point) 

119 

120 if self.point_type == 'vector': 

121 return tangent_vecs 

122 

123 if gs.all(Matrices.is_symmetric(tangent_vecs)): 

124 X = SymmetricMatrices.to_vector( 

125 tangent_vecs) 

126 elif gs.all(Matrices.is_skew_symmetric(tangent_vecs)): 

127 X = SkewSymmetricMatrices( 

128 tangent_vecs.shape[-1]).basis_representation(tangent_vecs) 

129 else: 

130 X = gs.reshape(tangent_vecs, (len(X), - 1)) 

131 

132 return X

134 def inverse_transform(self, X, base_point=None): 

135 """Reconstruction of X. 

136 

137 The reconstruction will match X_original whose transform would be X. 

138 

139 Parameters 

140 ---------- 

141 X : array-like, shape=[..., dim] 

142 New data, where dim is the dimension of the manifold data belong 

143 to. 

144 base_point : array-like, shape={dim, [n,n]}, optional (mean) 

145 Point on the manifold, where the input samples are tangent 

146 vectors. 

147 

148 Returns 

149 ------- 

150 X_original : array-like, shape=[..., {dim, [n, n]} 

151 Data lying on the manifold. 

152 """ 

153 if base_point is None: 

154 base_point = self.estimator.estimate_ 

155 

156 if self.estimator.estimate_ is None: 

157 raise RuntimeError('fit needs to be called first or a ' 

158 'base_point passed.') 

159 

160 if self.point_type == 'matrix': 

161 n_base_point = base_point.shape[-1] 

162 n_vecs = X.shape[-1] 

163 dim_sym = int(n_base_point * (n_base_point + 1) / 2) 

164 dim_skew = int(n_base_point * (n_base_point - 1) / 2) 

165 

166 if gs.all(Matrices.is_symmetric(base_point)) and dim_sym == n_vecs: 

167 tangent_vecs = SymmetricMatrices( 

168 base_point.shape[-1]).from_vector(X) 

169 elif dim_skew == n_vecs: 

170 tangent_vecs = SkewSymmetricMatrices( 

171 dim_skew).matrix_representation(X) 

172 else: 

173 dim = base_point.shape[-1] 

174 tangent_vecs = gs.reshape(X, (len(X), dim, dim)) 

175 else: 

176 tangent_vecs = X 

177 return self._used_geometry.exp(tangent_vecs, base_point)
