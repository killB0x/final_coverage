8class TIOEndpoint(APIEndpoint): 

9 def _parse_filters(self, finput, filterset=None, rtype='sjson'): 

10 ''' 

11 A centralized method to parse and munge the filter tuples into the 

12 anticipates response. 

13 

14 Args: 

15 finput (list): The list of filter tuples 

16 filterset (dict): The response of the allowed filters 

17 rtype (str, optional): 

18 The filter format. Allowed types are 'json', 'sjson', 

19 'accessgroup', and 'colon'. JSON is just a simple JSON list of 

20 dictionaries. SJSON format is effectively serialized JSON into 

21 the query params. COLON format denotes a colon-delimited format. 

22 

23 Returns: 

24 dict: 

25 The query parameters in the anticipated dictionary format to 

26 feed to requests. 

27 ''' 

28 resp = dict() 

29 

30 for f in finput: 

31 # First we need to validate the inputs are correct. We will do that 

32 # by comparing the filter to the filterset data we have and compare 

33 # the operators and values to make sure that the input is expected. 

34 fname = self._check('filter_name', f[0], str) 

35 if f[0] not in filterset: 

36 raise UnexpectedValueError( 

37 '{} is not a filterable option'.format(f[0])) 

38 

39 foper = self._check('filter_operator', f[1], str, 

40 choices=filterset.get(f[0], dict()).get('operators')) 

41 

42 if isinstance(f[2], str): 

43 rval = f[2].split(',') 

44 elif isinstance(f[2], list): 

45 rval = f[2] 

46 else: 

47 raise TypeError('filter_value is not a valid type.') 

48 

49 fval = self._check('filter_value', rval, list, 

50 choices=filterset.get(f[0], dict()).get('choices'), 

51 pattern=filterset.get(f[0], dict()).get('pattern')) 

52 if rtype not in ['accessgroup']: 

53 fval = ','.join(fval) 

54 

55 if rtype == 'sjson': 

56 # For the serialized JSON format, we will need to generate the 

57 # expanded input for each filter 

58 i = finput.index(f) 

59 resp['filter.{}.filter'.format(i)] = fname 

60 resp['filter.{}.quality'.format(i)] = foper 

61 resp['filter.{}.value'.format(i)] = fval 

62 elif rtype == 'json': 

63 # for standard JSON formats, we will simply build a 'filters' 

64 # list and store the information there. 

65 if 'filters' not in resp: 

66 resp['filters'] = list() 

67 resp['filters'].append({ 

68 'filter': fname, 

69 'quality': foper, 

70 'value': fval 

71 }) 

72 elif rtype == 'colon': 

73 # for the colon-delimited format, we simply need to generate the 

74 # filter as NAME:OPER:VAL and dump it all into a field named f. 

75 if 'f' not in resp: 

76 resp['f'] = list() 

77 resp['f'].append('{}:{}:{}'.format(fname, foper, fval)) 

78 elif rtype == 'accessgroup': 

79 # For the access group format, we will instead use the format of 

80 # "terms", "type", and "operator". Further all terms must be a 

81 # list of strings. 

82 if 'rules' not in resp: 

83 resp['rules'] = list() 

84 resp['rules'].append({ 

85 'operator': foper, 

86 'terms': fval, 

87 'type': fname 

88 }) 

89 

90 return resp

92 def _wait_for_download(self, path, resource, resource_id, file_id, **kw): 

93 ''' 

94 A simple method to centralize waiting for an export to enter a 

95 completed state. The initial request will be made and then we will 

96 recheck every two and a half seconds after that. Once the status 

97 returns one of the completed states, we will return the status. 

98 ''' 

99 status = self._api.get(path, **kw).json()['status'] 

100 while status not in ['error', 'ready']: 

101 time.sleep(2.5) 

102 status = self._api.get(path, **kw).json()['status'] 

103 

104 # If the status that has been reported back is "error", then we will 

105 # need to throw the appropriate error back to the user. 

106 if status == 'error': 

107 raise FileDownloadError(resource, resource_id, file_id) 

108 

109 # Return the status to the caller. 

110 return status

112class TIOIterator(APIResultsIterator): 

113 _path = None 

114 _resource = None 

115 _size = 100 

116 _page_num = 0 

117 _api_version = 1

119 def _get_data(self): 

120 ''' 

121 Request the next page of data 

122 ''' 

123 # The first thing that we need to do is construct the query with the 

124 # current offset and limits 

125 query = self._query 

126 

127 if self._api_version == 2: 

128 query['size'] = self._size 

129 query['page'] = self._page_num 

130 else: 

131 query['limit'] = self._limit 

132 query['offset'] = self._offset 

133 

134 # Lets make the actual call at this point. 

135 resp = self._api.get(self._path, params=query).json() 

136 

137 if self._api_version == 2: 

138 self._page_num += 1 

139 else: 

140 self._offset += self._limit 

141 

142 # Lastly we need to return the data from the response and the data key 

143 # so that _get_page() knows where the information is stored. 

144 return resp, self._resource

146 def _get_page(self): 

147 ''' 

148 Get the next page of records 

149 ''' 

150 # First we need to see if there is a page limit and if there is, have 

151 # we run into that limit. If we have, then return a StopIteration 

152 # exception. 

153 if self._pages_total and self._pages_requested >= self._pages_total: 

154 raise StopIteration() 

155 

156 # Lets make the actual call at this point. 

157 resp, key = self._get_data() 

158 

159 # Now that we have the response, lets reset any counters we need to, 

160 # and increment things like the page counter, offset, etc. 

161 self.page_count = 0 

162 self._pages_requested += 1 

163 

164 # Lastly we want to refresh the page data and the total based on the 

165 # most recent data we have. 

166 if self._api_version == 2: 

167 self.page = resp['data'][key] 

168 self.total = resp['total_count'] 

169 else: 

170 self.page = resp[key] 

171 self.total = resp['pagination']['total']
