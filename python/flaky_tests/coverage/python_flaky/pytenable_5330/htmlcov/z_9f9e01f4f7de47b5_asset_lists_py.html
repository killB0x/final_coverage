25 def _dynamic_rules_constructor(self, rule, sub=False): 

26 ''' 

27 Handles expanding the tuple format into the JSON formatted request. 

28 ''' 

29 if isinstance(rule, dict): 

30 # if the rule is a dictionary, then simply pass it through as-is. 

31 return rule 

32 elif isinstance(rule, tuple): 

33 # if the rule is a tuple, then we will want to convert it into the 

34 # expected dictionary format. 

35 if rule[0] in ['all', 'any']: 

36 # if the first parameter in the tuple is either "any" or "all", 

37 # we will then assume that this is a group of rules, and call 

38 # the rule constructor for every subsequent parameter in the 

39 # tuple. 

40 resp = { 

41 'operator': rule[0], 

42 'children': [self._dynamic_rules_constructor(r, sub=True) 

43 for r in rule[1:]] 

44 } 

45 if sub: 

46 resp['type'] = 'group' 

47 else: 

48 # as the first item was _not_ "all" or "any", we're safe to 

49 # assume that the rule is actually a rule clause. In this case 

50 # we will want to validate the fields based on the potential 

51 # known values that each attribute could have. The rule should 

52 # generally be constructed in the following format: 

53 # 

54 # ('filterName', 'operator', 'value') 

55 # 

56 # or in the case of a plugin constraint, then there will be a 

57 # fourth parameter like so: 

58 # 

59 # ('filterName', 'operator', 'value', int(pluginID)) 

60 # or 

61 # ('filterName', 'operator', 'value', list(id1, id2, id3, etc.)) 

62 resp = { 

63 'type': 'clause', 

64 'filterName': self._check('rule:name', rule[0], str, 

65 choices=['dns', 'exploitAvailable', 'exploitFrameworks', 

66 'firstseen', 'mac', 'os', 'ip', 'lastseen', 

67 'netbioshost', 'netbiosworkgroup', 'pluginid', 

68 'plugintext', 'port', 'severity', 'sshv1', 'sshv2', 

69 'tcpport', 'udpport', 'xref']), 

70 'operator': self._check('rule:operator', rule[1], str, 

71 choices=['contains', 'eq', 'lt', 'lte', 'ne', 'gt', 

72 'gte', 'regex', 'pcre']) 

73 } 

74 

75 # if the value is an integer, then we will want to ensure that 

76 # we wrap the value within an id dictionary. This is necessary 

77 # for pluginid and severity filter names. In all other cases 

78 # the value should be a string. 

79 if rule[0] in ['pluginid', 'severity']: 

80 resp['value'] = { 

81 'id': self._check('rule:value', rule[2], int)} 

82 else: 

83 resp['value'] = self._check('rule:value', rule[2], str) 

84 

85 # if there is a plugin constraint, then we will want to convert 

86 # the plugin constraint into a string value. If it's a single 

87 # plugin id, then we will simply convert from int to str. If 

88 # a list of values is provided, then we will build a comma-delim 

89 # string with the values that were passed. 

90 if len(rule) == 4: 

91 if isinstance(rule[3], int): 

92 resp['pluginIDConstraint'] = str(rule[3]) 

93 elif isinstance(rule[3], list): 

94 resp['pluginIDConstraint'] = ','.join( 

95 [str(r) for r in rule[3]]) 

96 else: 

97 raise TypeError( 

98 'rule {} has an invalid plugin constraint.'.format(rule)) 

99 else: 

100 raise TypeError('rules {} not a tuple or dict'.format(rule)) 

101 return resp

298 def create(self, name, list_type, **kw): 

299 ''' 

300 Creates an asset-list. 

301 

302 :sc-api:`asset-list: create <Asset.html#asset_POST>` 

303 

304 Args: 

305 name (str): 

306 The name for the asset list to create. 

307 list_type (str): 

308 The type of list to create. Supported values are 

309 ``combination``, ``dnsname``, ``dnsnameupload``, ``dynamic``, 

310 ``ldapquery``, ``static``, ``staticeventfilter``, 

311 ``staticvulnfilter``, ``templates``, ``upload``, ``watchlist``, 

312 ``watchlisteventfilter``, and ``watchlistupload``. 

313 combinations (tuple, optional): 

314 An asset combination tuple. For further information refer to 

315 the asset combination logic described at 

316 :mod:`tenable.sc.analysis`. 

317 data_fields (list, optional): 

318 A list of data fields as required for a given asset list type. 

319 Each item within the list should be formatted in the following 

320 way: ``{'fieldName': 'name', 'fieldValue': 'value'}`` 

321 description (str, optional): 

322 The description for the asset list being created. 

323 dn (str, optional): 

324 The base DN to use for an LDAP query. Must also provide a 

325 ``search_string`` and an ``ldap_id``. 

326 dns_names (list, optional): 

327 When defining a DNS asset list, use this attribute to provide 

328 the list of DNS addresses. 

329 exclude_managed_ips (bool, optional): 

330 Determines whether or not managed IPs should be excluded from 

331 the asset list. 

332 filters (list, optional): 

333 A list of filter tuples to use when defining filtered asset 

334 list types. Follows the same format as filters within the rest 

335 of pyTenable. 

336 fobj (FileObject, optional): 

337 A file-like object to use when uploading an asset list. 

338 ips (list, optional): 

339 A list of IP Addresses, CIDRs, and/or IP Address ranges to use 

340 for the purposes of a static asset list. 

341 lce_id (int, optional): 

342 When defining a event-based asset list, which LCE should be used 

343 to generate the asset list query. 

344 ldap_id (int, optional): 

345 The numeric identifier pertaining to the LDAP server to use for 

346 an LDAP query. must also provide a ``dn`` and a 

347 ``search_string``. 

348 prep (bool, optional): 

349 Should asset preperation be run after the list is created? If 

350 unspecified, the default action is ``True``. 

351 rules (tuple, optional): 

352 For a dynamic asset list, the tuple definition of the rules to 

353 determine what Ips are associated to this asset list. Rules 

354 follow a similar pattern to the asset combination logic and 

355 are written in a way to follow the same visual methodology as 

356 the UI. 

357 

358 For example, a simple dynamic ruleset may look like: 

359 

360 .. code-block:: python 

361 

362 ('any', ('dns', 'contains', 'svc.company.tld'), 

363 ('dns', 'contains', 'prod.company.tld')) 

364 

365 Which would match all assets with either svc.company.tld or 

366 prod.company.tld in their DNS names. Rule gropups can be nested 

367 as well, by supplying a new group tuple instead of a rule: 

368 

369 .. code-block:: python 

370 

371 ('any', ('dns', 'contains', 'svc.company.tld'), 

372 ('dns', 'contains', 'prod.company.tld'), 

373 ('any', ('ip', 'contains', '192.168.140'), 

374 ('ip', 'contains', '192.168.141'))) 

375 

376 In this example we have nested another group requiring that the 

377 ip may contain either of the values in addition to any of the 

378 DNS rules. 

379 

380 It's also possible to constrain the rule to a specific plugin or 

381 plugins as well by adding a 4th element in a rule tuple. 

382 Defining them would look like so: 

383 

384 .. code-block:: python 

385 

386 # Singular Plugin ID 

387 ('plugintext', 'contains', 'credentialed', 19506) 

388 # Multiple Plugin IDs 

389 ('plugintext', 'contains', 'stuff', [19506, 10180]) 

390 

391 * Available rules are ``dns``, ``exploitAvailable``, 

392 ``exploitFrameworks``, ``firstseen``, ``mac``, ``os``, ``ip``, 

393 ``lastseen``, ``netbioshost``, ``netbiosworkgroup``, 

394 ``pluginid``, ``plugintext``, ``port``, ``severity``, ``sshv1``, 

395 ``sshv2``, ``tcpport``, ``udpport``, and ``xref``. 

396 * Available operators are ``contains``, ``eq``, ``lt``, ``lte``, 

397 ``ne``, ``gt``, ``gte``, ``regex``, ``pcre``. 

398 * Group alauses are either ``any`` or ``all``. Any is a logical 

399 or. All is a logical and. 

400 scan_id (int, optional): 

401 When defining an "individual" source_type, the numeric id of the 

402 scan instance to base the query upon. 

403 search_string (str, optional): 

404 The search string to use as part of an LDAP Query. Must also 

405 provide a ``dn`` and an ``ldap_id``. 

406 sort_dir (str, optional): 

407 When defining a filtered asset list type, determines the 

408 direction of the sort to use. This field must be passed when 

409 defining a sort_field. 

410 sort_field (str, optional): 

411 When defining a filtered asset list type, determines what field 

412 to sort the resulting query on. 

413 source_type (str, optional): 

414 The source of the data to query from when defining a filtered 

415 asset list type. 

416 start_offset (int, optional): 

417 The start offset of the filter to use when defining a filtered 

418 asset list type. 

419 tags (str, optional): 

420 A tag to associate to the asset list. 

421 template (int, optional): 

422 The numeric id of the template to use. 

423 tool (str, optional): 

424 When specifying filtered asset list types, the analysis tool to 

425 use for determining what IPs should be included within the 

426 asset list. 

427 view (str, optional): 

428 When the source_type is "individual", the view defined what 

429 subset of the data to use. 

430 

431 Returns: 

432 dict: The newly created asset-list. 

433 

434 Examples: 

435 >>> asset-list = sc.asset_lists.create() 

436 ''' 

437 kw['name'] = name 

438 kw['type'] = list_type 

439 

440 payload = self._constructor(**kw) 

441 return self._api.post('asset', json=payload).json()['response']

635 def export_definition(self, id, fobj=None): 

636 ''' 

637 Exports an asset list definition and stored the data in the file-like 

638 object that was passed. 

639 

640 :sc-api:`asset-list: export <Asset.html#AssetRESTReference-/asset/{id}/export>` 

641 

642 Args: 

643 id (int): The numeric identifier for the asset list to export. 

644 fobj (FileObject): 

645 The file-like object to store the asset list XML definition. 

646 

647 Returns: 

648 :obj:`FileObject`: 

649 The file-like object containing the XML definition. 

650 

651 Examples: 

652 >>> with open('example.xml', 'wb') as fobj: 

653 ... sc.asset_lists.export_definition(1, fobj) 

654 ''' 

655 resp = self._api.get('asset/{}/export'.format( 

656 self._check('id', id, int)), stream=True) 

657 

658 # if no file-like object was passed, then we will instantiate a BytesIO 

659 # object to push the file into. 

660 if not fobj: 

661 fobj = BytesIO() 

662 

663 # Lets stream the file into the file-like object... 

664 for chunk in resp.iter_content(chunk_size=1024): 

665 if chunk: 

666 fobj.write(chunk) 

667 fobj.seek(0) 

668 resp.close() 

669 return fobj
