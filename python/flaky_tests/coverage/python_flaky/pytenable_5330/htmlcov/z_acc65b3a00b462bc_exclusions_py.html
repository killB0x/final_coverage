20from .base import TIOEndpoint 

21from datetime import datetime 

22 

23class ExclusionsAPI(TIOEndpoint): 

24 def create(self, name, members, start_time=None, end_time=None, 

25 timezone=None, description=None, frequency=None, 

26 interval=None, weekdays=None, day_of_month=None, 

27 enabled=True): 

28 ''' 

29 Create a scan target exclusion. 

30 

31 :devportal:`exclusions: create <exclusions-create>` 

32 

33 Args: 

34 name (str): The name of the exclusion to create. 

35 members (list): 

36 The exclusions members. Each member should be a string with 

37 either a FQDN, IP Address, IP Range, or CIDR. 

38 description (str, optional): 

39 Some further detail about the exclusion. 

40 start_time (datetime): When the exclusion should start. 

41 end_time (datetime): When the exclusion should end. 

42 timezone (str, optional): 

43 The timezone to use for the exclusion. The default if none is 

44 specified is to use UTC. For the list of usable timezones, 

45 please refer to :devportal:`scans-timezones` 

46 frequency (str, optional): 

47 The frequency of the rule. The string inputted will be up-cased. 

48 Valid values are: ``ONETIME``, ``DAILY``, ``WEEKLY``, 

49 ``MONTHLY``, ``YEARLY``. 

50 Default value is ``ONETIME``. 

51 interval (int, optional): 

52 The interval of the rule. The default interval is 1 

53 weekdays (list, optional): 

54 List of 2-character representations of the days of the week to 

55 repeat the frequency rule on. Valid values are: 

56 *SU, MO, TU, WE, TH, FR, SA* 

57 Default values: ``['SU', 'MO', 'TU', 'WE', 'TH', 'FR', 'SA']`` 

58 day_of_month (int, optional): 

59 The day of the month to repeat a **MONTHLY** frequency rule on. 

60 The default is today. 

61 enabled (bool, optional): 

62 Is the exclusion enabled? The default is ``True`` 

63 

64 Returns: 

65 :obj:`dict`: 

66 Dictionary of the newly minted exclusion. 

67 

68 Examples: 

69 Creating a one-time exclusion: 

70 

71 >>> from datetime import datetime, timedelta 

72 >>> exclusion = tio.exclusions.create( 

73 ... 'Example One-Time Exclusion', 

74 ... ['127.0.0.1'], 

75 ... start_time=datetime.utcnow(), 

76 ... end_time=datetime.utcnow() + timedelta(hours=1)) 

77 

78 Creating a daily exclusion: 

79 

80 >>> exclusion = tio.exclusions.create( 

81 ... 'Example Daily Exclusion', 

82 ... ['127.0.0.1'], 

83 ... frequency='daily', 

84 ... start_time=datetime.utcnow(), 

85 ... end_time=datetime.utcnow() + timedelta(hours=1)) 

86 

87 Creating a weekly exclusion: 

88 

89 >>> exclusion = tio.exclusions.create( 

90 ... 'Example Weekly Exclusion', 

91 ... ['127.0.0.1'], 

92 ... frequency='weekly', 

93 ... weekdays=['mo', 'we', 'fr'], 

94 ... start_time=datetime.utcnow(), 

95 ... end_time=datetime.utcnow() + timedelta(hours=1)) 

96 

97 Creating a monthly esxclusion: 

98 

99 >>> exclusion = tio.exclusions.create( 

100 ... 'Example Monthly Exclusion', 

101 ... ['127.0.0.1'], 

102 ... frequency='monthly', 

103 ... day_of_month=1, 

104 ... start_time=datetime.utcnow(), 

105 ... end_time=datetime.utcnow() + timedelta(hours=1)) 

106 

107 Creating a yearly exclusion: 

108 

109 >>> exclusion = tio.exclusions.create( 

110 ... 'Example Yearly Exclusion', 

111 ... ['127.0.0.1'], 

112 ... frequency='yearly', 

113 ... start_time=datetime.utcnow(), 

114 ... end_time=datetime.utcnow() + timedelta(hours=1)) 

115 ''' 

116 # Starting with the innermost part of the payload, lets construct the 

117 # rrules dictionary. 

118 frequency = self._check('frequency', frequency, str, 

119 choices=['ONETIME', 'DAILY', 'WEEKLY', 'MONTHLY', 'YEARLY'], 

120 default='ONETIME', 

121 case='upper') 

122 

123 rrules = { 

124 'freq': frequency, 

125 'interval': self._check('interval', interval, int, default=1) 

126 } 

127 

128 # if the frequency is a weekly one, then we will need to specify the 

129 # days of the week that the exclusion is run on. 

130 if frequency == 'WEEKLY': 

131 rrules['byweekday'] = ','.join(self._check( 

132 'weekdays', weekdays, list, 

133 choices=['SU', 'MO', 'TU', 'WE', 'TH', 'FR', 'SA'], 

134 default=['SU', 'MO', 'TU', 'WE', 'TH', 'FR', 'SA'], 

135 case='upper')) 

136 # In the same vein as the frequency check, we're accepting 

137 # case-insensitive input, comparing it to our known list of 

138 # acceptable responses, then joining them all together into a 

139 # comma-separated string. 

140 

141 # if the frequency is monthly, then we will need to specify the day of 

142 # the month that the rule will run on. 

143 if frequency == 'MONTHLY': 

144 rrules['bymonthday'] = self._check('day_of_month', day_of_month, int, 

145 choices=list(range(1,32)), 

146 default=datetime.today().day) 

147 

148 # Next we need to construct the rest of the payload 

149 payload = { 

150 'name': self._check('name', name, str), 

151 'members': ','.join(self._check('members', members, list)), 

152 'description': self._check('description', description, str, default=''), 

153 'schedule': { 

154 'enabled': self._check('enabled', enabled, bool, default=True), 

155 'starttime': self._check('start_time', start_time, datetime).strftime('%Y-%m-%d %H:%M:%S'), 

156 'endtime': self._check('end_time', end_time, datetime).strftime('%Y-%m-%d %H:%M:%S'), 

157 'timezone': self._check('timezone', timezone, str, 

158 choices=self._api._tz, 

159 default='Etc/UTC'), 

160 'rrules': rrules 

161 } 

162 } 

163 

164 # And now to make the call and return the data. 

165 return self._api.post('exclusions', json=payload).json()

205 def edit(self, id, name=None, members=None, start_time=None, 

206 end_time=None, timezone=None, description=None, frequency=None, 

207 interval=None, weekdays=None, day_of_month=None, enabled=None): 

208 ''' 

209 Edit an existing scan target exclusion. 

210 

211 :devportal:`exclusions: edit <exclusions-edit>` 

212 

213 The edit function will first gather the details of the exclusion that 

214 will be edited and will overlay the changes on top. The result will 

215 then be pushed back to the API to modify the exclusion. 

216 

217 Args: 

218 id (int): The id of the exclusion object in Tenable.io 

219 

220 scanner_id (int, optional): The scanner id. 

221 name (str, optional): The name of the exclusion to create. 

222 description (str, optional): 

223 Some further detail about the exclusion. 

224 start_time (datetime, optional): When the exclusion should start. 

225 end_time (datetime, optional): When the exclusion should end. 

226 timezone (str, optional): 

227 The timezone to use for the exclusion. The default if none is 

228 specified is to use UTC. 

229 frequency (str, optional): 

230 The frequency of the rule. The string inputted will be upcased. 

231 Valid values are: *ONETIME, DAILY, WEEKLY, MONTHLY, YEARLY*. 

232 interval (int, optional): The interval of the rule. 

233 weekdays (list, optional): 

234 List of 2-character representations of the days of the week to 

235 repeat the frequency rule on. Valid values are: 

236 *SU, MO, TU, WE, TH, FR, SA* 

237 Default values: ``['SU', 'MO', 'TU', 'WE', 'TH', 'FR', 'SA']`` 

238 day_of_month (int, optional): 

239 The day of the month to repeat a **MONTHLY** frequency rule on. 

240 

241 Returns: 

242 :obj:`dict`: 

243 Dictionary of the newly minted exclusion. 

244 

245 Examples: 

246 Modifying the name of an exclusion: 

247 

248 >>> exclusion = tio.exclusions.edit(1, name='New Name') 

249 ''' 

250 

251 # Lets start constructing the payload to be sent to the API... 

252 payload = self.details(id) 

253 

254 if name: 

255 payload['name'] = self._check('name', name, str) 

256 

257 if members: 

258 payload['members'] = ','.join(self._check('members', members, list)) 

259 

260 if description: 

261 payload['description'] = self._check('description', description, str) 

262 

263 if enabled is not None: 

264 payload['schedule']['enabled'] = self._check('enabled', enabled, bool) 

265 

266 if start_time: 

267 payload['schedule']['starttime'] = self._check( 

268 'start_time', start_time, datetime).strftime('%Y-%m-%d %H:%M:%S') 

269 

270 if end_time: 

271 payload['schedule']['endtime'] = self._check( 

272 'end_time', end_time, datetime).strftime('%Y-%m-%d %H:%M:%S') 

273 

274 if timezone: 

275 payload['schedule']['timezone'] = self._check( 

276 'timezone', timezone, str, choices=self._api._tz) 

277 

278 if frequency: 

279 payload['schedule']['rrules']['freq'] = self._check( 

280 'frequency', frequency, str, 

281 choices=['ONETIME', 'DAILY', 'WEEKLY', 'MONTHLY', 'YEARLY'], 

282 case='upper') 

283 

284 if interval: 

285 payload['schedule']['rrules']['interval'] = self._check( 

286 'interval', interval, int) 

287 

288 if weekdays: 

289 payload['schedule']['rrules']['byweekday'] = ','.join(self._check( 

290 'weekdays', weekdays, list, 

291 choices=['SU', 'MO', 'TU', 'WE', 'TH', 'FR', 'SA'], 

292 case='upper')) 

293 # In the same vein as the frequency check, we're accepting 

294 # case-insensitive input, comparing it to our known list of 

295 # acceptable responses, then joining them all together into a 

296 # comma-separated string. 

297 

298 if day_of_month is not None: 

299 payload['schedule']['rrules']['bymonthday'] = self._check( 

300 'day_of_month', day_of_month, int, choices=list(range(1,32))) 

301 

302 # Lets check to make sure that the scanner_id and exclusion_id are 

303 # integers as the API documentation requests and if we don't raise an 

304 # error, then lets make the call. 

305 return self._api.put( 

306 'exclusions/{}'.format( 

307 self._check('id', id, int) 

308 ), json=payload).json()
