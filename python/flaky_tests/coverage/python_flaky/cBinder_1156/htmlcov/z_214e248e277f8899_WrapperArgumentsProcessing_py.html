34 """

37 to check if a user passed these OUT arrays correctly.

89def _initialize_out_arrays_if_necessary_and_check_sizes(writer, parameters): 

90 """ 

91 (1) 

92 There is a possibility, that a function has multiple array parameters, both IN and OUT. 

93 When there is at least one IN array of a certain declared size, every OUT array of the same 

94 declared size can be initialized based on the length of the IN array (it is assumed that IN array 

95 is passed with correct size). 

96 This function, adds a correct initializing script to a wrapping function - when it is possible. 

97 For example, part of a doxygen comment: 

98 * @param[in] in_arr sample array (array of size n) 

99 * @param[out] out_arr sample array (array of size n) 

100 (2) 

101 This function also adds a check, when there is no IN parameter of a declared size, yet there are 

102 at least two OUT array parameters of this size - then, both of them should be passed with the same size. 

103 """ 

104 # First, get every array parameter 

105 array_params = [param for param in parameters if param.is_array] 

106 # Get the list of parameters with the same size (list of ArraysSameSize objects) 

107 arrays_same_sizes_list = _get_arrays_of_same_size_list(array_params) 

108 

109 for arr_same_size_obj in arrays_same_sizes_list: 

110 # Check, if there is at least one IN array parameter of a certain size 

111 arr_same_size_obj.check_for_decisive_in_param() 

112 

113 # Get a list of OUT array parameters (not IN, not IN/OUT) 

114 array_out_params = [param for param in parameters if param.is_array and param.is_out and not param.is_in] 

115 # Local list variable to perform a size consistency check later, if there is no IN array parameter 

116 # to infer proper size from 

117 arrays_out_to_check_size_consistency = [] 

118 for arr_out in array_out_params: 

119 same_arr_size_obj = _get_object_for_given_param(arrays_same_sizes_list, arr_out) 

120 decisive_parameter_name = same_arr_size_obj.decisive_param_name 

121 if decisive_parameter_name is not None: 

122 # If the initialization is possible, add a correct script 

123 _initialize_one_out_array(writer, arr_out, decisive_parameter_name) 

124 else: 

125 arrays_out_to_check_size_consistency.append(arr_out) 

126 if len(arrays_out_to_check_size_consistency) > 1: 

127 _check_if_every_array_of_the_same_size_and_type_has_indeed_same_size(writer, arrays_out_to_check_size_consistency, False)

130def _initialize_non_array_out_parameters_if_necessary(writer, parameters): 

131 """ 

132 When user wants to return a value from a C language function not through return statement 

133 but using OUT non-array parameter - that is, a pointer, the only option to actually 

134 return a value from a wrapping function is to treat this parameter as an array of size 1. 

135 (1) 

136 Firstly, because such parameters have to be processed as arrays of size 1, user has to 

137 pass it as array type in Python. Appropriate check is performed and error thrown in case 

138 incorrect argument is passed 

139 (2) 

140 If user passed an array type, but it does not have size 1, warning is raised and 

141 auto-initialization is done 

142 """ 

143 # First, get every OUT parameter of size 1 

144 non_array_out_params = [param for param in parameters if param.is_out and param.sizes[0] == 1] 

145 # Add checking if every parameter is passed as an array 

146 for param in non_array_out_params: 

147 with writer.write_if('not isinstance(' + param.name + ', list)'): 

148 writer.write_line('out_param_err = \"You passed OUT parameter not as an array.\"') 

149 writer.write_line('raise ValueError(out_param_err)') 

150 # Then add checking procedure with initialization 

151 for param in non_array_out_params: 

152 with writer.write_if('len(' + param.name + ') != 1'): 

153 writer.write_line('out_param_auto_init = \"\\nWarning: OUT parameter (not an array)' + ' was passed with incorrect size.\\n\" + \\') 

154 writer.write_line('\t\"Wrapper initializes it with size 1\"') 

155 writer.write_line('warnings.warn(out_param_auto_init)') 

156 writer.write_line(param.name + '.clear()') 

157 if 'char' in param.type: 

158 # Char type has to be initialized with string 

159 writer.write_line(param.name + ' += [\'\']') 

160 else: 

161 writer.write_line(param.name + ' += [0]')
