38def update_sim_ap_params(sim_params, delta, field=None): 

39 """Update the aperiodic parameter definition in a SimParams object. 

40 

41 Parameters 

42 ---------- 

43 sim_params : SimParams 

44 Object storing the current parameter definition. 

45 delta : float or list of float 

46 Value(s) by which to update the parameters. 

47 field : {'offset', 'knee', 'exponent'} or list of string 

48 Field of the aperiodic parameter(s) to update. 

49 

50 Returns 

51 ------- 

52 new_sim_params : SimParams 

53 Updated object storing the new parameter definition. 

54 

55 Raises 

56 ------ 

57 InconsistentDataError 

58 If the input parameters and update values are inconsistent. 

59 """ 

60 

61 # Grab the aperiodic parameters that need updating 

62 ap_params = sim_params.aperiodic_params.copy() 

63 

64 # If field isn't specified, check shapes line up and update across parameters 

65 if not field: 

66 if not len(ap_params) == len(delta): 

67 raise InconsistentDataError("The number of items to update and " 

68 "number of new values is inconsistent.") 

69 ap_params = [param + update for param, update in zip(ap_params, delta)] 

70 

71 # If labels are given, update deltas according to their labels 

72 else: 

73 # This loop checks & casts to list, to work for single or multiple passed in values 

74 for cur_field, cur_delta in zip(list([field]) if not isinstance(field, list) else field, 

75 list([delta]) if not isinstance(delta, list) else delta): 

76 data_ind = get_indices(infer_ap_func(ap_params))[cur_field] 

77 ap_params[data_ind] = ap_params[data_ind] + cur_delta 

78 

79 # Replace parameters. Note that this copies a new object, as data objects are immutable 

80 new_sim_params = sim_params._replace(aperiodic_params=ap_params) 

81 

82 return new_sim_params

85class Stepper(): 

86 """Object for stepping across parameter values. 

87 

88 Parameters 

89 ---------- 

90 start : float 

91 Start value to iterate from. 

92 stop : float 

93 End value to iterate to. 

94 step : float 

95 Increment of each iteration. 

96 

97 Attributes 

98 ---------- 

99 len : int 

100 Length of generator range. 

101 data : iterator 

102 Set of specified parameters to iterate across. 

103 

104 Examples 

105 -------- 

106 Define a stepper object for center frequency values for an alpha peak: 

107 

108 >>> alpha_cf_steps = Stepper(8, 12.5, 0.5) 

109 """ 

110 

111 def __init__(self, start, stop, step): 

112 """Initialize a Stepper object.""" 

113 

114 self._check_values(start, stop, step) 

115 

116 self.start = start 

117 self.stop = stop 

118 self.step = step 

119 self.len = round((stop-start)/step) 

120 self.data = iter(np.arange(start, stop, step)) 

121 

122 def __len__(self): 

123 

124 return self.len 

125 

126 def __next__(self): 

127 

128 return round(next(self.data), 4) 

129 

130 def __iter__(self): 

131 

132 return self.data

135 def _check_values(start, stop, step): 

136 """Checks if provided values are valid. 

137 

138 Parameters 

139 ---------- 

140 start, stop, step : float 

141 Definition of the parameter range to iterate over. 

142 

143 Raises 

144 ------ 

145 ValueError 

146 If the given values for defining the iteration range are inconsistent. 

147 """ 

148 

149 if any(ii < 0 for ii in [start, stop, step]): 

150 raise ValueError("Inputs 'start', 'stop', and 'step' should all be positive values.") 

151 

152 if not start < stop: 

153 raise ValueError("Input 'start' should be less than 'stop'.") 

154 

155 if not step < (stop - start): 

156 raise ValueError("Input 'step' is too large given values for 'start' and 'stop'.")

159def param_iter(params): 

160 """Create a generator to iterate across parameter ranges. 

161 

162 Parameters 

163 ---------- 

164 params : list of floats and Stepper 

165 Parameters over which to iterate, including a Stepper object. 

166 The Stepper defines the iterated parameter and its range. 

167 Floats define the other parameters, that will be held constant. 

168 

169 Yields 

170 ------ 

171 list of floats 

172 Next generated list of parameters. 

173 

174 Raises 

175 ------ 

176 ValueError 

177 If the number of Stepper objects given is greater than one. 

178 

179 Examples 

180 -------- 

181 Iterate across exponent values from 1 to 2, in steps of 0.1: 

182 

183 >>> aps = param_iter([Stepper(1, 2, 0.1), 1]) 

184 

185 Iterate over center frequency values from 8 to 12 in increments of 0.25: 

186 

187 >>> peaks = param_iter([Stepper(8, 12, .25), 0.5, 1]) 

188 """ 

189 

190 # If input is a list of lists, check each element, and flatten if needed 

191 if isinstance(params[0], list): 

192 params = [item for sublist in params for item in sublist] 

193 

194 # Finds where Stepper object is. If there is more than one, raise an error 

195 iter_ind = 0 

196 num_iters = 0 

197 for cur_ind, param in enumerate(params): 

198 

199 if isinstance(param, Stepper): 

200 num_iters += 1 

201 iter_ind = cur_ind 

202 

203 if num_iters > 1: 

204 raise ValueError("Iteration is only supported across one parameter at a time.") 

205 

206 # Generate and yield next set of parameters 

207 gen = params[iter_ind] 

208 while True: 

209 try: 

210 params[iter_ind] = next(gen) 

211 yield params 

212 except StopIteration: 

213 return

216def param_sampler(params, probs=None): 

217 """Create a generator to sample randomly from possible parameters. 

218 

219 Parameters 

220 ---------- 

221 params : list of lists or list of float 

222 Possible parameter values. 

223 probs : list of float, optional 

224 Probabilities with which to sample each parameter option. 

225 If None, each parameter option is sampled uniformly. 

226 

227 Yields 

228 ------ 

229 list of float 

230 A randomly sampled set of parameters. 

231 

232 Examples 

233 -------- 

234 Sample from aperiodic definitions with high and low exponents, with 50% probability of each: 

235 

236 >>> aps = param_sampler([[1, 1], [2, 1]], probs=[0.5, 0.5]) 

237 

238 Sample from peak definitions of alpha or alpha & beta, with 75% change of sampling just alpha: 

239 

240 >>> peaks = param_sampler([[10, 1, 1], [[10, 1, 1], [20, 0.5, 1]]], probs=[0.75, 0.25]) 

241 """ 

242 

243 # If input is a list of lists, check each element, and flatten if needed 

244 if isinstance(params[0], list): 

245 params = [check_flat(lst) for lst in params] 

246 

247 # In order to use numpy's choice, with probabilities, choices are made on indices 

248 # This is because the params can be a messy-sized list, that numpy choice does not like 

249 inds = np.array(range(len(params))) 

250 

251 # Check that length of options is same as length of probs, if provided 

252 if np.any(probs): 

253 if len(inds) != len(probs): 

254 raise ValueError("The number of options must match the number of probabilities.") 

255 

256 # While loop allows the generator to be called an arbitrary number of times 

257 while True: 

258 yield params[np.random.choice(inds, p=probs)]

261def param_jitter(params, jitters): 

262 """Create a generator that adds jitter to parameter definitions. 

263 

264 Parameters 

265 ---------- 

266 params : list of lists or list of float 

267 Possible parameter values. 

268 jitters : list of lists or list of float 

269 The scale of the jitter for each parameter. 

270 Must be the same shape and organization as `params`. 

271 

272 Yields 

273 ------ 

274 list of float 

275 A jittered set of parameters. 

276 

277 Notes 

278 ----- 

279 - Jitter is added as random samples from a normal (gaussian) distribution. 

280 

281 - The jitter specified corresponds to the standard deviation of the normal distribution. 

282 - For any parameter for which there should be no jitter, set the corresponding value to zero. 

283 

284 Examples 

285 -------- 

286 Jitter aperiodic definitions, for offset and exponent, each with the same amount of jitter: 

287 

288 >>> aps = param_jitter([1, 1], [0.1, 0.1]) 

289 

290 Jitter center frequency of peak definitions, by different amounts for alpha & beta: 

291 

292 >>> peaks = param_jitter([[10, 1, 1], [20, 1, 1]], [[0.1, 0, 0], [0.5, 0, 0]]) 

293 """ 

294 

295 # Check if inputs are list of lists, and flatten if so 

296 if isinstance(params[0], list): 

297 params = check_flat(params) 

298 jitters = check_flat(jitters) 

299 

300 # While loop allows the generator to be called an arbitrary number of times 

301 while True: 

302 

303 out_params = [None] * len(params) 

304 for ind, (param, jitter) in enumerate(zip(params, jitters)): 

305 out_params[ind] = param + np.random.normal(0, jitter) 

306 

307 yield out_params
