28class FOOOFGroup(FOOOF): 

29 """Model a group of power spectra as a combination of aperiodic and periodic components. 

30 

31 WARNING: FOOOF expects frequency and power values in linear space. 

32 

33 Passing in logged frequencies and/or power spectra is not detected, 

34 and will silently produce incorrect results. 

35 

36 Parameters 

37 ---------- 

38 peak_width_limits : tuple of (float, float), optional, default: (0.5, 12.0) 

39 Limits on possible peak width, as (lower_bound, upper_bound). 

40 max_n_peaks : int, optional, default: inf 

41 Maximum number of gaussians to be fit in a single spectrum. 

42 min_peak_height : float, optional, default: 0 

43 Absolute threshold for detecting peaks, in units of the input data. 

44 peak_threshold : float, optional, default: 2.0 

45 Relative threshold for detecting peaks, in units of standard deviation of the input data. 

46 aperiodic_mode : {'fixed', 'knee'} 

47 Which approach to take for fitting the aperiodic component. 

48 verbose : bool, optional, default: True 

49 Verbosity mode. If True, prints out warnings and general status updates. 

50 

51 Attributes 

52 ---------- 

53 freqs : 1d array 

54 Frequency values for the power spectra. 

55 power_spectra : 2d array 

56 Power values for the group of power spectra, as [n_power_spectra, n_freqs]. 

57 Power values are stored internally in log10 scale. 

58 freq_range : list of [float, float] 

59 Frequency range of the power spectra, as [lowest_freq, highest_freq]. 

60 freq_res : float 

61 Frequency resolution of the power spectra. 

62 group_results : list of FOOOFResults 

63 Results of FOOOF model fit for each power spectrum. 

64 has_data : bool 

65 Whether data is loaded to the object. 

66 has_model : bool 

67 Whether model results are available in the object. 

68 n_peaks_ : int 

69 The number of peaks fit in the model. 

70 n_failed_fits_ : int 

71 The number of models that failed to fit. 

72 failed_fit_inds_ : list of int 

73 The indices of any models that failed to fit. 

74 

75 Notes 

76 ----- 

77 - Commonly used abbreviations used in this module include: 

78 CF: center frequency, PW: power, BW: Bandwidth, AP: aperiodic 

79 - Input power spectra must be provided in linear scale. 

80 Internally they are stored in log10 scale, as this is what the model operates upon. 

81 - Input power spectra should be smooth, as overly noisy power spectra may lead to bad fits. 

82 For example, raw FFT inputs are not appropriate. Where possible and appropriate, use 

83 longer time segments for power spectrum calculation to get smoother power spectra, 

84 as this will give better model fits. 

85 - The gaussian params are those that define the gaussian of the fit, where as the peak 

86 params are a modified version, in which the CF of the peak is the mean of the gaussian, 

87 the PW of the peak is the height of the gaussian over and above the aperiodic component, 

88 and the BW of the peak, is 2*std of the gaussian (as 'two sided' bandwidth). 

89 - The FOOOFGroup object inherits from the FOOOF object. As such it also has data 

90 attributes (`power_spectrum` & `fooofed_spectrum_`), and parameter attributes 

91 (`aperiodic_params_`, `peak_params_`, `gaussian_params_`, `r_squared_`, `error_`) 

92 which are defined in the context of individual model fits. These attributes are 

93 used during the fitting process, but in the group context do not store results 

94 post-fitting. Rather, all model fit results are collected and stored into the 

95 `group_results` attribute. To access individual parameters of the fit, use 

96 the `get_params` method. 

97 """

255 def fit(self, freqs=None, power_spectra=None, freq_range=None, n_jobs=1, progress=None): 

256 """Fit a group of power spectra. 

257 

258 Parameters 

259 ---------- 

260 freqs : 1d array, optional 

261 Frequency values for the power_spectra, in linear space. 

262 power_spectra : 2d array, shape: [n_power_spectra, n_freqs], optional 

263 Matrix of power spectrum values, in linear space. 

264 freq_range : list of [float, float], optional 

265 Desired frequency range to run FOOOF on. If not provided, fits the entire given range. 

266 n_jobs : int, optional, default: 1 

267 Number of jobs to run in parallel. 

268 1 is no parallelization. -1 uses all available cores. 

269 progress : {None, 'tqdm', 'tqdm.notebook'}, optional 

270 Which kind of progress bar to use. If None, no progress bar is used. 

271 

272 Notes 

273 ----- 

274 Data is optional, if data has already been added to the object. 

275 """ 

276 

277 # If freqs & power spectra provided together, add data to object 

278 if freqs is not None and power_spectra is not None: 

279 self.add_data(freqs, power_spectra, freq_range) 

280 

281 # If 'verbose', print out a marker of what is being run 

282 if self.verbose and not progress: 

283 print('Running FOOOFGroup across {} power spectra.'.format(len(self.power_spectra))) 

284 

285 # Run linearly 

286 if n_jobs == 1: 

287 self._reset_group_results(len(self.power_spectra)) 

288 for ind, power_spectrum in \ 

289 _progress(enumerate(self.power_spectra), progress, len(self)): 

290 self._fit(power_spectrum=power_spectrum) 

291 self.group_results[ind] = self._get_results() 

292 

293 # Run in parallel 

294 else: 

295 self._reset_group_results() 

296 n_jobs = cpu_count() if n_jobs == -1 else n_jobs 

297 with Pool(processes=n_jobs) as pool: 

298 self.group_results = list(_progress(pool.imap(partial(_par_fit, fg=self), 

299 self.power_spectra), 

300 progress, len(self.power_spectra))) 

301 

302 # Clear the individual power spectrum and fit results of the current fit 

303 self._reset_data_results(clear_spectrum=True, clear_results=True)

332 def get_params(self, name, col=None): 

333 """Return model fit parameters for specified feature(s). 

334 

335 Parameters 

336 ---------- 

337 name : {'aperiodic_params', 'peak_params', 'gaussian_params', 'error', 'r_squared'} 

338 Name of the data field to extract across the group. 

339 col : {'CF', 'PW', 'BW', 'offset', 'knee', 'exponent'} or int, optional 

340 Column name / index to extract from selected data, if requested. 

341 Only used for name of {'aperiodic_params', 'peak_params', 'gaussian_params'}. 

342 

343 Returns 

344 ------- 

345 out : ndarray 

346 Requested data. 

347 

348 Raises 

349 ------ 

350 NoModelError 

351 If there are no model fit results available. 

352 ValueError 

353 If the input for the `col` input is not understood. 

354 

355 Notes 

356 ----- 

357 For further description of the data you can extract, check the FOOOFResults documentation. 

358 """ 

359 

360 if not self.has_model: 

361 raise NoModelError("No model fit results are available, can not proceed.") 

362 

363 # Allow for shortcut alias, without adding `_params` 

364 if name in ['aperiodic', 'peak', 'gaussian']: 

365 name = name + '_params' 

366 

367 # If col specified as string, get mapping back to integer 

368 if isinstance(col, str): 

369 col = get_indices(self.aperiodic_mode)[col] 

370 elif isinstance(col, int): 

371 if col not in [0, 1, 2]: 

372 raise ValueError("Input value for `col` not valid.") 

373 

374 # Pull out the requested data field from the group data 

375 # As a special case, peak_params are pulled out in a way that appends 

376 # an extra column, indicating which FOOOF run each peak comes from 

377 if name in ('peak_params', 'gaussian_params'): 

378 out = np.array([np.insert(getattr(data, name), 3, index, axis=1) 

379 for index, data in enumerate(self.group_results)]) 

380 # This updates index to grab selected column, and the last column 

381 # This last column is the 'index' column (FOOOF object source) 

382 if col is not None: 

383 col = [col, -1] 

384 else: 

385 out = np.array([getattr(data, name) for data in self.group_results]) 

386 

387 # Some data can end up as a list of separate arrays 

388 # If so, concatenate it all into one 2d array 

389 if isinstance(out[0], np.ndarray): 

390 out = np.concatenate([arr.reshape(1, len(arr)) \ 

391 if arr.ndim == 1 else arr for arr in out], 0) 

392 

393 # Select out a specific column, if requested 

394 if col is not None: 

395 out = out[:, col] 

396 

397 return out

419 def load(self, file_name, file_path=None): 

420 """Load FOOOFGroup data from file. 

421 

422 Parameters 

423 ---------- 

424 file_name : str 

425 File to load data from. 

426 file_path : str, optional 

427 Path to directory to load from. If None, loads from current directory. 

428 """ 

429 

430 # Clear results so as not to have possible prior results interfere 

431 self._reset_group_results() 

432 

433 power_spectra = [] 

434 for ind, data in enumerate(load_jsonlines(file_name, file_path)): 

435 

436 self._add_from_dict(data) 

437 

438 # If settings are loaded, check and update based on the first line 

439 if ind == 0: 

440 self._check_loaded_settings(data) 

441 

442 # If power spectra data is part of loaded data, collect to add to object 

443 if 'power_spectrum' in data.keys(): 

444 power_spectra.append(data['power_spectrum']) 

445 

446 # If results part of current data added, check and update object results 

447 if set(OBJ_DESC['results']).issubset(set(data.keys())): 

448 self._check_loaded_results(data) 

449 self.group_results.append(self._get_results()) 

450 

451 # Reconstruct frequency vector, if information is available to do so 

452 if self.freq_range: 

453 self._regenerate_freqs() 

454 

455 # Add power spectra data, if they were loaded 

456 if power_spectra: 

457 self.power_spectra = np.array(power_spectra) 

458 

459 # Reset peripheral data from last loaded result, keeping freqs info 

460 self._reset_data_results(clear_spectrum=True, clear_results=True)

463 def get_fooof(self, ind, regenerate=True): 

464 """Get a FOOOF object for a specified model fit. 

465 

466 Parameters 

467 ---------- 

468 ind : int 

469 The index of the FOOOFResults in FOOOFGroup.group_results to load. 

470 regenerate : bool, optional, default: False 

471 Whether to regenerate the model fits from the given fit parameters. 

472 

473 Returns 

474 ------- 

475 fm : FOOOF 

476 The FOOOFResults data loaded into a FOOOF object. 

477 """ 

478 

479 # Initialize a FOOOF object, with same settings & check data mode as current FOOOFGroup 

480 fm = FOOOF(*self.get_settings(), verbose=self.verbose) 

481 fm.set_check_data_mode(self._check_data) 

482 

483 # Add data for specified single power spectrum, if available 

484 # The power spectrum is inverted back to linear, as it is re-logged when added to FOOOF 

485 if self.has_data: 

486 fm.add_data(self.freqs, np.power(10, self.power_spectra[ind])) 

487 # If no power spectrum data available, copy over data information & regenerate freqs 

488 else: 

489 fm.add_meta_data(self.get_meta_data()) 

490 

491 # Add results for specified power spectrum, regenerating full fit if requested 

492 fm.add_results(self.group_results[ind]) 

493 if regenerate: 

494 fm._regenerate_model() 

495 

496 return fm

499 def get_group(self, inds): 

500 """Get a FOOOFGroup object with the specified sub-selection of model fits. 

501 

502 Parameters 

503 ---------- 

504 inds : array_like of int or array_like of bool 

505 Indices to extract from the object. 

506 If a boolean mask, True indicates indices to select. 

507 

508 Returns 

509 ------- 

510 fg : FOOOFGroup 

511 The requested selection of results data loaded into a new FOOOFGroup object. 

512 """ 

513 

514 # Check and convert indices encoding to list of int 

515 inds = check_inds(inds) 

516 

517 # Initialize a new FOOOFGroup object, with same settings as current FOOOFGroup 

518 fg = FOOOFGroup(*self.get_settings(), verbose=self.verbose) 

519 

520 # Add data for specified power spectra, if available 

521 # The power spectra are inverted back to linear, as they are re-logged when added to FOOOF 

522 if self.has_data: 

523 fg.add_data(self.freqs, np.power(10, self.power_spectra[inds, :])) 

524 # If no power spectrum data available, copy over data information & regenerate freqs 

525 else: 

526 fg.add_meta_data(self.get_meta_data()) 

527 

528 # Add results for specified power spectra 

529 fg.group_results = [self.group_results[ind] for ind in inds] 

530 

531 return fg

576def _progress(iterable, progress, n_to_run): 

577 """Add a progress bar to an iterable to be processed. 

578 

579 Parameters 

580 ---------- 

581 iterable : list or iterable 

582 Iterable object to potentially apply progress tracking to. 

583 progress : {None, 'tqdm', 'tqdm.notebook'} 

584 Which kind of progress bar to use. If None, no progress bar is used. 

585 n_to_run : int 

586 Number of jobs to complete. 

587 

588 Returns 

589 ------- 

590 pbar : iterable or tqdm object 

591 Iterable object, with tqdm progress functionality, if requested. 

592 

593 Raises 

594 ------ 

595 ValueError 

596 If the input for `progress` is not understood. 

597 

598 Notes 

599 ----- 

600 The explicit `n_to_run` input is required as tqdm requires this in the parallel case. 

601 The `tqdm` object that is potentially returned acts the same as the underlying iterable, 

602 with the addition of printing out progress every time items are requested. 

603 """ 

604 

605 # Check progress specifier is okay 

606 tqdm_options = ['tqdm', 'tqdm.notebook'] 

607 if progress is not None and progress not in tqdm_options: 

608 raise ValueError("Progress bar option not understood.") 

609 

610 # Set the display text for the progress bar 

611 pbar_desc = 'Running FOOOFGroup' 

612 

613 # Use a tqdm, progress bar, if requested 

614 if progress: 

615 

616 # Try loading the tqdm module 

617 tqdm = safe_import(progress) 

618 

619 if not tqdm: 

620 

621 # If tqdm isn't available, proceed without a progress bar 

622 print(("A progress bar requiring the 'tqdm' module was requested, " 

623 "but 'tqdm' is not installed. \nProceeding without using a progress bar.")) 

624 pbar = iterable 

625 

626 else: 

627 

628 # If tqdm loaded, apply the progress bar to the iterable 

629 pbar = tqdm.tqdm(iterable, desc=pbar_desc, total=n_to_run, dynamic_ncols=True) 

630 

631 # If progress is None, return the original iterable without a progress bar applied 

632 else: 

633 pbar = iterable 

634 

635 return pbar
