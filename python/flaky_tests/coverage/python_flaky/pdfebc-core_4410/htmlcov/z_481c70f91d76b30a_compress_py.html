36def _get_pdf_filenames_at(source_directory): 

37 """Find all PDF files in the specified directory. 

38 

39 Args: 

40 source_directory (str): The source directory. 

41 

42 Returns: 

43 list(str): Filepaths to all PDF files in the specified directory. 

44 

45 Raises: 

46 ValueError 

47 """ 

48 if not os.path.isdir(source_directory): 

49 raise ValueError("%s is not a directory!" % source_directory) 

50 return [os.path.join(source_directory, filename) 

51 for filename in os.listdir(source_directory) 

52 if filename.endswith(PDF_EXTENSION)]

54def compress_pdf(filepath, output_path, ghostscript_binary): 

55 """Compress a single PDF file. 

56 

57 Args: 

58 filepath (str): Path to the PDF file. 

59 output_path (str): Output path. 

60 ghostscript_binary (str): Name/alias of the Ghostscript binary. 

61 

62 Raises: 

63 ValueError 

64 FileNotFoundError 

65 """ 

66 if not filepath.endswith(PDF_EXTENSION): 

67 raise ValueError("Filename must end with .pdf!\n%s does not." % filepath) 

68 try: 

69 file_size = os.stat(filepath).st_size 

70 if file_size < FILE_SIZE_LOWER_LIMIT: 

71 LOGGER.info(NOT_COMPRESSING.format(filepath, file_size, FILE_SIZE_LOWER_LIMIT)) 

72 process = subprocess.Popen(['cp', filepath, output_path]) 

73 else: 

74 LOGGER.info(COMPRESSING.format(filepath)) 

75 process = subprocess.Popen( 

76 [ghostscript_binary, "-sDEVICE=pdfwrite", 

77 "-dCompatabilityLevel=1.4", "-dPDFSETTINGS=/ebook", 

78 "-dNOPAUSE", "-dQUIET", "-dBATCH", 

79 "-sOutputFile=%s" % output_path, filepath] 

80 ) 

81 except FileNotFoundError: 

82 msg = GS_NOT_INSTALLED.format(ghostscript_binary) 

83 raise FileNotFoundError(msg) 

84 process.communicate() 

85 LOGGER.info(FILE_DONE.format(output_path))

87def compress_multiple_pdfs(source_directory, output_directory, ghostscript_binary): 

88 """Compress all PDF files in the current directory and place the output in the 

89 given output directory. This is a generator function that first yields the amount 

90 of files to be compressed, and then yields the output path of each file. 

91 

92 Args: 

93 source_directory (str): Filepath to the source directory. 

94 output_directory (str): Filepath to the output directory. 

95 ghostscript_binary (str): Name of the Ghostscript binary. 

96 

97 Returns: 

98 list(str): paths to outputs. 

99 """ 

100 source_paths = _get_pdf_filenames_at(source_directory) 

101 yield len(source_paths) 

102 for source_path in source_paths: 

103 output = os.path.join(output_directory, os.path.basename(source_path)) 

104 compress_pdf(source_path, output, ghostscript_binary) 

105 yield output
