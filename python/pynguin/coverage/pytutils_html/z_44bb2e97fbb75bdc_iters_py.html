7def accumulate(iterable, func=operator.add): 

8 """ 

9 Iterate over running totals, ie [a,b,c,d] -> func( func( func(a, b), c), d) with each func result yielded. 

10 Func is operator.add by default. 

11 

12 >>> list(accumulate([1,2,3,4,5])) 

13 [1, 3, 6, 10, 15] 

14 >>> list(accumulate([1,2,3,4,5], operator.mul)) 

15 [1, 2, 6, 24, 120] 

16 

17 :param iterable: Iterable 

18 :param func: method (default=operator.add) to call for each pair of (last call result or first item, next item) 

19 :return generator: Generator 

20 """ 

21 it = iter(iterable) 

22 try: 

23 total = next(it) 

24 except StopIteration: 

25 return 

26 yield total 

27 for element in it: 

28 total = func(total, element) 

29 yield total

32def consume(iterator, n=None): 

33 """ 

34 Efficiently advance an iterator n-steps ahead. If n is none, consume entirely. 

35 Consumes at C level (and therefore speed) in cpython. 

36 """ 

37 if n is None: 

38 # feed the entire iterator into a zero-length deque 

39 collections.deque(iterator, maxlen=0) 

40 else: 

41 # advance to the empty slice starting at position n 

42 next(itertools.islice(iterator, n, n), None)

45def dedupe_iter(iterator, hashfunc=hash): 

46 """" 

47 Deduplicates an iterator iteratively using hashed values in a set. 

48 Not exactly memory efficient because of that of course. 

49 If you have a large dataset with high cardinality look at HyperLogLog instead. 

50 

51 :return generator: Iterator of deduplicated results. 

52 """ 

53 done = set() 

54 for item in iterator: 

55 hashed = hashfunc(item) 

56 

57 if hashed in done: 

58 continue 

59 

60 done.add(hashed) 

61 yield item
