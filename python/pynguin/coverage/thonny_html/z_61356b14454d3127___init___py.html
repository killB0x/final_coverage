107 # First download devices

123 def get_download_url_and_size(self, board_id): 

124 # TODO: should take vid/pid also into account. It looks like different models may have same board_id 

125 if self._release_info is None or self._devices_info is None: 

126 return None 

127 

128 if not "tag_name" in self._release_info: 

129 raise RuntimeError("Could not find tag_name from %s" % self._get_release_info_url()) 

130 

131 release = self._release_info["tag_name"] 

132 

133 if not self._devices_info.get(board_id, {}).get("FIRMWARE_DOWNLOAD", None): 

134 raise RuntimeError( 

135 "Could not find your board (%s) or its download url from %s (consider making a PR). " 

136 % (board_id, self._get_devices_info_url()) 

137 + "Please find the firmware from https://circuitpython.org/ and install it manually." 

138 ) 

139 

140 url = self._devices_info[board_id]["FIRMWARE_DOWNLOAD"].format( 

141 lang="en_US", release=release 

142 ) 

143 

144 # reporting approximate size for now. Downloader can take precise value from the header later 

145 size = 2 ** 20 # 1 MiB 

146 return (url, size)
